58 void ofCamera::setupPerspective(bool _vFlip, float fov, float nearDist, float farDist, const ofVec2f & lensOffset){
59  ofRectangle orientedViewport = ofGetNativeViewport();
60  float eyeX = orientedViewport.width / 2;
61  float eyeY = orientedViewport.height / 2;
62  float halfFov = PI * fov / 360;
63  float theTan = tanf(halfFov);
64  float dist = eyeY / theTan;
65 
66  if(nearDist == 0) nearDist = dist / 10.0f;
67  if(farDist == 0) farDist = dist * 10.0f;
68 
69  setFov(fov);
70  setNearClip(nearDist);
71  setFarClip(farDist);
72  setLensOffset(lensOffset);
73  setForceAspectRatio(false);
74 
75  setPosition(eyeX,eyeY,dist);
76  lookAt(ofVec3f(eyeX,eyeY,0),ofVec3f(0,1,0));
77  vFlip = _vFlip;
78 }

81 void ofCamera::setupOffAxisViewPortal(const ofVec3f & topLeft, const ofVec3f & bottomLeft, const ofVec3f & bottomRight){
82  ofVec3f bottomEdge = bottomRight - bottomLeft; // plane x axis
83  ofVec3f leftEdge = topLeft - bottomLeft; // plane y axis
84  ofVec3f bottomEdgeNorm = bottomEdge.normalized();
85  ofVec3f leftEdgeNorm = leftEdge.normalized();
86  ofVec3f bottomLeftToCam = this->getPosition() - bottomLeft;
87 
88  ofVec3f cameraLookVector = leftEdgeNorm.getCrossed(bottomEdgeNorm);
89 
90  ofVec3f cameraUpVector = bottomEdgeNorm.getCrossed(cameraLookVector);
91 
92  lookAt(cameraLookVector + this->getPosition(), cameraUpVector);
93 
94  //lensoffset
95  ofVec2f lensOffset;
96  lensOffset.x = -bottomLeftToCam.dot(bottomEdgeNorm) * 2.0f / bottomEdge.length() + 1.0f;
97  lensOffset.y = -bottomLeftToCam.dot(leftEdgeNorm) * 2.0f / leftEdge.length() + 1.0f;
98  setLensOffset(lensOffset);
99  setAspectRatio( bottomEdge.length() / leftEdge.length() );
100  float distanceAlongOpticalAxis = abs(bottomLeftToCam.dot(cameraLookVector));
101  setFov(2.0f * RAD_TO_DEG * atan( (leftEdge.length() / 2.0f) / distanceAlongOpticalAxis));
102 }

164 ofMatrix4x4 ofCamera::getProjectionMatrix(ofRectangle viewport) const {
165  if(isOrtho) {
166  return ofMatrix4x4::newOrthoMatrix(0, viewport.width, 0, viewport.height, nearClip, farClip);
167  }else{
168  float aspect = forceAspectRatio ? aspectRatio : viewport.width/viewport.height;
169  ofMatrix4x4 matProjection;
170  matProjection.makePerspectiveMatrix(fov, aspect, nearClip, farClip);
171  matProjection.translate(-lensOffset.x, -lensOffset.y, 0);

173  }

203 
204  //convert from screen to camera
