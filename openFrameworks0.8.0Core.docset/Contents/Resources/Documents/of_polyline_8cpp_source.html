641  if (k <= j+1) // there is nothing to simplify
642  return;
643 
644  // check for adequate approximation by segment S from v[j] to v[k]
645  int maxi = j; // index of vertex farthest from S
646  float maxd2 = 0; // distance squared of farthest vertex
647  float tol2 = tol * tol; // tolerance squared
648  Segment S = {v[j], v[k]}; // segment from v[j] to v[k]
649  ofPoint u;
650  u = S.P1 - S.P0; // segment direction vector
651  double cu = dot(u,u); // segment length squared
652

654  // compute using the Feb 2001 Algorithm's dist_ofPoint_to_Segment()
655  // Note: this works in any dimension (2D, 3D, ...)
656  ofPoint w;
657  ofPoint Pb; // base of perpendicular from v[i] to S
658  float b, cw, dv2; // dv2 = distance v[i] to S squared
659 
660  for (int i=j+1; i<k; i++){
661  // compute distance squared
662  w = v[i] - S.P0;
663  cw = dot(w,u);
664  if ( cw <= 0 ) dv2 = d2(v[i], S.P0);
665  else if ( cu <= cw ) dv2 = d2(v[i], S.P1);
666  else {

668  Pb = S.P0 + u*b;
669  dv2 = d2(v[i], Pb);
670  }
671  // test with current max distance squared
672  if (dv2 <= maxd2) continue;
673 
674  // v[i] is a new max vertex
675  maxi = i;
676  maxd2 = dv2;
677  }
678  if (maxd2 > tol2) // error is worse than the tolerance
679  {
680  // split the polyline at the farthest vertex from S
681  mk[maxi] = 1; // mark v[maxi] for the simplified polyline
682  // recursively simplify the two subpolylines at v[maxi]
683  simplifyDP( tol, v, j, maxi, mk ); // polyline v[j] to v[maxi]
684  simplifyDP( tol, v, maxi, k, mk ); // polyline v[maxi] to v[k]
685  }
686  // else the approximation is OK, so ignore intermediate vertices
687  return;
688 }

691 void ofPolyline::simplify(float tol){
692  if(points.size() < 2) return;
693 
694  int n = size();
695 
696  if(n == 0) {
697  return;
698  }
699 
700  vector <ofPoint> sV;
701  sV.resize(n);
702 
703  int i, k, m, pv; // misc counters
704  float tol2 = tol * tol; // tolerance squared
705  vector<ofPoint> vt;
706  vector<int> mk;
707  vt.resize(n);
708  mk.resize(n,0);
709 
710 
711  // STAGE 1. Vertex Reduction within tolerance of prior vertex cluster
712  vt[0] = points[0]; // start at the beginning
713  for (i=k=1, pv=0; i<n; i++) {
714  if (d2(points[i], points[pv]) < tol2) continue;
715 
716  vt[k++] = points[i];
717  pv = i;
718  }
719  if (pv < n-1) vt[k++] = points[n-1]; // finish at the end
720 
721  // STAGE 2. Douglas-Peucker polyline simplification
722  mk[0] = mk[k-1] = 1; // mark the first and last vertices
723  simplifyDP( tol, &vt[0], 0, k-1, &mk[0] );
724 
725  // copy marked vertices to the output simplified polyline
726  for (i=m=0; i<k; i++) {
727  if (mk[i]) sV[m++] = vt[i];
728  }
729 
730  //get rid of the unused points
731  if( m < (int)sV.size() ){
732  points.assign( sV.begin(),sV.begin()+m );
733  }else{
734  points = sV;
735  }
736 
737 }

757 float ofPolyline::getIndexAtLength(float length) const {
758  if(points.size() < 2) return 0;
759  updateCache();
760 
761  float totalLength = getPerimeter();
762  length = ofClamp(length, 0, totalLength);
763 
764  int lastPointIndex = isClosed() ? points.size() : points.size()-1;
765 
766  int i1 = ofClamp(floor(length / totalLength * lastPointIndex), 0, lengths.size()-2); // start approximation here
767  int leftLimit = 0;
768  int rightLimit = lastPointIndex;
769 
770  float distAt1, distAt2;
771  for(int iterations = 0; iterations < 32; iterations ++) { // limit iterations
772  i1 = ofClamp(i1, 0, lengths.size()-1);
773  distAt1 = lengths[i1];
774  if(distAt1 <= length) { // if Length at i1 is less than desired Length (this is good)
775  distAt2 = lengths[i1+1];
776  if(distAt2 > length) {
777  float t = ofMap(length, distAt1, distAt2, 0, 1);
778  return i1 + t;
779  } else {
780  leftLimit = i1;
781  }
782  } else {
783  rightLimit = i1;
784  }
785  i1 = (leftLimit + rightLimit)/2;
786  }
787  return 0;

949 void ofPolyline::updateCache(bool bForceUpdate) const {
950  if(bCacheIsDirty || bForceUpdate) {
951  lengths.clear();
952  angles.clear();
953  rotations.clear();
954  normals.clear();
955  tangents.clear();
956  area = 0;
957  centroid2D.set(0, 0, 0);
958  bCacheIsDirty = false;
959 
960  if(points.size() < 2) return;
961 
962  // area
963  for(int i=0;i<(int)points.size()-1;i++){
964  area += points[i].x * points[i+1].y - points[i+1].x * points[i].y;
965  }
966  area += points[points.size()-1].x * points[0].y - points[0].x * points[points.size()-1].y;
967  area *= 0.5;
968 
969 
970  // centroid
971  // TODO: doesn't seem to work on all concave shapes
972  for(int i=0;i<(int)points.size()-1;i++){
973  centroid2D.x += (points[i].x + points[i+1].x) * (points[i].x*points[i+1].y - points[i+1].x*points[i].y);
974  centroid2D.y += (points[i].y + points[i+1].y) * (points[i].x*points[i+1].y - points[i+1].x*points[i].y);
975  }
976  centroid2D.x += (points[points.size()-1].x + points[0].x) * (points[points.size()-1].x*points[0].y - points[0].x*points[points.size()-1].y);
977  centroid2D.y += (points[points.size()-1].y + points[0].y) * (points[points.size()-1].x*points[0].y - points[0].x*points[points.size()-1].y);
978 
979  centroid2D.x /= (6*area);
980  centroid2D.y /= (6*area);
981 
982 
983  // per vertex cache
984  lengths.resize(points.size());
985  tangents.resize(points.size());
986  angles.resize(points.size());
987  normals.resize(points.size());
988  rotations.resize(points.size());
989 
990  float angle;
991  ofVec3f rotation;
992  ofVec3f normal;
993  ofVec3f tangent;
994 
995  float length = 0;
996  for(int i=0; i<(int)points.size(); i++) {
997  lengths[i] = length;
998 
999  calcData(i, tangent, angle, rotation, normal);
1000  tangents[i] = tangent;
1001  angles[i] = angle;
1002  rotations[i] = rotation;
1003  normals[i] = normal;
1004 
1005  length += points[i].distance(points[getWrappedIndex(i + 1)]);
1006  }
1007 
1008  if(isClosed()) lengths.push_back(length);
1009  }
