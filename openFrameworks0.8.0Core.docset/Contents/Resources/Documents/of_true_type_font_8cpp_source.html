647  areaSum += (cps[i].width+border*2)*(cps[i].height+border*2);
648  }
649 
650  vector<charProps> sortedCopy = cps;
651  sort(sortedCopy.begin(),sortedCopy.end(),&compare_cps);
652 
653  // pack in a texture, algorithm to calculate min w/h from
654  // http://upcommons.upc.edu/pfc/bitstream/2099.1/7720/1/TesiMasterJonas.pdf
655  //ofLogNotice("ofTrueTypeFont") << "loadFont(): areaSum: " << areaSum
656 
657  bool packed = false;
658  float alpha = logf(areaSum)*1.44269;
659 
660  int w;
661  int h;
662  while(!packed){
663  w = pow(2,floor((alpha/2.f) + 0.5)); // there doesn't seem to be a round in cmath for windows.
664  //w = pow(2,round(alpha/2.f));
665  h = w;//pow(2,round(alpha - round(alpha/2.f)));
666  int x=0;
667  int y=0;
668  int maxRowHeight = sortedCopy[0].tH + border*2;
669  for(int i=0;i<(int)cps.size();i++){
670  if(x+sortedCopy[i].tW + border*2>w){
671  x = 0;
672  y += maxRowHeight;
673  maxRowHeight = sortedCopy[i].tH + border*2;
674  if(y + maxRowHeight > h){
675  alpha++;
676  break;
677  }
678  }
679  x+= sortedCopy[i].tW + border*2;
680  if(i==(int)cps.size()-1) packed = true;
681  }
682 
683  }
684 
685 
686 
687  ofPixels atlasPixelsLuminanceAlpha;
688  atlasPixelsLuminanceAlpha.allocate(w,h,2);
689  atlasPixelsLuminanceAlpha.set(0,255);
690  atlasPixelsLuminanceAlpha.set(1,0);
691 
692 
693  int x=0;
694  int y=0;
695  int maxRowHeight = sortedCopy[0].tH + border*2;
696  for(int i=0;i<(int)cps.size();i++){
697  ofPixels & charPixels = expanded_data[sortedCopy[i].character];
698 
699  if(x+sortedCopy[i].tW + border*2>w){
700  x = 0;
701  y += maxRowHeight;
702  maxRowHeight = sortedCopy[i].tH + border*2;
703  }
704 
705  cps[sortedCopy[i].character].t2 = float(x + border)/float(w);
706  cps[sortedCopy[i].character].v2 = float(y + border)/float(h);
707  cps[sortedCopy[i].character].t1 = float(cps[sortedCopy[i].character].tW + x + border)/float(w);
708  cps[sortedCopy[i].character].v1 = float(cps[sortedCopy[i].character].tH + y + border)/float(h);
709  charPixels.pasteInto(atlasPixelsLuminanceAlpha,x+border,y+border);
710  x+= sortedCopy[i].tW + border*2;
711  }
712 
713  ofPixels atlasPixels;
714  atlasPixels.allocate(atlasPixelsLuminanceAlpha.getWidth(),atlasPixelsLuminanceAlpha.getHeight(),4);
715  atlasPixels.setChannel(0,atlasPixelsLuminanceAlpha.getChannel(0));
716  atlasPixels.setChannel(1,atlasPixelsLuminanceAlpha.getChannel(0));
717  atlasPixels.setChannel(2,atlasPixelsLuminanceAlpha.getChannel(0));
718  atlasPixels.setChannel(3,atlasPixelsLuminanceAlpha.getChannel(1));
719  texAtlas.allocate(atlasPixels,false);
720 
721  if(bAntiAliased && fontSize>20){
722  texAtlas.setTextureMinMagFilter(GL_LINEAR,GL_LINEAR);
723  }else{
724  texAtlas.setTextureMinMagFilter(GL_NEAREST,GL_NEAREST);
725  }
726  texAtlas.loadData(atlasPixels);
727 
728  // ------------- close the library and typeface
729  FT_Done_Face(face);
730  bLoadedOk = true;
731  return true;

973  xoffset += cps[cy].setWidth * letterSpacing * spaceSize;

979  GLint lextent = cps[cy].leftExtent;

986  y2 = (y + yoffset + -top + corr);

989  minx = x2;

992  maxy = y1;

994  } else {

996  if (y2 < miny) miny = y2;
997  if (x1 > maxx) maxx = x1;
998  if (y1 > maxy) maxy = y1;
999  }
1000  }
1001  }
1002  index++;
1003  }
1004

1006  myRect.y = miny;
1007  myRect.width = maxx-minx;
