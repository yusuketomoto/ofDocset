421 void ofBackgroundGradient(const ofColor& start, const ofColor& end, ofGradientMode mode) {
422  float w = ofGetWidth(), h = ofGetHeight();
423  gradientMesh.clear();
424  gradientMesh.setMode(OF_PRIMITIVE_TRIANGLE_FAN);
425 #ifdef TARGET_OPENGLES
426  if(ofIsGLProgrammableRenderer()) gradientMesh.setUsage(GL_STREAM_DRAW);
427 #else
428  gradientMesh.setUsage(GL_STREAM_DRAW);
429 #endif
430  if(mode == OF_GRADIENT_CIRCULAR) {
431  // this could be optimized by building a single mesh once, then copying
432  // it and just adding the colors whenever the function is called.
433  ofVec2f center(w / 2, h / 2);
434  gradientMesh.addVertex(center);
435  gradientMesh.addColor(start);
436  int n = 32; // circular gradient resolution
437  float angleBisector = TWO_PI / (n * 2);
438  float smallRadius = ofDist(0, 0, w / 2, h / 2);
439  float bigRadius = smallRadius / cos(angleBisector);
440  for(int i = 0; i <= n; i++) {
441  float theta = i * TWO_PI / n;
442  gradientMesh.addVertex(center + ofVec2f(sin(theta), cos(theta)) * bigRadius);
443  gradientMesh.addColor(end);
444  }
445  } else if(mode == OF_GRADIENT_LINEAR) {
446  gradientMesh.addVertex(ofVec2f(0, 0));
447  gradientMesh.addVertex(ofVec2f(w, 0));
448  gradientMesh.addVertex(ofVec2f(w, h));
449  gradientMesh.addVertex(ofVec2f(0, h));
450  gradientMesh.addColor(start);
451  gradientMesh.addColor(start);
452  gradientMesh.addColor(end);
453  gradientMesh.addColor(end);
454  } else if(mode == OF_GRADIENT_BAR) {
455  gradientMesh.addVertex(ofVec2f(w / 2, h / 2));
456  gradientMesh.addVertex(ofVec2f(0, h / 2));
457  gradientMesh.addVertex(ofVec2f(0, 0));
458  gradientMesh.addVertex(ofVec2f(w, 0));
459  gradientMesh.addVertex(ofVec2f(w, h / 2));
460  gradientMesh.addVertex(ofVec2f(w, h));
461  gradientMesh.addVertex(ofVec2f(0, h));
462  gradientMesh.addVertex(ofVec2f(0, h / 2));
463  gradientMesh.addColor(start);
464  gradientMesh.addColor(start);
465  gradientMesh.addColor(end);
466  gradientMesh.addColor(end);
467  gradientMesh.addColor(start);
468  gradientMesh.addColor(end);
469  gradientMesh.addColor(end);
470  gradientMesh.addColor(start);
471  }
472  GLboolean depthMaskEnabled;
473  glGetBooleanv(GL_DEPTH_WRITEMASK,&depthMaskEnabled);
474  glDepthMask(GL_FALSE);
475  gradientMesh.draw();
476  if(depthMaskEnabled){
477  glDepthMask(GL_TRUE);
478  }
479 }
