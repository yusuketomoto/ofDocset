339 void ofPath::rectRounded(float x, float y, float z, float w, float h, float topLeftRadius,
340  float topRightRadius,
341  float bottomRightRadius,
342  float bottomLeftRadius){
343  // since we support w / h < 0, canonicalize the rectangle for easier drawing
344  if(w < 0.0f) {
345  x += w;
346  w *= -1.0f;
347  }
348 
349  if(h < 0.0f) {
350  y += h;
351  h *= -1.0f;
352  }
353 
354  // keep radii in check
355  float maxRadius = MIN(w / 2.0f, h / 2.0f);
356  topLeftRadius = MIN(topLeftRadius, maxRadius);
357  topRightRadius = MIN(topRightRadius, maxRadius);
358  bottomRightRadius = MIN(bottomRightRadius, maxRadius);
359  bottomLeftRadius = MIN(bottomLeftRadius, maxRadius);
360 
361  // if all radii are ~= 0.0f, then render as a normal rectangle
362  if((fabs(topLeftRadius) < FLT_EPSILON) &&
363  (fabs(topRightRadius) < FLT_EPSILON) &&
364  (fabs(bottomRightRadius) < FLT_EPSILON) &&
365  (fabs(bottomLeftRadius) < FLT_EPSILON)) {
366 
367  // rect mode respect happens in ofRect
368  rectangle(x, y, z, w, h);
369  } else {
370  float left = x;
371  float right = x + w;
372  float top = y;
373  float bottom = y + h;
374 
375 
376  moveTo(left + topLeftRadius, top, z);
377 
378  // top right
379  if(fabs(topRightRadius) >= FLT_EPSILON) {
380  arc(right - topRightRadius, top + topRightRadius, z, topRightRadius, topRightRadius, 270, 360);
381  } else {

383  }
384

386  // bottom right

388  arc(right - bottomRightRadius, bottom - bottomRightRadius, z, bottomRightRadius, bottomRightRadius, 0, 90);
389  }

395  arc(left + bottomLeftRadius, bottom - bottomLeftRadius, z, bottomLeftRadius, bottomLeftRadius, 90, 180);

492 void ofPath::generatePolylinesFromCommands(){
493  if(mode==POLYLINES || commands.empty()) return;
494  if(bNeedsPolylinesGeneration || curveResolution!=prevCurveRes){
495  prevCurveRes = curveResolution;
496 
497  polylines.clear();
498  int j=-1;
499 
500  for(int i=0; i<(int)commands.size();i++){
501  switch(commands[i].type){
502  case Command::moveTo:
503  polylines.push_back(ofPolyline());
504  j++;
505  polylines[j].addVertex(commands[i].to);
506  break;
507  case Command::lineTo:
508  polylines[j].addVertex(commands[i].to);
509  break;

511  polylines[j].curveTo(commands[i].to, curveResolution);
512  break;

514  polylines[j].bezierTo(commands[i].cp1,commands[i].cp2,commands[i].to, curveResolution);
515  break;

517  polylines[j].quadBezierTo(commands[i].cp1,commands[i].cp2,commands[i].to, curveResolution);
518  break;
519  case Command::arc:
520  polylines[j].arc(commands[i].to,commands[i].radiusX,commands[i].radiusY,commands[i].angleBegin,commands[i].angleEnd, circleResolution);
521  break;
522  case Command::arcNegative:
523  polylines[j].arcNegative(commands[i].to,commands[i].radiusX,commands[i].radiusY,commands[i].angleBegin,commands[i].angleEnd, circleResolution);
524  break;
525  case Command::close:
526  polylines[j].setClosed(true);
527  break;
528  }
529  }

531  bNeedsPolylinesGeneration = false;
532  bNeedsTessellation = true;
