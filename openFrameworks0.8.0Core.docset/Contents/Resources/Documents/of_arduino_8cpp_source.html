102  // analog pins used as digital

104  _digitalPinValue[i] = -1;
105  }
106 
107  for (int i=0; i<_totalDigitalPins; ++i) {

211 
212  _digitalPinValue[pin] = value;
213

215  int bit=0;

218 
219  // support Firmata 2.3/Arduino 1.0 with backwards compatibility
220  // to previous protocol versions
221  if (_firmwareVersionSum >= FIRMWARE2_3) {
222  port1Offset = 16;

232  }

393 void ofArduino::processData(unsigned char inputData){
394 
395  char msg[100];
396  sprintf(msg, "Received Byte: %i", inputData);
397  //Logger::get("Application").information(msg);
398 
399  // we have command data
400  if(_waitForData>0 && inputData<128) {
401  _waitForData--;
402 
403  // collect the data
404  _storedInputData[_waitForData] = inputData;
405 
406  // we have all data executeMultiByteCommand
407  if(_waitForData==0) {
408  switch (_executeMultiByteCommand) {
409  case FIRMATA_DIGITAL_MESSAGE:
410  processDigitalPort(_multiByteChannel, (_storedInputData[0] << 7) | _storedInputData[1]);
411  break;
412  case FIRMATA_REPORT_VERSION: // report version
413  _majorProtocolVersion = _storedInputData[1];
414  _minorProtocolVersion = _storedInputData[0];
415  ofNotifyEvent(EProtocolVersionReceived, _majorProtocolVersion, this);
416  break;
417  case FIRMATA_ANALOG_MESSAGE:
418  if(_analogHistory[_multiByteChannel].size()>0){
419  int previous = _analogHistory[_multiByteChannel].front();
420 
421  _analogHistory[_multiByteChannel].push_front((_storedInputData[0] << 7) | _storedInputData[1]);
422  if((int)_analogHistory[_multiByteChannel].size()>_analogHistoryLength)
423  _analogHistory[_multiByteChannel].pop_back();
424 
425  // trigger an event if the pin has changed value
426  if(_analogHistory[_multiByteChannel].front()!=previous)
427  ofNotifyEvent(EAnalogPinChanged, _multiByteChannel, this);
428  }else{
429  _analogHistory[_multiByteChannel].push_front((_storedInputData[0] << 7) | _storedInputData[1]);
430  if((int)_analogHistory[_multiByteChannel].size()>_analogHistoryLength)
431  _analogHistory[_multiByteChannel].pop_back();
432  }
433  break;
434  }
435 
436  }
437  }
438  // we have SysEx command data
439  else if(_waitForData<0){
440 
441  // we have all sysex data
442  if(inputData==FIRMATA_END_SYSEX){
443  _waitForData=0;
444  processSysExData(_sysExData);
445  _sysExData.clear();
446  }
447  // still have data, collect it
448  else {
449  _sysExData.push_back((unsigned char)inputData);
450  }
451  }
452  // we have a command
453  else{
454 
455  int command;
456 
457  // extract the command and channel info from a byte if it is less than 0xF0
458  if(inputData < 0xF0) {
459  command = inputData & 0xF0;
460  _multiByteChannel = inputData & 0x0F;
461  }
462  else {
463  // commands in the 0xF* range don't use channel data
464  command = inputData;
465  }
466 
467  switch (command) {
468  case FIRMATA_REPORT_VERSION:
469  case FIRMATA_DIGITAL_MESSAGE:
470  case FIRMATA_ANALOG_MESSAGE:
471  _waitForData = 2; // 2 bytes needed
472  _executeMultiByteCommand = command;
473  break;
474  case FIRMATA_START_SYSEX:
475  _sysExData.clear();
476  _waitForData = -1; // n bytes needed, -1 is used to indicate sysex message
477  _executeMultiByteCommand = command;
478  break;
479  }
480 
481  }
482 }

485 void ofArduino::processSysExData(vector<unsigned char> data){
486 
487  string str;
488 
489  vector<unsigned char>::iterator it;
490  unsigned char buffer;
491  //int i = 1;
492 
493  // act on reserved sysEx messages (extended commands) or trigger SysEx event...
494  switch(data.front()) { //first byte in buffer is command
495  case FIRMATA_SYSEX_REPORT_FIRMWARE:
496  it = data.begin();
497  it++; // skip the first byte, which is the firmware version command
498  _majorFirmwareVersion = *it;
499  it++;
500  _minorFirmwareVersion = *it;
501  it++;
502 
503  while( it != data.end() ) {
504  buffer = *it;
505  it++;
506  buffer += *it << 7;
507  it++;
508  str+=buffer;
509  }
510  _firmwareName = str;
511 
512  _firmwareVersionSum = _majorFirmwareVersion * 10 + _minorFirmwareVersion;
513  ofNotifyEvent(EFirmwareVersionReceived, _majorFirmwareVersion, this);
514 
515  // trigger the initialization event
516  if (!_initialized) {
517  initPins();
518  ofNotifyEvent(EInitialized, _majorFirmwareVersion, this);
519 
520  }
521 
522  break;
523  case FIRMATA_SYSEX_FIRMATA_STRING:
524  it = data.begin();
525  it++; // skip the first byte, which is the string command
526  while( it != data.end() ) {
527  buffer = *it;
528  it++;
529  buffer += *it << 7;
530  it++;
531  str+=buffer;
532  }
533 
534  _stringHistory.push_front(str);
535  if((int)_stringHistory.size()>_stringHistoryLength)
536  _stringHistory.pop_back();
537 
538  ofNotifyEvent(EStringReceived, str, this);
539  break;
540  default: // the message isn't in Firmatas extended command set
541  _sysExHistory.push_front(data);
542  if((int)_sysExHistory.size()>_sysExHistoryLength)
543  _sysExHistory.pop_back();
544  ofNotifyEvent(ESysExReceived, data, this);
545  break;
546 
547  }

553  int previous;
554  int i;
555  int pin;
556  int port1Pins;
557  int port2Pins;
558

561  port1Pins = 8;
562  port2Pins = 4;
563  } else {
564  port1Pins = 6;
565  port2Pins = 6;
566  }
567

569  case 0: // pins 2-7 (0,1 are ignored as serial RX/TX)

571  pin = i;
572  previous = -1;
573  if(_digitalPinMode[pin]==ARD_INPUT){
