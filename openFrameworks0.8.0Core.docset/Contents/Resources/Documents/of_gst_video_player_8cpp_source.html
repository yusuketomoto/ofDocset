175 bool ofGstVideoPlayer::allocate(int bpp){
176  if(bIsAllocated) return true;
177 
178  guint64 durationNanos = videoUtils.getDurationNanos();
179 
180  nFrames = 0;
181  if(GstPad* pad = gst_element_get_static_pad(videoUtils.getSink(), "sink")){
182 #if GST_VERSION_MAJOR==0
183  int width,height;
184  if(gst_video_get_size(GST_PAD(pad), &width, &height)){
185  if(!videoUtils.allocate(width,height,bpp)) return false;
186  }else{
187  ofLogError("ofGstVideoPlayer") << "allocate(): couldn't query width and height";
188  return false;
189  }
190 
191  const GValue *framerate = gst_video_frame_rate(pad);
192  fps_n=0;
193  fps_d=0;
194  if(framerate && GST_VALUE_HOLDS_FRACTION (framerate)){
195  fps_n = gst_value_get_fraction_numerator (framerate);
196  fps_d = gst_value_get_fraction_denominator (framerate);
197  nFrames = (float)(durationNanos / GST_SECOND) * (float)fps_n/(float)fps_d;
198  ofLogVerbose("ofGstVideoPlayer") << "allocate(): framerate: " << fps_n << "/" << fps_d;
199  }else{
200  ofLogWarning("ofGstVideoPlayer") << "allocate(): cannot get framerate, frame seek won't work";
201  }
202  bIsAllocated = true;
203 #else
204  if(GstCaps *caps = gst_pad_get_current_caps (GST_PAD (pad))){
205  GstVideoInfo info;
206  gst_video_info_init (&info);
207  if (gst_video_info_from_caps (&info, caps)){
208  if(!videoUtils.allocate(info.width,info.height,bpp)) return false;
209  }else{
210  ofLogError("ofGstVideoPlayer") << "allocate(): couldn't query width and height";
211  return false;
212  }
213 
214  fps_n = info.fps_n;
215  fps_d = info.fps_d;
216  nFrames = (float)(durationNanos / GST_SECOND) * (float)fps_n/(float)fps_d;
217  gst_caps_unref(caps);
218  bIsAllocated = true;
219  }else{
220  ofLogError("ofGstVideoPlayer") << "allocate(): cannot get pipeline caps";
221  bIsAllocated = false;
222  }
223 #endif
224  gst_object_unref(GST_OBJECT(pad));
225  }else{
226  ofLogError("ofGstVideoPlayer") << "allocate(): cannot get sink pad";
227  bIsAllocated = false;
228  }
