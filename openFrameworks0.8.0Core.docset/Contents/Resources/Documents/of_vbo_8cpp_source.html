632 void ofVbo::bind(){
633  if(supportVAOs){
634  if(vaoID==0){
635  glGenVertexArrays(1, &vaoID);
636  if(vaoID!=0){
637  retainVAO(vaoID);
638  }else{
639  supportVAOs = false;
640  ofLogVerbose("ofVbo") << "bind(): error allocating VAO, disabling VAO support";
641  }
642  }
643 
644  glBindVertexArray(vaoID);
645  }
646 
647  if(vaoChanged || !supportVAOs){
648  bool programmable = ofIsGLProgrammableRenderer();
649  if(bUsingVerts){
650  glBindBuffer(GL_ARRAY_BUFFER, vertId);
651  if(!programmable){
652  glEnableClientState(GL_VERTEX_ARRAY);
653  glVertexPointer(vertSize, GL_FLOAT, vertStride, 0);
654  }else{
655  glEnableVertexAttribArray(ofShader::POSITION_ATTRIBUTE);
656  glVertexAttribPointer(ofShader::POSITION_ATTRIBUTE, 3, GL_FLOAT, GL_FALSE, vertStride, 0);
657  }
658  }else if(supportVAOs){
659  if(!programmable){
660  glDisableClientState(GL_VERTEX_ARRAY);
661  }else{
662  glDisableVertexAttribArray(ofShader::POSITION_ATTRIBUTE);
663  }
664  }
665 
666  if(bUsingColors) {
667  glBindBuffer(GL_ARRAY_BUFFER, colorId);
668  if(!programmable){
669  glEnableClientState(GL_COLOR_ARRAY);
670  glColorPointer(4, GL_FLOAT, colorStride, 0);
671  }else{
672  glEnableVertexAttribArray(ofShader::COLOR_ATTRIBUTE);
673  glVertexAttribPointer(ofShader::COLOR_ATTRIBUTE, 4, GL_FLOAT, GL_FALSE, colorStride, 0);
674  }
675  }else if(supportVAOs){
676  if(!programmable){
677  glDisableClientState(GL_COLOR_ARRAY);
678  }else{
679  glDisableVertexAttribArray(ofShader::COLOR_ATTRIBUTE);
680  }
681  }
682 
683  if(bUsingNormals) {
684  glBindBuffer(GL_ARRAY_BUFFER, normalId);
685  if(!programmable){
686  glEnableClientState(GL_NORMAL_ARRAY);
687  glNormalPointer(GL_FLOAT, normalStride, 0);
688  }else{
689  // tig: note that we set the 'Normalize' flag to true here, assuming that mesh normals need to be
690  // normalized while being uploaded to GPU memory.
691  // http://www.opengl.org/sdk/docs/man/xhtml/glVertexAttribPointer.xml
692  // Normalizing the normals on the shader is probably faster, but sending non-normalized normals is
693  // more prone to lead to artifacts difficult to diagnose, especially with the built-in 3D primitives.
694  // If you need to optimise this, and you've dug this far through the code, you are most probably
695  // able to roll your own client code for binding & rendering vbos anyway...
696  glEnableVertexAttribArray(ofShader::NORMAL_ATTRIBUTE);
697  glVertexAttribPointer(ofShader::NORMAL_ATTRIBUTE, 3, GL_FLOAT, GL_TRUE, normalStride, 0);
698  }
699  }else if(supportVAOs){
700  if(!programmable){
701  glDisableClientState(GL_NORMAL_ARRAY);
702  }else{
703  glDisableVertexAttribArray(ofShader::NORMAL_ATTRIBUTE);
704  }
705  }
706 
707  if(bUsingTexCoords) {
708  glBindBuffer(GL_ARRAY_BUFFER, texCoordId);
709  if(!programmable){
710  glEnableClientState(GL_TEXTURE_COORD_ARRAY);
711  glTexCoordPointer(2, GL_FLOAT, texCoordStride, 0);
712  }else{
713  glEnableVertexAttribArray(ofShader::TEXCOORD_ATTRIBUTE);
714  glVertexAttribPointer(ofShader::TEXCOORD_ATTRIBUTE, 2, GL_FLOAT, GL_FALSE, texCoordStride, 0);
715  }
716  }else if(supportVAOs){
717  if(!programmable){
718  glDisableClientState(GL_TEXTURE_COORD_ARRAY);
719  }else{
720  glDisableVertexAttribArray(ofShader::TEXCOORD_ATTRIBUTE);
721  }
722  }
723 
724  map<int,GLuint>::iterator it;
725  for(it=attributeIds.begin();it!=attributeIds.end();it++){
726  glBindBuffer(GL_ARRAY_BUFFER, attributeIds[it->first]);
727  glEnableVertexAttribArray(it->first);
728  glVertexAttribPointer(it->first, attributeNumCoords[it->first], GL_FLOAT, GL_FALSE, attributeStrides[it->first], 0);
729  }
730 
731  vaoChanged=false;
732  }
733 
734 
735  ofPtr<ofGLProgrammableRenderer> renderer = ofGetGLProgrammableRenderer();
736  if(renderer){
737  renderer->setAttributes(bUsingVerts,bUsingColors,bUsingTexCoords,bUsingNormals);
738  }
739  bBound = true;
740 }
