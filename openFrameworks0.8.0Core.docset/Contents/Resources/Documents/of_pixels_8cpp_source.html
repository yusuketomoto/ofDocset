739  a00 + a01 * y + a02 * y2 + a03 * y3 +
740  a10 * x + a11 * x * y + a12 * x * y2 + a13 * x * y3 +
741  a20 * x2 + a21 * x2 * y + a22 * x2 * y2 + a23 * x2 * y3 +
742  a30 * x3 + a31 * x3 * y + a32 * x3 * y2 + a33 * x3 * y3;

748 template<typename PixelType>
749 bool ofPixels_<PixelType>::resizeTo(ofPixels_<PixelType>& dst, ofInterpolationMethod interpMethod){
750  if(&dst == this){
751  return true;
752  }
753 
754  if (!(isAllocated()) || !(dst.isAllocated()) || getBytesPerPixel() != dst.getBytesPerPixel()) return false;
755 
756  int srcWidth = getWidth();
757  int srcHeight = getHeight();
758  int dstWidth = dst.getWidth();
759  int dstHeight = dst.getHeight();
760  int bytesPerPixel = getBytesPerPixel();
761 
762 
763  PixelType * dstPixels = dst.getPixels();
764 
765  switch (interpMethod){
766 
767  //----------------------------------------
768  case OF_INTERPOLATE_NEAREST_NEIGHBOR:{
769  int dstIndex = 0;
770  float srcxFactor = (float)srcWidth/dstWidth;
771  float srcyFactor = (float)srcHeight/dstHeight;
772  float srcy = 0.5;
773  for (int dsty=0; dsty<dstHeight; dsty++){
774  float srcx = 0.5;
775  int srcIndex = int(srcy)*srcWidth;
776  for (int dstx=0; dstx<dstWidth; dstx++){
777  int pixelIndex = int(srcIndex + srcx) * bytesPerPixel;
778  for (int k=0; k<bytesPerPixel; k++){
779  dstPixels[dstIndex] = pixels[pixelIndex];
780  dstIndex++;
781  pixelIndex++;
782  }
783  srcx+=srcxFactor;
784  }
785  srcy+=srcyFactor;
786  }
787  }break;
788 
789  //----------------------------------------
790  case OF_INTERPOLATE_BILINEAR:
791  // not implemented yet
792  ofLogError("ofPixels") << "resizeTo(): bilinear resize not implemented, not resizing";
793  break;
794 
795  //----------------------------------------
796  case OF_INTERPOLATE_BICUBIC:
797  float px1, py1;
798  float px2, py2;
799  float px3, py3;
800 
801  float srcColor = 0;
802  float interpCol;
803  int patchRow;
804  int patchIndex;
805  float patch[16];
806 
807  int srcRowBytes = srcWidth*bytesPerPixel;
808  int loIndex = (srcRowBytes)+1;
809  int hiIndex = (srcWidth*srcHeight*bytesPerPixel)-(srcRowBytes)-1;
810 
811  for (int dsty=0; dsty<dstHeight; dsty++){
812  for (int dstx=0; dstx<dstWidth; dstx++){
813 
814  int dstIndex0 = (dsty*dstWidth + dstx) * bytesPerPixel;
815  float srcxf = srcWidth * (float)dstx/(float)dstWidth;
816  float srcyf = srcHeight * (float)dsty/(float)dstHeight;
817  int srcx = (int) MIN(srcWidth-1, srcxf);
818  int srcy = (int) MIN(srcHeight-1, srcyf);
819  int srcIndex0 = (srcy*srcWidth + srcx) * bytesPerPixel;
820 
821  px1 = srcxf - srcx;
822  py1 = srcyf - srcy;
823  px2 = px1 * px1;
824  px3 = px2 * px1;
825  py2 = py1 * py1;
826  py3 = py2 * py1;
827 
828  for (int k=0; k<bytesPerPixel; k++){
829  int dstIndex = dstIndex0+k;
830  int srcIndex = srcIndex0+k;
831 
832  for (int dy=0; dy<4; dy++) {
833  patchRow = srcIndex + ((dy-1)*srcRowBytes);
834  for (int dx=0; dx<4; dx++) {
835  patchIndex = patchRow + (dx-1)*bytesPerPixel;
836  if ((patchIndex >= loIndex) && (patchIndex < hiIndex)) {
837  srcColor = pixels[patchIndex];
838  }
839  patch[dx*4 + dy] = srcColor;
840  }
841  }
842 
843  interpCol = (PixelType)bicubicInterpolate(patch, px1,py1, px2,py2, px3,py3);
844  dstPixels[dstIndex] = interpCol;
845  }
846 
847  }
848  }
849  break;
850  }
851 
852  return true;
853 }
