369 void ofFbo::allocate(Settings _settings) {
370  if(!checkGLSupport()) return;
371 
372  destroy();
373 
374  if(_settings.width == 0) _settings.width = ofGetWidth();
375  if(_settings.height == 0) _settings.height = ofGetHeight();
376  if(_settings.numSamples > maxSamples() && maxSamples() > -1) {
377  ofLogWarning("ofFbo") << "allocate(): clamping numSamples " << _settings.numSamples << " to maxSamples " << maxSamples() << " for frame buffer object" << fbo;
378  _settings.numSamples = maxSamples();
379  }
380  settings = _settings;
381 
382  // create main fbo
383  // this is the main one we bind for drawing into
384  // all the renderbuffers are attached to this (whether MSAA is enabled or not)
385  glGenFramebuffers(1, &fbo);
386  retainFB(fbo);
387  bind();
388 
389  if(settings.depthStencilAsTexture && settings.numSamples){
390  ofLogWarning("ofFbo") << "allocate(): multisampling not supported with depthStencilAsTexture, setting 0 samples for frame buffer object " << fbo;
391  settings.numSamples = 0;
392  }
393 
394  //currently depth only works if stencil is enabled.
395  // http://forum.openframeworks.cc/index.php/topic,6837.0.html
396 #ifdef TARGET_OPENGLES
397  if(settings.useDepth){
398  settings.useStencil = true;
399  }
400  if( settings.depthStencilAsTexture ){
401  settings.depthStencilAsTexture = false;
402  ofLogWarning("ofFbo") << "allocate(): depthStencilAsTexture is not available for iOS";
403  }
404 #endif
405 
406  GLenum depthAttachment = GL_DEPTH_ATTACHMENT;
407 
408  if( settings.useDepth && settings.useStencil ){
409  settings.depthStencilInternalFormat = GL_DEPTH_STENCIL;
410  #ifdef TARGET_OPENGLES
411  depthAttachment = GL_DEPTH_ATTACHMENT;
412  #else
413  depthAttachment = GL_DEPTH_STENCIL_ATTACHMENT;
414  #endif
415  }else if(settings.useDepth){
416  depthAttachment = GL_DEPTH_ATTACHMENT;
417  }else if(settings.useStencil){
418  depthAttachment = GL_STENCIL_ATTACHMENT;
419  settings.depthStencilInternalFormat = GL_STENCIL_INDEX;
420  }
421 
422  //- USE REGULAR RENDER BUFFER
423  if(!settings.depthStencilAsTexture){
424  if(settings.useDepth && settings.useStencil){
425  stencilBuffer = depthBuffer = createAndAttachRenderbuffer(settings.depthStencilInternalFormat, depthAttachment);
426  retainRB(stencilBuffer);
427  retainRB(depthBuffer);
428  }else if(settings.useDepth){
429  depthBuffer = createAndAttachRenderbuffer(settings.depthStencilInternalFormat, depthAttachment);
430  retainRB(depthBuffer);
431  }else if(settings.useStencil){
432  stencilBuffer = createAndAttachRenderbuffer(settings.depthStencilInternalFormat, depthAttachment);
433  retainRB(stencilBuffer);
434  }
435  //- INSTEAD USE TEXTURE
436  }else{
437  if(settings.useDepth || settings.useStencil){
438  createAndAttachDepthStencilTexture(settings.textureTarget,settings.depthStencilInternalFormat,depthAttachment);
439 #ifdef TARGET_OPENGLES
440  // if there's depth and stencil the texture should be attached as
441  // depth and stencil attachments
442  // http://www.khronos.org/registry/gles/extensions/OES/OES_packed_depth_stencil.txt
443  if(settings.useDepth && settings.useStencil){
444  glFramebufferTexture2D(GL_FRAMEBUFFER,
445  GL_STENCIL_ATTACHMENT,
446  GL_TEXTURE_2D, depthBufferTex.texData.textureID, 0);
447  }
448 #endif
449  }
450  }
451 
452  // if we want MSAA, create a new fbo for textures
453  #ifndef TARGET_OPENGLES
454  if(settings.numSamples){
455  glGenFramebuffers(1, &fboTextures);
456  retainFB(fboTextures);
457  }else{
458  fboTextures = fbo;
459  }
460  #else
461  fboTextures = fbo;
462  if(settings.numSamples){
463  ofLogWarning("ofFbo") << "allocate(): multisampling not supported in OpenGL ES";
464  }
465  #endif
466 
467  // now create all textures and color buffers
468  if(settings.colorFormats.size() > 0) {
469  for(int i=0; i<(int)settings.colorFormats.size(); i++) createAndAttachTexture(settings.colorFormats[i], i);
470  }
471  else if(settings.numColorbuffers > 0) {
472  for(int i=0; i<settings.numColorbuffers; i++) createAndAttachTexture(settings.internalformat, i);
473  }
474  else {
475  ofLogWarning("ofFbo") << "allocate(): no color buffers specified for frame buffer object " << fbo;
476  }
477 
478 
479  // if textures are attached to a different fbo (e.g. if using MSAA) check it's status
480  if(fbo != fboTextures) {
481  glBindFramebuffer(GL_FRAMEBUFFER, fboTextures);
482  }
483 
484  // check everything is ok with this fbo
485  bIsAllocated = checkStatus();
486 
487  // unbind it
488  unbind();

510 void ofFbo::createAndAttachTexture(GLenum internalFormat, GLenum attachmentPoint) {
511  // bind fbo for textures (if using MSAA this is the newly created fbo, otherwise its the same fbo as before)
512  GLint temp;
513  glGetIntegerv(GL_FRAMEBUFFER_BINDING, &temp);
514  glBindFramebuffer(GL_FRAMEBUFFER, fboTextures);
515 
516  ofTexture tex;
517  tex.allocate(settings.width, settings.height, internalFormat, settings.textureTarget == GL_TEXTURE_2D ? false : true);
518  tex.texData.bFlipTexture = false;
519  tex.setTextureWrap(settings.wrapModeHorizontal, settings.wrapModeVertical);
520  tex.setTextureMinMagFilter(settings.minFilter, settings.maxFilter);
521 
522  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0 + attachmentPoint, tex.texData.textureTarget, tex.texData.textureID, 0);
523  textures.push_back(tex);
524 
525  settings.colorFormats.resize(attachmentPoint + 1);
526  settings.colorFormats[attachmentPoint] = internalFormat;
527  settings.numColorbuffers = settings.colorFormats.size();
528 
529 
530  // if MSAA, bind main fbo and attach renderbuffer
531  if(settings.numSamples) {
532  glBindFramebuffer(GL_FRAMEBUFFER, fbo);
533 
534  GLuint colorBuffer = createAndAttachRenderbuffer(internalFormat, GL_COLOR_ATTACHMENT0 + attachmentPoint);
535  colorBuffers.push_back(colorBuffer);
536  retainRB(colorBuffer);
537  }
538  glBindFramebuffer(GL_FRAMEBUFFER, temp);
539 }

733  if(!bIsAllocated) return;
734  // TODO: flag to see if this is dirty or not

736  if(fbo != fboTextures && dirty) {

738 
739  if (!ofIsGLProgrammableRenderer()){
740  // save current drawbuffer
741  glPushAttrib(GL_COLOR_BUFFER_BIT);
742  }
743  // save current readbuffer
744  GLint readBuffer;
745  glGetIntegerv(GL_READ_BUFFER, &readBuffer);
746 
747  glBindFramebuffer(GL_READ_FRAMEBUFFER, fbo);
748  glBindFramebuffer(GL_DRAW_FRAMEBUFFER, fboTextures);
749  glDrawBuffer(GL_COLOR_ATTACHMENT0 + attachmentPoint);
750  glReadBuffer(GL_COLOR_ATTACHMENT0 + attachmentPoint);
751  glBlitFramebuffer(0, 0, settings.width, settings.height, 0, 0, settings.width, settings.height, GL_COLOR_BUFFER_BIT, GL_NEAREST);
752 
753  glBindFramebuffer(GL_READ_FRAMEBUFFER, savedFramebuffer);
754  glBindFramebuffer(GL_DRAW_FRAMEBUFFER, savedFramebuffer);
755  glBindFramebuffer( GL_FRAMEBUFFER, savedFramebuffer );
756 
757  // restore readbuffer
758  glReadBuffer(readBuffer);
759 
760  if(!ofIsGLProgrammableRenderer()){
761  // restore drawbuffer
762  glPopAttrib();
763  }
764 
765  dirty = false;
766 
767  }
