104 static void release(ofLight & light){
105  int id = light.getLightID();
106  if(id==-1) return;
107  bool lastRef=false;
108  if(getIds().find(id)!=getIds().end()){
109  getIds()[id]--;
110  if(getIds()[id]==0){
111  lastRef=true;
112  getIds().erase(id);
113  }
114  }else{
115  ofLogWarning("ofLight") << "release(): something's wrong here, releasing unknown light id " << id;
116  lastRef=true;
117  }
118  if(lastRef){
119  light.setAmbientColor(ofColor(0,0,0,255));
120  if(id>0){
121  light.setDiffuseColor(ofColor(0,0,0,255));
122  light.setSpecularColor(ofColor(0,0,0,255));
123  }else{
124  light.setDiffuseColor(ofColor(255,255,255,255));
125  light.setSpecularColor(ofColor(255,255,255,255));
126  }
127  GLfloat cc[] = {0,0,1, 0};
128  glLightfv(GL_LIGHT0 + id, GL_POSITION, cc);
129 
130  light.disable();
131  getActiveLights()[id] = false;
132  }
133 }

201 void ofLight::setup() {
202  if(glIndex==-1){
203  bool bLightFound = false;
204  // search for the first free block
205  for(int i=0; i<OF_MAX_LIGHTS; i++) {
206  if(getActiveLights()[i] == false) {
207  glIndex = i;
208  retain(glIndex);
209  bLightFound = true;
210  break;
211  }
212  }
213  if( !bLightFound ){
214  ofLogError("ofLight") << "setup(): couldn't get active GL light, maximum number of "<< OF_MAX_LIGHTS << " reached";
215  }
216  if(bLightFound) {
217  // run this the first time, since it was not found before //
218  onPositionChanged();
219  setAmbientColor( getAmbientColor() );
220  setDiffuseColor( getDiffuseColor() );
221  setSpecularColor( getSpecularColor() );
222  setAttenuation( getAttenuationConstant(), getAttenuationLinear(), getAttenuationQuadratic() );
223  if(getIsSpotlight()) {
224  setSpotlightCutOff(getSpotlightCutOff());
225  setSpotConcentration(getSpotConcentration());
226  }
227  if(getIsSpotlight() || getIsDirectional()) {
228  onOrientationChanged();
229  }
230  }
231  }
232 }
