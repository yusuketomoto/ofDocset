723 void ofGstVideoUtils::update(){
724  if (isLoaded()){
725  if(!isFrameByFrame()){
726  mutex.lock();
727  bHavePixelsChanged = bBackPixelsChanged;
728  if (bHavePixelsChanged){
729  bBackPixelsChanged=false;
730  pixels.swap(backPixels);
731 #if GST_VERSION_MAJOR==0
732  if(prevBuffer) gst_buffer_unref (prevBuffer);
733 #else
734  if(prevBuffer) gst_sample_unref (prevBuffer);
735 #endif
736  prevBuffer = buffer;
737  }
738 
739  mutex.unlock();
740  }else{
741 #if GST_VERSION_MAJOR==0
742  GstBuffer *buffer;
743 
744  //get the buffer from appsink
745  if(isPaused()) buffer = gst_app_sink_pull_preroll (GST_APP_SINK (getSink()));
746  else buffer = gst_app_sink_pull_buffer (GST_APP_SINK (getSink()));
747 
748  if(buffer){
749  if(pixels.isAllocated()){
750  if(prevBuffer) gst_buffer_unref (prevBuffer);
751  pixels.setFromExternalPixels(GST_BUFFER_DATA (buffer),pixels.getWidth(),pixels.getHeight(),pixels.getNumChannels());
752  prevBuffer = buffer;
753  bHavePixelsChanged=true;
754  }
755  }
756  }
757 #else
758  GstBuffer *buffer;
759  GstSample * sample;
760 
761  //get the buffer from appsink
762  if(isPaused()){
763  sample = gst_app_sink_pull_preroll (GST_APP_SINK (getSink()));
764  }else{
765  sample = gst_app_sink_pull_sample (GST_APP_SINK (getSink()));
766  }
767  buffer = gst_sample_get_buffer(sample);
768 
769  if(buffer){
770  if(pixels.isAllocated()){
771  if(prevBuffer) gst_sample_unref (prevBuffer);
772  gst_buffer_map (buffer, &mapinfo, GST_MAP_READ);
773  //TODO: stride = mapinfo.size / height;
774  pixels.setFromExternalPixels(mapinfo.data,pixels.getWidth(),pixels.getHeight(),pixels.getNumChannels());
775  prevBuffer = sample;
776  bHavePixelsChanged=true;
777  gst_buffer_unmap(buffer,&mapinfo);
778  }
779  }
780  }
781 #endif
782  }else{
783  ofLogWarning("ofGstVideoUtils") << "update(): ofGstVideoUtils not loaded";
784  }
785  bIsFrameNew = bHavePixelsChanged;
786  bHavePixelsChanged = false;
