396 void ofColor_<PixelType>::getHsb(float& hue, float& saturation, float& brightness) const {
397  float max = getBrightness();
398 
399  float min = r;
400  if(g < min) {
401  min = g;
402  }
403  if(b < min) {
404  min = b;
405  }
406 
407  if(max == min) { // grays
408  hue = 0.f;
409  saturation = 0.f;
410  brightness = max;
411  return;
412  }
413 
414  float hueSixth;
415  if(r == max) {
416  hueSixth = (g - b) / (max - min);
417  if(hueSixth < 0.f)
418  hueSixth += 6.f;
419  } else if (g == max) {
420  hueSixth = 2.f + (b - r) / (max - min);
421  } else {
422  hueSixth = 4.f + (r - g) / (max - min);
423  }
424  hue = limit() * hueSixth / 6.f;
425  saturation = limit() * (max - min) / max;
426  brightness = max;
427 }

454 template<typename PixelType>
455 void ofColor_<PixelType>::setHsb(float hue, float saturation, float brightness, float alpha ) {
456  saturation = ofClamp(saturation, 0, limit());
457  brightness = ofClamp(brightness, 0, limit());
458  if(brightness == 0) { // black
459  set(0);
460  } else if(saturation == 0) { // grays
461  set(brightness);
462  } else {
463  float hueSix = hue * 6. / limit();
464  float saturationNorm = saturation / limit();
465  int hueSixCategory = (int) floorf(hueSix);
466  float hueSixRemainder = hueSix - hueSixCategory;
467  PixelType pv = (PixelType) ((1.f - saturationNorm) * brightness);
468  PixelType qv = (PixelType) ((1.f - saturationNorm * hueSixRemainder) * brightness);
469  PixelType tv = (PixelType) ((1.f - saturationNorm * (1.f - hueSixRemainder)) * brightness);
470  switch(hueSixCategory) {
471  case 0: case 6: // r
472  r = brightness;
473  g = tv;
474  b = pv;
475  break;
476  case 1: // g
477  r = qv;
478  g = brightness;
479  b = pv;
480  break;
481  case 2:
482  r = pv;
483  g = brightness;
484  b = tv;
485  break;
486  case 3: // b
487  r = pv;
488  g = qv;
489  b = brightness;
490  break;
491  case 4:
492  r = tv;
493  g = pv;
494  b = brightness;
495  break;
496  case 5: // back to r
497  r = brightness;
498  g = pv;
499  b = qv;
500  break;
501  }
502  }
503 
504  // finally assign the alpha
505  a = alpha;
506 }
