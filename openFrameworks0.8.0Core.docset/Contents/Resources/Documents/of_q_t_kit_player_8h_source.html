12 // ofQTKitDecodeMode allows for highly efficient decoding of video
13 // pixel data. Options are available for direct-to-screen/texture rendering
14 // as well as more traditional CPU/RAM pixel-based rendering.
15 
16 enum ofQTKitDecodeMode {
17  OF_QTKIT_DECODE_PIXELS_ONLY = 0,
18  // OF_QTKIT_DECODE_PIXELS_ONLY: This mode decodes pixels only and can't
19  // be drawn directly to the screen without an an additional step.
20 
21  OF_QTKIT_DECODE_TEXTURE_ONLY,
22  // OF_QTKIT_DECODE_TEXTURE_ONLY: This is the fastest mode, but doesn't
23  // allow direct pixel access because video rendering skips the CPU and RAM.
24  // Pixels ARE available to texture / fragment shaders though.
25 
26  OF_QTKIT_DECODE_PIXELS_AND_TEXTURE
27  // OF_QTKIT_DECODE_PIXELS_AND_TEXTURE: This mode combines both
28  // OF_QTKIT_DECODE_PIXELS_ONLY AND OF_QTKIT_DECODE_TEXTURE_ONLY modes.
29  // It is slightly slower OF_QTKIT_DECODE_TEXTURE_ONLY, but faster than
30  // managing your own ofTexture.
31 };
