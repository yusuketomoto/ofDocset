504 - (void) update
505 {
506  @synchronized(self){
507  if(hasNewFrame){
508  if(pixels == NULL){
509  ofLogError("ofQTKitGrabber") << "pixels not set on Obj-C Class from C++ wrapper";
510  return;
511  }
512 
513  //pixels won't be allocated if we didn't force the video width and height
514  //allocate upon receving the first frame
515  if(!pixels->isAllocated()){
516  pixels->allocate(CVPixelBufferGetWidth(cvFrame),
517  CVPixelBufferGetHeight(cvFrame), OF_IMAGE_COLOR);
518  width = CVPixelBufferGetWidth(cvFrame);
519  height = CVPixelBufferGetHeight(cvFrame);
520  }
521 
522  CVPixelBufferLockBaseAddress(cvFrame, kCVPixelBufferLock_ReadOnly);
523 #if( MAC_OS_X_VERSION_10_8 && MAC_OS_X_VERSION_MIN_REQUIRED >= MAC_OS_X_VERSION_10_8)
524  //fast conversion of BGRA -> RGB introduced in 10.8
525  vImage_Buffer src;
526  src.height = CVPixelBufferGetHeight(cvFrame);
527  src.width = CVPixelBufferGetWidth(cvFrame);
528  src.data = CVPixelBufferGetBaseAddress(cvFrame);
529  src.rowBytes = CVPixelBufferGetBytesPerRow(cvFrame);
530 
531  vImage_Buffer dest;
532  dest.height = height;
533  dest.width = width;
534  dest.rowBytes = width*3;
535  dest.data = pixels->getPixels();
536 
537  vImage_Error err = vImageConvert_BGRA8888toRGB888(&src, &dest, kvImageNoFlags);
538  if(err != kvImageNoError){
539  NSLog(@"[error] ofxQTKitGrabber: update(): pixel copy vImage_error %ld", err);
540  }
541 #else
542  //manually convert BGRA -> RGB
543  int dstBpp = 3;
544  int srcBpp = 4;
545  int dstRowBytes = width*dstBpp;
546  int srcRowBytes = CVPixelBufferGetBytesPerRow(cvFrame);
547  for(int y = 0; y < height; y++){
548  unsigned char* src = ((unsigned char*)CVPixelBufferGetBaseAddress(cvFrame)) + y*srcRowBytes;
549  unsigned char* dst = pixels->getPixels() + y*dstRowBytes;
550  for(int x = 0; x < width; x++){
551  dst[0] = src[2];
552  dst[1] = src[1];
553  dst[2] = src[0];
554  dst+=dstBpp;
555  src+=srcBpp;
556  }
557  }
558 #endif
559  CVPixelBufferUnlockBaseAddress(cvFrame, kCVPixelBufferLock_ReadOnly);
560 
561  hasNewFrame = NO;
562  isFrameNew = YES;
563  }
564  else{
565  isFrameNew = NO;
566  }
567  }
