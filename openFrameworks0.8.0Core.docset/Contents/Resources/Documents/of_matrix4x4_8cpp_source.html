595 /******************************************
596  Matrix inversion technique:
597  Given a matrix mat, we want to invert it.
598  mat = [ r00 r01 r02 a
599  r10 r11 r12 b
600  r20 r21 r22 c
601  tx ty tz d ]
602  We note that this matrix can be split into three matrices.
603  mat = rot * trans * corr, where rot is rotation part, trans is translation part, and corr is the correction due to perspective (if any).
604  rot = [ r00 r01 r02 0
605  r10 r11 r12 0
606  r20 r21 r22 0
607  0 0 0 1 ]
608  trans = [ 1 0 0 0
609  0 1 0 0
610  0 0 1 0
611  tx ty tz 1 ]
612  corr = [ 1 0 0 px
613  0 1 0 py
614  0 0 1 pz
615  0 0 0 s ]
616  where the elements of corr are obtained from linear combinations of the elements of rot, trans, and mat.
617  So the inverse is mat' = (trans * corr)' * rot', where rot' must be computed the traditional way, which is easy since it is only a 3x3 matrix.
618  This problem is simplified if [px py pz s] = [0 0 0 1], which will happen if mat was composed only of rotations, scales, and translations (which is common). In this case, we can ignore corr entirely which saves on a lot of computations.
619  ******************************************/
620 
621 bool invert_4x3( const ofMatrix4x4& src, ofMatrix4x4 & dst )
622 {
623  if (&src==&dst)
624  {
625  ofMatrix4x4 tm(src);
626  return invert_4x3(tm,dst);
627  }
628 
629  register float r00, r01, r02,
630  r10, r11, r12,
631  r20, r21, r22;
632  // Copy rotation components directly into registers for speed
633  r00 = src._mat[0][0]; r01 = src._mat[0][1]; r02 = src._mat[0][2];
634  r10 = src._mat[1][0]; r11 = src._mat[1][1]; r12 = src._mat[1][2];
635  r20 = src._mat[2][0]; r21 = src._mat[2][1]; r22 = src._mat[2][2];
636 
637  // Partially compute inverse of rot
638  dst._mat[0][0] = r11*r22 - r12*r21;
639  dst._mat[0][1] = r02*r21 - r01*r22;
640  dst._mat[0][2] = r01*r12 - r02*r11;
641 
642  // Compute determinant of rot from 3 elements just computed
643  register float one_over_det = 1.0/(r00*dst._mat[0][0] + r10*dst._mat[0][1] + r20*dst._mat[0][2]);
644  r00 *= one_over_det; r10 *= one_over_det; r20 *= one_over_det; // Saves on later computations
645 
646  // Finish computing inverse of rot
647  dst._mat[0][0] *= one_over_det;
648  dst._mat[0][1] *= one_over_det;
649  dst._mat[0][2] *= one_over_det;
650  dst._mat[0][3] = 0.0;
651  dst._mat[1][0] = r12*r20 - r10*r22; // Have already been divided by det
652  dst._mat[1][1] = r00*r22 - r02*r20; // same
653  dst._mat[1][2] = r02*r10 - r00*r12; // same
654  dst._mat[1][3] = 0.0;
655  dst._mat[2][0] = r10*r21 - r11*r20; // Have already been divided by det
656  dst._mat[2][1] = r01*r20 - r00*r21; // same
657  dst._mat[2][2] = r00*r11 - r01*r10; // same
658  dst._mat[2][3] = 0.0;
659  dst._mat[3][3] = 1.0;
660 
661  // We no longer need the rxx or det variables anymore, so we can reuse them for whatever we want. But we will still rename them for the sake of clarity.
662 
663 #define d r22
664  d = src._mat[3][3];
665 
666  if( square(d-1.0) > 1.0e-6 ) // Involves perspective, so we must
667  { // compute the full inverse
668 
669  ofMatrix4x4 TPinv;
670  dst._mat[3][0] = dst._mat[3][1] = dst._mat[3][2] = 0.0;
671 
672 #define px r00
673 #define py r01
674 #define pz r02
675 #define one_over_s one_over_det
676 #define a r10
677 #define b r11
678 #define c r12
679 
680  a = src._mat[0][3]; b = src._mat[1][3]; c = src._mat[2][3];
681  px = dst._mat[0][0]*a + dst._mat[0][1]*b + dst._mat[0][2]*c;
682  py = dst._mat[1][0]*a + dst._mat[1][1]*b + dst._mat[1][2]*c;
683  pz = dst._mat[2][0]*a + dst._mat[2][1]*b + dst._mat[2][2]*c;
684 
685 #undef a
686 #undef b
687 #undef c
688 #define tx r10
689 #define ty r11
690 #define tz r12
691 
692  tx = src._mat[3][0]; ty = src._mat[3][1]; tz = src._mat[3][2];
693  one_over_s = 1.0/(d - (tx*px + ty*py + tz*pz));
694 
695  tx *= one_over_s; ty *= one_over_s; tz *= one_over_s; // Reduces number of calculations later on
696 
697  // Compute inverse of trans*corr
698  TPinv._mat[0][0] = tx*px + 1.0;
699  TPinv._mat[0][1] = ty*px;
700  TPinv._mat[0][2] = tz*px;
701  TPinv._mat[0][3] = -px * one_over_s;
702  TPinv._mat[1][0] = tx*py;
703  TPinv._mat[1][1] = ty*py + 1.0;
704  TPinv._mat[1][2] = tz*py;
705  TPinv._mat[1][3] = -py * one_over_s;
706  TPinv._mat[2][0] = tx*pz;
707  TPinv._mat[2][1] = ty*pz;
708  TPinv._mat[2][2] = tz*pz + 1.0;
709  TPinv._mat[2][3] = -pz * one_over_s;
710  TPinv._mat[3][0] = -tx;
711  TPinv._mat[3][1] = -ty;
712  TPinv._mat[3][2] = -tz;
713  TPinv._mat[3][3] = one_over_s;
714 
715  dst.preMult(TPinv); // Finish computing full inverse of mat
716 
717 #undef px
718 #undef py
719 #undef pz
720 #undef one_over_s
721 #undef d
722  }
723  else // Rightmost column is [0; 0; 0; 1] so it can be ignored
724  {
725  tx = src._mat[3][0]; ty = src._mat[3][1]; tz = src._mat[3][2];
726 
727  // Compute translation components of mat'
728  dst._mat[3][0] = -(tx*dst._mat[0][0] + ty*dst._mat[1][0] + tz*dst._mat[2][0]);
729  dst._mat[3][1] = -(tx*dst._mat[0][1] + ty*dst._mat[1][1] + tz*dst._mat[2][1]);
730  dst._mat[3][2] = -(tx*dst._mat[0][2] + ty*dst._mat[1][2] + tz*dst._mat[2][2]);
731 
732 #undef tx
733 #undef ty
734 #undef tz
735  }
736 
737  return true;
738 }

1246 /* Compute the spectral decomposition of symmetric positive semi-definite S.
1247 * Returns rotation in U and scale factors in result, so that if K is a diagonal
1248 * matrix of the scale factors, then S = U K (U transpose). Uses Jacobi method.
1249 * See Gene H. Golub and Charles F. Van Loan. Matrix Computations. Hopkins 1983.

1256  static char nxt[] = {Y,Z,X};

1260  OffD[X] = S[Y][Z]; OffD[Y] = S[Z][X]; OffD[Z] = S[X][Y];

1277  }
