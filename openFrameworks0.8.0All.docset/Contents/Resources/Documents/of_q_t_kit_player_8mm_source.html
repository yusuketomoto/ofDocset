31 bool ofQTKitPlayer::loadMovie(string movieFilePath, ofQTKitDecodeMode mode) {
32  if(mode != OF_QTKIT_DECODE_PIXELS_ONLY && mode != OF_QTKIT_DECODE_TEXTURE_ONLY && mode != OF_QTKIT_DECODE_PIXELS_AND_TEXTURE){
33  ofLogError("ofQTKitPlayer") << "loadMovie(): unknown ofQTKitDecodeMode mode";
34  return false;
35  }
36 
37  if(isLoaded()){
38  close(); //auto released
39  }
40 
41  NSAutoreleasePool* pool = [[NSAutoreleasePool alloc] init];
42 
43  decodeMode = mode;
44  bool useTexture = (mode == OF_QTKIT_DECODE_TEXTURE_ONLY || mode == OF_QTKIT_DECODE_PIXELS_AND_TEXTURE);
45  bool usePixels = (mode == OF_QTKIT_DECODE_PIXELS_ONLY || mode == OF_QTKIT_DECODE_PIXELS_AND_TEXTURE);
46  bool useAlpha = (pixelFormat == OF_PIXELS_RGBA);
47 
48  bool isURL = false;
49 
50  if (Poco::icompare(movieFilePath.substr(0,7), "http://") == 0 ||
51  Poco::icompare(movieFilePath.substr(0,8), "https://") == 0 ||
52  Poco::icompare(movieFilePath.substr(0,7), "rtsp://") == 0) {
53  isURL = true;
54  }
55  else {
56  movieFilePath = ofToDataPath(movieFilePath, false);
57  }
58 
59  moviePlayer = [[QTKitMovieRenderer alloc] init];
60  BOOL success = [moviePlayer loadMovie:[NSString stringWithCString:movieFilePath.c_str() encoding:NSUTF8StringEncoding]
61  pathIsURL:isURL
62  allowTexture:useTexture
63  allowPixels:usePixels
64  allowAlpha:useAlpha];
65 
66  if(success){
67  moviePlayer.synchronousSeek = bSynchronousSeek;
68  reallocatePixels();
69  moviePath = movieFilePath;
70  duration = moviePlayer.duration;
71 
72  setLoopState(currentLoopState);
73  setSpeed(1.0f);
74  firstFrame(); //will load the first frame
75  }
76  else {
77  ofLogError("ofQTKitPlayer") << "loadMovie(): couldn't load \"" << movieFilePath << "\"";
78  [moviePlayer release];
79  moviePlayer = NULL;
80  }
81 
82  [pool release];
83 
84  return success;
85 }
