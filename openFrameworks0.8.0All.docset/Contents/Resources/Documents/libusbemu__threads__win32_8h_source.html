77 struct QuickThread
78 {
79 private:
80  HANDLE hThread;
81 
82  // Type-safe wrapper that converts arbitrary function signatures into the
83  // required signature of Win32 Thread Procedures (LPTHREAD_START_ROUTINE).
84  // Any Win32 LPTHREAD_START_ROUTINE declared routine can be wrapped as well.
85  // The wrapper is also capable of cleaning up itself upon thread termination.
86  // This wrapper can be extended in the future to support member-functions
87  // to run as thread procedures.
88  template<typename F>
89  struct ThreadWrapper
90  {
91  struct State
92  {
93  F* routine;
94  void* params;
95  bool release;
96  QuickThread* instance;
97  QuickEvent* sigclone;
98  };
99  static DWORD WINAPI Thunk(LPVOID lpParameter)
100  {
101  State state = *((State*)lpParameter); // clone state (no heap alloc!)
102  state.sigclone->Signal(); // done cloning, signal back to creator
103  state.sigclone = NULL;
104 
105  // start wrapped thread procedure
106  DWORD ret = (DWORD)state.routine(state.params);
107 
108  // release the associated QuickThread instance if requested
109  if (state.release)
110  delete(state.instance);
111 
112  return(ret);
113  }
114  };
115 
116  // allow the creation of pseudo-handles to the calling thread
117  // this constructor cannot and should never be called explicitly!
118  // use QuickThread::Myself() to spawn a pseudo-handle QuickThread
119  inline QuickThread() : hThread(GetCurrentThread()) {}
120 
121 public:
122  template<typename F>
123  inline QuickThread(F* proc, void* params, const bool auto_release=false) : hThread(NULL)
124  {
125  // the 'typename' is required here because of dependent names...
126  // MSVC relaxes this constraint, but it goes against the standard.
127  typename ThreadWrapper<F>::State state;
128  state.routine = proc;
129  state.params = params;
130  state.release = auto_release;
131  state.instance = this;
132  // in order to prevent heap allocation, an event is created so that the
133  // thunk function can signal back when it is done cloning the state; this
134  // may look like unnecessary overhead, but the less heap memory control,
135  // the better becomes the management and maintenance of this class.
136  QuickEvent hWaitThunkCloneState;
137  state.sigclone = &hWaitThunkCloneState;
138 
139  // Ready to issue the thread creation:
140  hThread = CreateThread(NULL, 0, &ThreadWrapper<F>::Thunk, (LPVOID)&state, 0, NULL);
141 
142  // Wait for the thread thunk to clone the state...
143  hWaitThunkCloneState.Wait();
144  // Event object will then be automatically released upon return
145  }
146 
147  inline ~QuickThread()
148  {
149  // only if not a pseudo-handle...
150  if (hThread == GetCurrentThread())
151  return;
152  CloseHandle(hThread);
153  }

271  int WaitAnyUntilTimeout(const unsigned int milliseconds)
272  {
273  int index (-1);
274  mutex.Enter();
275  DWORD ret (WAIT_FAILED);
276  const unsigned int nHandles (m_vHandles.size());
277  if (nHandles > 0)
278  ret = WaitForMultipleObjects(nHandles, &m_vHandles[0], FALSE, milliseconds);
279  if (ret - WAIT_OBJECT_0 < nHandles)
280  index = (int)(ret - WAIT_OBJECT_0);
281  mutex.Leave();
282  return(index);
283  }
