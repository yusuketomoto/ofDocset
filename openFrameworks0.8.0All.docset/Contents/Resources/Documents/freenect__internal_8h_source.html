97 static inline int16_t fn_le16s(int16_t s)
98 {
99  // reinterpret cast to unsigned, use the normal fn_le16, and then reinterpret cast back
100  union {
101  int16_t s;
102  uint16_t u;
103  } conversion_union;
104  conversion_union.s = s;
105  conversion_union.u = fn_le16(conversion_union.u);
106  return conversion_union.s;
107 }
