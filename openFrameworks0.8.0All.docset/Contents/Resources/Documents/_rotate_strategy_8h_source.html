91  int index = 0;

93  switch (timestr.count())
94  {
95  case 3: // day,hh:mm
96  {

98  _day = DateTimeParser::parseDayOfWeek(it, timestr[index].end());
99  ++index;
100  }
101  case 2: // hh:mm
102  _hour = NumberParser::parse(timestr[index]);
103  ++index;
104  case 1: // mm
105  _minute = NumberParser::parse(timestr[index]);
106  break;
107  default:

109  }

111  }

135  while (!(_threshold.minute() == _minute &&

137  (-1 == _day || _threshold.dayOfWeek() == _day)));
