59 FN_INTERNAL int fnusb_list_device_attributes(fnusb_ctx *ctx, struct freenect_device_attributes** attribute_list)
60 {
61  *attribute_list = NULL; // initialize some return value in case the user is careless.
62  libusb_device **devs;
63  //pointer to pointer of device, used to retrieve a list of devices
64  ssize_t count = libusb_get_device_list (ctx->ctx, &devs);
65  if (count < 0)
66  return -1;
67 
68  struct freenect_device_attributes** camera_prev_next = attribute_list;
69 
70  // Pass over the list. For each camera seen, if we already have a camera
71  // for the newest_camera device, allocate a new one and append it to the list,
72  // incrementing num_devs. Likewise for each audio device.
73  struct libusb_device_descriptor desc;
74  int num_cams = 0;
75  int i;
76  for (i = 0; i < count; i++) {
77  int r = libusb_get_device_descriptor (devs[i], &desc);
78  if (r < 0)
79  continue;
80  if (desc.idVendor == VID_MICROSOFT && (desc.idProduct == PID_NUI_CAMERA || desc.idProduct == PID_K4W_CAMERA)) {
81  // Verify that a serial number exists to query. If not, don't touch the device.
82  if (desc.iSerialNumber == 0) {
83  continue;
84  }
85 
86  // Open device.
87  int res;
88  libusb_device_handle *this_device;
89  res = libusb_open(devs[i], &this_device);
90  unsigned char string_desc[256]; // String descriptors are at most 256 bytes.
91  if (res != 0) {
92  continue;
93  }
94 
95  // Read string descriptor referring to serial number.
96  res = libusb_get_string_descriptor_ascii(this_device, desc.iSerialNumber, string_desc, 256);
97  libusb_close(this_device);
98  if (res < 0) {
99  continue;
100  }
101 
102  // Add item to linked list.
103  struct freenect_device_attributes* new_dev_attrs = (struct freenect_device_attributes*)malloc(sizeof(struct freenect_device_attributes));
104  memset(new_dev_attrs, 0, sizeof(*new_dev_attrs));
105

107  // Copy string with serial number
108  new_dev_attrs->camera_serial = strdup((char*)string_desc);
109  camera_prev_next = &(new_dev_attrs->next);
110  // Increment number of cameras found

450 static void iso_callback(struct libusb_transfer *xfer)
451 {
452  int i;
453  fnusb_isoc_stream *strm = (fnusb_isoc_stream*)xfer->user_data;
454  freenect_context *ctx = strm->parent->parent->parent;
455 
456  if (strm->dead) {
457  strm->dead_xfers++;
458  FN_SPEW("EP %02x transfer complete, %d left\n", xfer->endpoint, strm->num_xfers - strm->dead_xfers);
459  return;
460  }
461 
462  switch(xfer->status) {
463  case LIBUSB_TRANSFER_COMPLETED: // Normal operation.
464  {
465  uint8_t *buf = (uint8_t*)xfer->buffer;
466  for (i=0; i<strm->pkts; i++) {
467  strm->cb(strm->parent->parent, buf, xfer->iso_packet_desc[i].actual_length);
468  buf += strm->len;
469  }
470  int res;
471  res = libusb_submit_transfer(xfer);
472  if (res != 0) {
473  FN_ERROR("iso_callback(): failed to resubmit transfer after successful completion: %d\n", res);
474  strm->dead_xfers++;
475  if (res == LIBUSB_ERROR_NO_DEVICE) {
476  strm->parent->device_dead = 1;
477  }
478  }
479  break;
480  }
481  case LIBUSB_TRANSFER_NO_DEVICE:
482  {
483  // We lost the device we were talking to. This is a large problem,
484  // and one that we should eventually come up with a way to
485  // properly propagate up to the caller.
486  if(!strm->parent->device_dead) {
487  FN_ERROR("USB device disappeared, cancelling stream %02x :(\n", xfer->endpoint);
488  }
489  strm->dead_xfers++;
490  strm->parent->device_dead = 1;
491  break;
492  }
493  case LIBUSB_TRANSFER_CANCELLED:
494  {
495  if(strm->dead) {
496  FN_SPEW("EP %02x transfer cancelled\n", xfer->endpoint);
497  } else {
498  // This seems to be a libusb bug on OSX - instead of completing
499  // the transfer with LIBUSB_TRANSFER_NO_DEVICE, the transfers
500  // simply come back cancelled by the OS. We can detect this,
501  // though - the stream should be marked dead if we're
502  // intentionally cancelling transfers.
503  if(!strm->parent->device_dead) {
504  FN_ERROR("Got cancelled transfer, but we didn't request it - device disconnected?\n");
505  }
506  strm->parent->device_dead = 1;
507  }
508  strm->dead_xfers++;
509  break;
510  }
511  default:
512  {
513  // On other errors, resubmit the transfer - in particular, libusb
514  // on OSX tends to hit random errors a lot. If we don't resubmit
515  // the transfers, eventually all of them die and then we don't get
516  // any more data from the Kinect.
517  FN_WARNING("Isochronous transfer error: %d\n", xfer->status);
518  int res;
519  res = libusb_submit_transfer(xfer);
520  if (res != 0) {
521  FN_ERROR("Isochronous transfer resubmission failed after unknown error: %d\n", res);
522  strm->dead_xfers++;
523  if (res == LIBUSB_ERROR_NO_DEVICE) {
524  strm->parent->device_dead = 1;
525  }
526  }
527  break;
528  }
529  }
530 }
