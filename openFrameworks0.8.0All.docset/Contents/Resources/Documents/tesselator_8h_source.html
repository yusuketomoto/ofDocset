52 // Tesselation result element types:
53 // TESS_TRIANGLES
54 // Each element is defined as array of 3 TESSindex.
55 // The 3 values in the array defines indices to each vertex of a triangle.
56 // TESS_CONNECTED_TRIANGLES
57 // Each element is defined as array of 6 TESSindex.
58 // The first 3 values in the array defines indices to each vertex of a triangle.
59 // The second 3 values in the array defines indices to each of neighbour element or -1 if there is no neighbour.
60 // TESS_BOUNDARY_CONTOURS
61 // Each element is defined as array of 2 TESSindex.
62 // The first value is index to first vertex in contour and the second value is number of vertices in the contour.

83 // Custom memory allocator interface.
84 // The internal memory allocator allocates mesh edges, vertices and faces
85 // as well as dictionary nodes and active regions in buckets and uses simple
86 // freelist to speed up the allocation. The bucket size should roughly match your
87 // expected input data. For example if you process only hundreds of vertices,
88 // a bucket size of 128 might be ok, where as when processing thousands of vertices
89 // bucket size of 1024 might be approproate. The bucket size is a compromise between
90 // how often to allocate memory from the system versus how much extra space the system
91 // should allocate. Reasonable defaults are show in commects below, they will be used if
92 // the bucket sizes are zero.
93 //
94 // The use may left the memrealloc to be null. In that case, the tesselator will not try to
95 // dynamically grow int's internal arrays. The tesselator only needs the reallocation when it
96 // has found intersecting segments and needs to add new vertex. This defency can be cured by
97 // allocating some extra vertices beforehand. The 'extraVertices' variable allows to specify
98 // number of expected extra vertices.
99 struct TESSalloc
100 {
101  void *(*memalloc)( void *userData, unsigned int size );
102  void *(*memrealloc)( void *userData, void* ptr, unsigned int size );
103  void (*memfree)( void *userData, void *ptr );
104  void* userData; // User data passed to the allocator functions.
105  int meshEdgeBucketSize; // 512
106  int meshVertexBucketSize; // 512
107  int meshFaceBucketSize; // 256
108  int dictNodeBucketSize; // 512
109  int regionBucketSize; // 256
110  int extraVertices; // Number of extra vertices allocated for the priority queue.
111 };
