49 static void Normalize( TESSreal v[3] )
50 {
51  TESSreal len = v[0]*v[0] + v[1]*v[1] + v[2]*v[2];
52 
53  if( len <= 0 ) return;
54  len = sqrtf( len );
55  v[0] /= len;
56  v[1] /= len;
57  v[2] /= len;
58 }

71 static void ComputeNormal( TESStesselator *tess, TESSreal norm[3] )
72 {
73  TESSvertex *v, *v1, *v2;
74  TESSreal c, tLen2, maxLen2;
75  TESSreal maxVal[3], minVal[3], d1[3], d2[3], tNorm[3];
76  TESSvertex *maxVert[3], *minVert[3];
77  TESSvertex *vHead = &tess->mesh->vHead;
78  int i;
79 
80  v = vHead->next;
81  for( i = 0; i < 3; ++i ) {
82  c = v->coords[i];
83  minVal[i] = c;
84  minVert[i] = v;
85  maxVal[i] = c;
86  maxVert[i] = v;
87  }
88 
89  for( v = vHead->next; v != vHead; v = v->next ) {
90  for( i = 0; i < 3; ++i ) {
91  c = v->coords[i];
92  if( c < minVal[i] ) { minVal[i] = c; minVert[i] = v; }
93  if( c > maxVal[i] ) { maxVal[i] = c; maxVert[i] = v; }
94  }
95  }
96 
97  /* Find two vertices separated by at least 1/sqrt(3) of the maximum
98  * distance between any two vertices
99  */
100  i = 0;
101  if( maxVal[1] - minVal[1] > maxVal[0] - minVal[0] ) { i = 1; }
102  if( maxVal[2] - minVal[2] > maxVal[i] - minVal[i] ) { i = 2; }
103  if( minVal[i] >= maxVal[i] ) {
104  /* All vertices are the same -- normal doesn't matter */
105  norm[0] = 0; norm[1] = 0; norm[2] = 1;
106  return;
107  }
108 
109  /* Look for a third vertex which forms the triangle with maximum area
110  * (Length of normal == twice the triangle area)
111  */
112  maxLen2 = 0;
113  v1 = minVert[i];
114  v2 = maxVert[i];
115  d1[0] = v1->coords[0] - v2->coords[0];
116  d1[1] = v1->coords[1] - v2->coords[1];
117  d1[2] = v1->coords[2] - v2->coords[2];
118  for( v = vHead->next; v != vHead; v = v->next ) {
119  d2[0] = v->coords[0] - v2->coords[0];
120  d2[1] = v->coords[1] - v2->coords[1];
121  d2[2] = v->coords[2] - v2->coords[2];
122  tNorm[0] = d1[1]*d2[2] - d1[2]*d2[1];
123  tNorm[1] = d1[2]*d2[0] - d1[0]*d2[2];
124  tNorm[2] = d1[0]*d2[1] - d1[1]*d2[0];
125  tLen2 = tNorm[0]*tNorm[0] + tNorm[1]*tNorm[1] + tNorm[2]*tNorm[2];
126  if( tLen2 > maxLen2 ) {
127  maxLen2 = tLen2;
128  norm[0] = tNorm[0];
129  norm[1] = tNorm[1];
130  norm[2] = tNorm[2];
131  }
132  }
133 
134  if( maxLen2 <= 0 ) {
135  /* All points lie on a single line -- any decent normal will do */
136  norm[0] = norm[1] = norm[2] = 0;
137  norm[LongAxis(d1)] = 1;
138  }
139 }

142 static void CheckOrientation( TESStesselator *tess )
143 {
144  TESSreal area;
145  TESSface *f, *fHead = &tess->mesh->fHead;
146  TESSvertex *v, *vHead = &tess->mesh->vHead;
147  TESShalfEdge *e;
148 
149  /* When we compute the normal automatically, we choose the orientation
150  * so that the the sum of the signed areas of all contours is non-negative.
151  */
152  area = 0;
153  for( f = fHead->next; f != fHead; f = f->next ) {
154  e = f->anEdge;
155  if( e->winding <= 0 ) continue;
156  do {
157  area += (e->Org->s - e->Dst->s) * (e->Org->t + e->Dst->t);
158  e = e->Lnext;
159  } while( e != f->anEdge );
160  }
161  if( area < 0 ) {
162  /* Reverse the orientation by flipping all the t-coordinates */
163  for( v = vHead->next; v != vHead; v = v->next ) {
164  v->t = - v->t;
165  }
166  tess->tUnit[0] = - tess->tUnit[0];
167  tess->tUnit[1] = - tess->tUnit[1];
168  tess->tUnit[2] = - tess->tUnit[2];
169  }
170 }

196 /* Determine the polygon normal and project vertices onto the plane
197 * of the polygon.

199 void tessProjectPolygon( TESStesselator *tess )
200 {
201  TESSvertex *v, *vHead = &tess->mesh->vHead;
202  TESSreal norm[3];
203  TESSreal *sUnit, *tUnit;
204  int i, first, computedNormal = FALSE;
205 
206  norm[0] = tess->normal[0];
207  norm[1] = tess->normal[1];
208  norm[2] = tess->normal[2];
209  if( norm[0] == 0 && norm[1] == 0 && norm[2] == 0 ) {
210  ComputeNormal( tess, norm );
211  computedNormal = TRUE;
212  }
213  sUnit = tess->sUnit;
214  tUnit = tess->tUnit;
215  i = LongAxis( norm );
216 
217 #if defined(FOR_TRITE_TEST_PROGRAM) || defined(TRUE_PROJECT)
218  /* Choose the initial sUnit vector to be approximately perpendicular
219  * to the normal.
220  */
221  Normalize( norm );
222 
223  sUnit[i] = 0;
224  sUnit[(i+1)%3] = S_UNIT_X;
225  sUnit[(i+2)%3] = S_UNIT_Y;
226 
227  /* Now make it exactly perpendicular */
228  w = Dot( sUnit, norm );
229  sUnit[0] -= w * norm[0];
230  sUnit[1] -= w * norm[1];
231  sUnit[2] -= w * norm[2];
232  Normalize( sUnit );
233 
234  /* Choose tUnit so that (sUnit,tUnit,norm) form a right-handed frame */
235  tUnit[0] = norm[1]*sUnit[2] - norm[2]*sUnit[1];
236  tUnit[1] = norm[2]*sUnit[0] - norm[0]*sUnit[2];
237  tUnit[2] = norm[0]*sUnit[1] - norm[1]*sUnit[0];
238  Normalize( tUnit );
239 #else
240  /* Project perpendicular to a coordinate axis -- better numerically */
241  sUnit[i] = 0;
242  sUnit[(i+1)%3] = S_UNIT_X;
243  sUnit[(i+2)%3] = S_UNIT_Y;
244 
245  tUnit[i] = 0;
246  tUnit[(i+1)%3] = (norm[i] > 0) ? -S_UNIT_Y : S_UNIT_Y;
247  tUnit[(i+2)%3] = (norm[i] > 0) ? S_UNIT_X : -S_UNIT_X;
248 #endif
249 
250  /* Project the vertices onto the sweep plane */
251  for( v = vHead->next; v != vHead; v = v->next )
252  {
253  v->s = Dot( v->coords, sUnit );
254  v->t = Dot( v->coords, tUnit );
255  }
256  if( computedNormal ) {
257  CheckOrientation( tess );
258  }
259 
260  /* Compute ST bounds. */
261  first = 1;
262  for( v = vHead->next; v != vHead; v = v->next )
263  {
264  if (first)
265  {
266  tess->bmin[0] = tess->bmax[0] = v->s;
267  tess->bmin[1] = tess->bmax[1] = v->t;
268  first = 0;
269  }
270  else
271  {
272  if (v->s < tess->bmin[0]) tess->bmin[0] = v->s;
273  if (v->s > tess->bmax[0]) tess->bmax[0] = v->s;
274  if (v->t < tess->bmin[1]) tess->bmin[1] = v->t;
275  if (v->t > tess->bmax[1]) tess->bmax[1] = v->t;
276  }
277  }
278 }

283 /* tessMeshTessellateMonoRegion( face ) tessellates a monotone region
284 * (what else would it do??) The region must consist of a single
285 * loop of half-edges (see mesh.h) oriented CCW. "Monotone" in this
286 * case means that any vertical line intersects the interior of the
287 * region in a single interval.
288 *
289 * Tessellation consists of adding interior edges (actually pairs of
290 * half-edges), to split the region into non-overlapping triangles.
291 *
292 * The basic idea is explained in Preparata and Shamos (which I don''t
293 * have handy right now), although their implementation is more
294 * complicated than this one. The are two edge chains, an upper chain
295 * and a lower chain. We process all vertices from both chains in order,
296 * from right to left.
297 *
298 * The algorithm ensures that the following invariant holds after each
299 * vertex is processed: the untessellated region consists of two
300 * chains, where one chain (say the upper) is a single edge, and
301 * the other chain is concave. The left vertex of the single edge
302 * is always to the left of all vertices in the concave chain.
303 *
304 * Each step consists of adding the rightmost unprocessed vertex to one
305 * of the two chains, and forming a fan of triangles from the rightmost
306 * of two chain endpoints. Determining whether we can add each triangle
307 * to the fan is a simple orientation test. By making the fan as large
308 * as possible, we restore the invariant (check it yourself).
309 */
310 int tessMeshTessellateMonoRegion( TESSmesh *mesh, TESSface *face )
311 {
312  TESShalfEdge *up, *lo;
313 
314  /* All edges are oriented CCW around the boundary of the region.
315  * First, find the half-edge whose origin vertex is rightmost.
316  * Since the sweep goes from left to right, face->anEdge should
317  * be close to the edge we want.
318  */
319  up = face->anEdge;
320  if(!( up->Lnext != up && up->Lnext->Lnext != up )) return 1;
321 
322  for( ; VertLeq( up->Dst, up->Org ); up = up->Lprev )
323  ;
324  for( ; VertLeq( up->Org, up->Dst ); up = up->Lnext )
325  ;
326  lo = up->Lprev;
327 
328  while( up->Lnext != lo ) {
329  if( VertLeq( up->Dst, lo->Org )) {
330  /* up->Dst is on the left. It is safe to form triangles from lo->Org.
331  * The EdgeGoesLeft test guarantees progress even when some triangles
332  * are CW, given that the upper and lower chains are truly monotone.
333  */
334  while( lo->Lnext != up && (EdgeGoesLeft( lo->Lnext )
335  || EdgeSign( lo->Org, lo->Dst, lo->Lnext->Dst ) <= 0 )) {
336  TESShalfEdge *tempHalfEdge= tessMeshConnect( mesh, lo->Lnext, lo );
337  if (tempHalfEdge == NULL) return 0;
338  lo = tempHalfEdge->Sym;
339  }
340  lo = lo->Lprev;
341  } else {
342  /* lo->Org is on the left. We can make CCW triangles from up->Dst. */
343  while( lo->Lnext != up && (EdgeGoesRight( up->Lprev )
344  || EdgeSign( up->Dst, up->Org, up->Lprev->Org ) >= 0 )) {
345  TESShalfEdge *tempHalfEdge= tessMeshConnect( mesh, up, up->Lprev );
346  if (tempHalfEdge == NULL) return 0;
347  up = tempHalfEdge->Sym;
348  }
349  up = up->Lnext;
350  }
351  }
352 
353  /* Now lo->Org == up->Dst == the leftmost vertex. The remaining region
354  * can be tessellated in a fan from this leftmost vertex.
355  */
356  if( lo->Lnext == up ) return 1;
357  while( lo->Lnext->Lnext != up ) {
358  TESShalfEdge *tempHalfEdge= tessMeshConnect( mesh, lo->Lnext, lo );
359  if (tempHalfEdge == NULL) return 0;
360  lo = tempHalfEdge->Sym;
361  }
362 
363  return 1;
364 }

407 /* tessMeshSetWindingNumber( mesh, value, keepOnlyBoundary ) resets the
408 * winding numbers on all edges so that regions marked "inside" the
409 * polygon have a winding number of "value", and regions outside
410 * have a winding number of 0.
411 *
412 * If keepOnlyBoundary is TRUE, it also deletes all edges which do not
413 * separate an interior region from an exterior one.
414 */
415 int tessMeshSetWindingNumber( TESSmesh *mesh, int value,
416  int keepOnlyBoundary )
417 {
418  TESShalfEdge *e, *eNext;
419 
420  for( e = mesh->eHead.next; e != &mesh->eHead; e = eNext ) {
421  eNext = e->next;
422  if( e->Rface->inside != e->Lface->inside ) {
423 
424  /* This is a boundary edge (one side is interior, one is exterior). */
425  e->winding = (e->Lface->inside) ? value : -value;
426  } else {
427 
428  /* Both regions are interior, or both are exterior. */
429  if( ! keepOnlyBoundary ) {
430  e->winding = 0;
431  } else {
432  if ( !tessMeshDelete( mesh, e ) ) return 0;
433  }
434  }
435  }
436  return 1;
437 }

566 void OutputPolymesh( TESStesselator *tess, TESSmesh *mesh, int elementType, int polySize, int vertexSize )
567 {
568  TESSvertex* v = 0;
569  TESSface* f = 0;
570  TESShalfEdge* edge = 0;
571  TESSindex maxFaceCount = 0;
572  TESSindex maxVertexCount = 0;
573  TESSindex faceVerts, i;
574  TESSindex *elements = 0;
575  TESSreal *vert;
576 
577  // Assume that the input data is triangles now.
578  // Try to merge as many polygons as possible
579  if (polySize > 3)
580  {
581  if (!tessMeshMergeConvexFaces( mesh, polySize ))
582  {
583  tess->outOfMemory = 1;
584  return;
585  }
586  }
587 
588  // Mark unused
589  for ( v = mesh->vHead.next; v != &mesh->vHead; v = v->next )
590  v->n = TESS_UNDEF;
591 
592  // Create unique IDs for all vertices and faces.
593  for ( f = mesh->fHead.next; f != &mesh->fHead; f = f->next )
594  {
595  f->n = TESS_UNDEF;
596  if( !f->inside ) continue;
597 
598  edge = f->anEdge;
599  faceVerts = 0;
600  do
601  {
602  v = edge->Org;
603  if ( v->n == TESS_UNDEF )
604  {
605  v->n = maxVertexCount;
606  maxVertexCount++;
607  }
608  faceVerts++;
609  edge = edge->Lnext;
610  }
611  while (edge != f->anEdge);
612 
613  if(faceVerts > polySize ) continue;
614 
615  f->n = maxFaceCount;
616  ++maxFaceCount;
617  }
618 
619  tess->elementCount = maxFaceCount;

677  edge = edge->Lnext;
678  }

685  if ( elementType == TESS_CONNECTED_POLYGONS )
686  {

688  do
689  {
690  *elements++ = GetNeighbourFace( edge );
691  edge = edge->Lnext;
692  }
693  while (edge != f->anEdge);
694  // Fill unused.
695  for (i = faceVerts; i < polySize; ++i)
696  *elements++ = TESS_UNDEF;
697  }

786 void tessAddContour( TESStesselator *tess, int size, const void* vertices,
787  int stride, int numVertices )
788 {
789  const unsigned char *src = (const unsigned char*)vertices;
790  TESShalfEdge *e;
791  int i;
792 
793  if ( tess->mesh == NULL )
794  tess->mesh = tessMeshNewMesh( &tess->alloc );
795  if ( tess->mesh == NULL ) {
796  tess->outOfMemory = 1;
797  return;
798  }
799 
800  if ( size < 2 )
801  size = 2;
802  if ( size > 3 )
803  size = 3;
804 
805  e = NULL;
806 
807  for( i = 0; i < numVertices; ++i )
808  {
809  const TESSreal* coords = (const TESSreal*)src;
810  src += stride;
811 
812  if( e == NULL ) {
813  /* Make a self-loop (one vertex, one edge). */
814  e = tessMeshMakeEdge( tess->mesh );
815  if ( e == NULL ) {
816  tess->outOfMemory = 1;
817  return;
818  }
819  if ( !tessMeshSplice( tess->mesh, e, e->Sym ) ) {
820  tess->outOfMemory = 1;
821  return;
822  }
823  } else {
824  /* Create a new vertex and edge which immediately follow e
825  * in the ordering around the left face.
826  */
827  if ( tessMeshSplitEdge( tess->mesh, e ) == NULL ) {
828  tess->outOfMemory = 1;
829  return;
830  }
831  e = e->Lnext;
832  }
833 
834  /* The new vertex is now e->Org. */
835  e->Org->coords[0] = coords[0];
836  e->Org->coords[1] = coords[1];
837  if ( size > 2 )
838  e->Org->coords[2] = coords[2];
839  else
840  e->Org->coords[2] = 0;
841  /* Store the insertion number so that the vertex can be later recognized. */
842  e->Org->idx = tess->vertexIndexCounter++;
843 
844  /* The winding of an edge says how the winding number changes as we
845  * cross from the edge''s right face to its left face. We add the
846  * vertices in such an order that a CCW contour will add +1 to
847  * the winding number of the region inside the contour.
848  */
849  e->winding = 1;
850  e->Sym->winding = -1;
851  }
852 }

854 int tessTesselate( TESStesselator *tess, int windingRule, int elementType,
855  int polySize, int vertexSize, const TESSreal* normal )
856 {
857  TESSmesh *mesh;
858  int rc = 1;
859 
860  if (tess->vertices != NULL) {
861  tess->alloc.memfree( tess->alloc.userData, tess->vertices );
862  tess->vertices = 0;
863  }
864  if (tess->elements != NULL) {
865  tess->alloc.memfree( tess->alloc.userData, tess->elements );
866  tess->elements = 0;
867  }
868  if (tess->vertexIndices != NULL) {
869  tess->alloc.memfree( tess->alloc.userData, tess->vertexIndices );
870  tess->vertexIndices = 0;
871  }
872 
873  tess->vertexIndexCounter = 0;
874 
875  if (normal)
876  {
877  tess->normal[0] = normal[0];
878  tess->normal[1] = normal[1];
879  tess->normal[2] = normal[2];
880  }
881 
882  tess->windingRule = windingRule;
883 
884  if (vertexSize < 2)
885  vertexSize = 2;
886  if (vertexSize > 3)
887  vertexSize = 3;
888 
889  if (setjmp(tess->env) != 0) {
890  /* come back here if out of memory */
891  return 0;
892  }
893 
894  if (!tess->mesh)
895  {
896  return 0;
897  }
898 
899  /* Determine the polygon normal and project vertices onto the plane
900  * of the polygon.
901  */
902  tessProjectPolygon( tess );
903 
904  /* tessComputeInterior( tess ) computes the planar arrangement specified
905  * by the given contours, and further subdivides this arrangement
906  * into regions. Each region is marked "inside" if it belongs
907  * to the polygon, according to the rule given by tess->windingRule.
908  * Each interior region is guaranteed be monotone.
909  */
910  if ( !tessComputeInterior( tess ) ) {
911  longjmp(tess->env,1); /* could've used a label */
912  }
913 
914  mesh = tess->mesh;
915 
916  /* If the user wants only the boundary contours, we throw away all edges
917  * except those which separate the interior from the exterior.
918  * Otherwise we tessellate all the regions marked "inside".
919  */
920  if (elementType == TESS_BOUNDARY_CONTOURS) {
921  rc = tessMeshSetWindingNumber( mesh, 1, TRUE );
922  } else {
923  rc = tessMeshTessellateInterior( mesh );
924  }
925  if (rc == 0) longjmp(tess->env,1); /* could've used a label */
926 
927  //tessMeshCheckMesh( mesh );
928 
929  if (elementType == TESS_BOUNDARY_CONTOURS) {
930  OutputContours( tess, mesh, vertexSize ); /* output contours */
931  }
932  else
933  {
934  OutputPolymesh( tess, mesh, elementType, polySize, vertexSize ); /* output polygons */
935  }
936 
937  tessMeshDeleteMesh( &tess->alloc, mesh );
938  tess->mesh = NULL;
939 
940  if (tess->outOfMemory)
941  return 0;
942  return 1;
943 }
