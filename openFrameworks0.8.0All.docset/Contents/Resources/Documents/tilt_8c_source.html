47 int freenect_update_tilt_state(freenect_device *dev)
48 {
49  freenect_context *ctx = dev->parent;
50 
51  uint8_t buf[10];
52  uint16_t ux, uy, uz;
53 
54  if(!(ctx->enabled_subdevices & FREENECT_DEVICE_MOTOR)){
55  //this is needed for OS X. without it the camera timesout within 2-3 seconds
56  #ifdef TARGET_OS_MAC
57  int ret = fnusb_control(&dev->usb_cam, 0xC0, 0x32, 0x0, 0x0, buf, 10);
58  #endif
59  return 0;
60  }
61 
62  int ret = fnusb_control(&dev->usb_motor, 0xC0, 0x32, 0x0, 0x0, buf, 10);
63  if (ret != 10) {
64  FN_ERROR("Error in accelerometer reading, libusb_control_transfer returned %d\n", ret);
65  return ret < 0 ? ret : -1;
66  }
67 
68  ux = ((uint16_t)buf[2] << 8) | buf[3];
69  uy = ((uint16_t)buf[4] << 8) | buf[5];
70  uz = ((uint16_t)buf[6] << 8) | buf[7];
71 
72  dev->raw_state.accelerometer_x = (int16_t)ux;
73  dev->raw_state.accelerometer_y = (int16_t)uy;
74  dev->raw_state.accelerometer_z = (int16_t)uz;
75  dev->raw_state.tilt_angle = (int8_t)buf[8];
76  dev->raw_state.tilt_status = (freenect_tilt_status_code)buf[9];
77 
78  return ret;
79 }
80 int freenect_set_tilt_degs(freenect_device *dev, double angle)
81 {
82  freenect_context *ctx = dev->parent;
83  if(!(ctx->enabled_subdevices & FREENECT_DEVICE_MOTOR))
84  return 0;
85  int ret;
86  uint8_t empty[0x1];
87 
88  angle = (angle<MIN_TILT_ANGLE) ? MIN_TILT_ANGLE : ((angle>MAX_TILT_ANGLE) ? MAX_TILT_ANGLE : angle);
89  angle = angle * 2;
90 
91  ret = fnusb_control(&dev->usb_motor, 0x40, 0x31, (uint16_t)angle, 0x0, empty, 0x0);
92  return ret;
