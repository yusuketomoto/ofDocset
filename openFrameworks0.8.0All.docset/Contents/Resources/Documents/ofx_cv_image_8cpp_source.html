738 void ofxCvImage::warpIntoMe( ofxCvImage& mom, const ofPoint src[4], const ofPoint dst[4] ){
739

753  CvPoint2D32f cvsrc[4];

756  cvSetZero( translate );

758  cvsrc[i].x = src[i].x;
759  cvsrc[i].y = src[i].y;
760  cvdst[i].x = dst[i].x;
761  cvdst[i].y = dst[i].y;

764  cvWarpPerspective( mom.getCvImage(), cvImage, translate);
765  flagImageChanged();
766  cvReleaseMat( &translate );
767 
768  } else {
