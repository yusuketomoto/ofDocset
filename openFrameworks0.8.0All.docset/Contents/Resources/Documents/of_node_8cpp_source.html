232  if(parent) upVector = upVector * ofMatrix4x4::getInverseOf(parent->getGlobalTransformMatrix());
233  ofVec3f zaxis = (getGlobalPosition() - lookAtPosition).normalized();
234  if (zaxis.length() > 0) {
235  ofVec3f xaxis = upVector.getCrossed(zaxis).normalized();
236  ofVec3f yaxis = zaxis.getCrossed(xaxis);
237 
238  ofMatrix4x4 m;
239  m._mat[0].set(xaxis.x, xaxis.y, xaxis.z, 0);
240  m._mat[1].set(yaxis.x, yaxis.y, yaxis.z, 0);
241  m._mat[2].set(zaxis.x, zaxis.y, zaxis.z, 0);
242 
243  setGlobalOrientation(m.getRotate());
244  }
