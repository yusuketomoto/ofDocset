409 template<typename PixelType>
410 static void saveImage(ofPixels_<PixelType> & pix, ofBuffer & buffer, ofImageFormat format, ofImageQualityType qualityLevel) {
411  // thanks to alvaro casinelli for the implementation
412 
413  ofInitFreeImage();
414 
415  if (pix.isAllocated() == false){
416  ofLogError("ofImage","saveImage(): couldn't save to ofBuffer, pixels are not allocated");
417  return;
418  }
419 
420  if(format==OF_IMAGE_FORMAT_JPEG && pix.getNumChannels()==4){
421  ofPixels pix3 = pix;
422  pix3.setNumChannels(3);
423  saveImage(pix3,buffer,format,qualityLevel);
424  return;
425  }
426 
427  #ifdef TARGET_LITTLE_ENDIAN
428  if(sizeof(PixelType) == 1) {
429  pix.swapRgb();
430  }
431  #endif
432 
433  FIBITMAP * bmp = getBmpFromPixels(pix);
434 
435  #ifdef TARGET_LITTLE_ENDIAN
436  if(sizeof(PixelType) == 1) {
437  pix.swapRgb();
438  }
439  #endif
440 
441  if (bmp) // bitmap successfully created
442  {
443  // (b) open a memory stream to compress the image onto mem_buffer:
444  //
445  FIMEMORY *hmem = FreeImage_OpenMemory();
446  // (c) encode and save the image to the memory (on dib FIBITMAP image):
447  //
448  if(FREE_IMAGE_FORMAT(format) == FIF_JPEG) {
449  int quality = JPEG_QUALITYSUPERB;
450  switch(qualityLevel) {
451  case OF_IMAGE_QUALITY_WORST: quality = JPEG_QUALITYBAD; break;
452  case OF_IMAGE_QUALITY_LOW: quality = JPEG_QUALITYAVERAGE; break;
453  case OF_IMAGE_QUALITY_MEDIUM: quality = JPEG_QUALITYNORMAL; break;
454  case OF_IMAGE_QUALITY_HIGH: quality = JPEG_QUALITYGOOD; break;
455  case OF_IMAGE_QUALITY_BEST: quality = JPEG_QUALITYSUPERB; break;
456  }
457  FreeImage_SaveToMemory(FIF_JPEG, bmp, hmem, quality);
458  }else{
459  FreeImage_SaveToMemory((FREE_IMAGE_FORMAT)format, bmp, hmem);
460  }
461 
462  /*
463 
464  NOTE: at this point, hmem contains the entire data in memory stored in fif format. the
465  amount of space used by the memory is equal to file_size:
466  long file_size = FreeImage_TellMemory(hmem);
467  but can also be retrieved by FreeImage_AcquireMemory that retrieves both the
468  length of the buffer, and the buffer memory address.
469  */
470  #ifdef TARGET_WIN32
471  DWORD size_in_bytes = 0;
472  #else
473  uint32_t size_in_bytes = 0;
474  #endif
475  // Save compressed data on mem_buffer
476  // note: FreeImage_AquireMemory allocates space for aux_mem_buffer):
477  //
478  unsigned char *mem_buffer = NULL;
479  if (!FreeImage_AcquireMemory(hmem, &mem_buffer, &size_in_bytes))
480  ofLogError("ofImage") << "saveImage(): couldn't save to ofBuffer, aquiring compressed image from memory failed";
481 
482  /*
483  Now, before closing the memory stream, copy the content of mem_buffer
484  to an auxiliary buffer
485  */
486 
487  buffer.set((char*)mem_buffer,size_in_bytes);
488 
489  // Finally, close the FIBITMAP object, or we will get a memory leak:
490  FreeImage_Unload(bmp);
491 
492  // Close the memory stream (otherwise we may get a memory leak).
493  FreeImage_CloseMemory(hmem);
494  }
495 }

930  int sh = ofGetViewportHeight(); // if we are in a FBO or other viewport, this fails: ofGetHeight();
931 
932 
933  #ifndef TARGET_OPENGLES
934 
935  if(ofIsVFlipped()){
936  _y = sh - _y;
937  _y -= _h; // top, bottom issues
938  }
939

941  glPixelStorei(GL_PACK_ALIGNMENT, 1);
942  glReadPixels(_x, _y, _w, _h, ofGetGlFormat(pixels), GL_UNSIGNED_BYTE, pixels.getPixels()); // read the memory....

947  PixelType * linea;

950  linea = pixels.getPixels() + i * sizeOfOneLineOfPixels;
951  lineb = pixels.getPixels() + (pixels.getHeight()-i-1) * sizeOfOneLineOfPixels;

953  memcpy(linea, lineb, sizeOfOneLineOfPixels);

1042  }
1043  }
1044  else if(ofGetOrientation() == OF_ORIENTATION_90_LEFT) {
