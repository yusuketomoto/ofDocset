42  int maxArea,
43  int nConsidered,
44  bool bFindHoles,
45  bool bUseApproximation) {
46 
47  // get width/height disregarding ROI
48  IplImage* ipltemp = input.getCvImage();
49  _width = ipltemp->width;
50  _height = ipltemp->height;
51 
52  reset();
53 
54  // opencv will clober the image it detects contours on, so we want to
55  // copy it into a copy before we detect contours. That copy is allocated
56  // if necessary (necessary = (a) not allocated or (b) wrong size)
57  // so be careful if you pass in different sized images to "findContours"
58  // there is a performance penalty, but we think there is not a memory leak
59  // to worry about better to create mutiple contour finders for different
60  // sizes, ie, if you are finding contours in a 640x480 image but also a
61  // 320x240 image better to make two ofxCvContourFinder objects then to use
62  // one, because you will get penalized less.
63 
64  inputCopy.setUseTexture(false);
65  if( inputCopy.getWidth() == 0 ) {
66  inputCopy.allocate( _width, _height );
67  } else if( inputCopy.getWidth() != _width || inputCopy.getHeight() != _height ) {
68  // reallocate to new size
69  inputCopy.clear();
70  inputCopy.allocate( _width, _height );
71  }
72 
73  inputCopy.setROI( input.getROI() );
74  inputCopy = input;
75 
76  CvSeq* contour_list = NULL;
77  contour_storage = cvCreateMemStorage( 1000 );
78  storage = cvCreateMemStorage( 1000 );
79 
80  CvContourRetrievalMode retrieve_mode
81  = (bFindHoles) ? CV_RETR_LIST : CV_RETR_EXTERNAL;
82  cvFindContours( inputCopy.getCvImage(), contour_storage, &contour_list,
83  sizeof(CvContour), retrieve_mode, bUseApproximation ? CV_CHAIN_APPROX_SIMPLE : CV_CHAIN_APPROX_NONE );
84  CvSeq* contour_ptr = contour_list;
85 
86  // put the contours from the linked list, into an array for sorting
87  while( (contour_ptr != NULL) ) {
88  float area = cvContourArea(contour_ptr, CV_WHOLE_SEQ, bFindHoles); // oriented=true for holes
89  if(bFindHoles && area < 0) { // areas can be non negative in the case of holes
90  area = fabs(area);
91  }
92  if((area > minArea) && (area < maxArea)) {
93  cvSeqBlobs.push_back(contour_ptr);
94  }
95  contour_ptr = contour_ptr->h_next;
96  }
97 
98 
99  // sort the pointers based on size
100  if( cvSeqBlobs.size() > 1 ) {
101  sort( cvSeqBlobs.begin(), cvSeqBlobs.end(), sort_carea_compare );
102  }
103 
104 
105  // now, we have cvSeqBlobs.size() contours, sorted by size in the array
106  // cvSeqBlobs let's get the data out and into our structures that we like
107  for( int i = 0; i < MIN(nConsidered, (int)cvSeqBlobs.size()); i++ ) {
108  blobs.push_back( ofxCvBlob() );
109  float area = cvContourArea( cvSeqBlobs[i], CV_WHOLE_SEQ, bFindHoles ); // oriented=true for holes
110  CvRect rect = cvBoundingRect( cvSeqBlobs[i], 0 );
111  cvMoments( cvSeqBlobs[i], myMoments );
112 
113  blobs[i].area = bFindHoles ? fabs(area) : area; // only return positive areas
114  blobs[i].length = cvArcLength(cvSeqBlobs[i]);
115  blobs[i].boundingRect.x = rect.x;
116  blobs[i].boundingRect.y = rect.y;
117  blobs[i].boundingRect.width = rect.width;
118  blobs[i].boundingRect.height = rect.height;
119  blobs[i].centroid.x = (myMoments->m10 / myMoments->m00);
120  blobs[i].centroid.y = (myMoments->m01 / myMoments->m00);
121 
122  if(bFindHoles) {
123  // for some reason, changing the orientation when looking for holes
124  // yields negative areas for non holes and positive areas for holes
125  //
126  // negating the value here works, even though it feels like a hack
127  blobs[i].hole = -area < 0 ? true : false; // negative area denotes a hole
128  }
129  else {
130  blobs[i].hole = false; // no holes
131  }
132 
133  // get the points for the blob:
134  CvPoint pt;
135  CvSeqReader reader;
136  cvStartReadSeq( cvSeqBlobs[i], &reader, 0 );
137 
138  for( int j=0; j < cvSeqBlobs[i]->total; j++ ) {
139  CV_READ_SEQ_ELEM( pt, reader );
140  blobs[i].pts.push_back( ofPoint((float)pt.x, (float)pt.y) );
141  }
142  blobs[i].nPts = blobs[i].pts.size();
143 
144  }
145 
146  nBlobs = blobs.size();
147 
148  // Free the storage memory.
149  // Warning: do this inside this function otherwise a strange memory leak
150  if( contour_storage != NULL ) { cvReleaseMemStorage(&contour_storage); }
151  if( storage != NULL ) { cvReleaseMemStorage(&storage); }
152 
153  return nBlobs;
154
