22  * A test caller provides access to a test case method
23  * on a test case class. Test callers are useful when
24  * you want to run an individual test or add it to a
25  * suite.
26  *
27  * Here is an example:
28  *
29  * class MathTest : public TestCase {
30  * ...
31  * public:
32  * void setUp ();
33  * void tearDown ();
34  *
35  * void testAdd ();
36  * void testSubtract ();
37  * };
38  *
39  * Test *MathTest::suite () {
40  * TestSuite *suite = new TestSuite;
41  *
42  * suite->addTest (new TestCaller<MathTest> ("testAdd", testAdd));
43  * return suite;
44  * }
45  *
46  * You can use a TestCaller to bind any test method on a TestCase
47  * class, as long as it returns accepts void and returns void.
48  *
49  * See TestCase
50  */
