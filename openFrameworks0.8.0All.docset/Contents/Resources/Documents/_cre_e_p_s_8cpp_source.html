302 bool
303 CAt::inheritLineDash( const CAt &A,
304  FILE *file,
305  bool gsave_written )
306 {
307  bool changed = false;
308  CAt tempA( A );
309 
310  // line dashes set with the enum-constant
311  if( (tempA.m_LineDashType != LD_KEEP ) &&
312  (tempA.m_LineDashType != m_LineDashType ||
313  tempA.m_LineDashFactor != m_LineDashFactor ||
314  tempA.m_LineThickness != m_LineThickness ||
315  tempA.m_LineCap != m_LineCap ) )
316  {
317  m_LineDashType = tempA.m_LineDashType;
318  if( tempA.m_LineDashFactor >= 1 ) {
319  m_LineDashFactor = tempA.m_LineDashFactor;
320  }
321  m_LineDashOffset = 0;
322 
323  float solid_dot = 0,
324  zero_dist = 0,
325  linethickness = tempA.m_LineThickness,
326  factor = m_LineDashFactor > 1 ? m_LineDashFactor : 1;
327  int linecap = tempA.m_LineCap >= 0 ? tempA.m_LineCap : m_LineCap;
328 
329  if( linethickness < 0 ) linethickness = m_LineThickness;
330  if( linecap < 0 ) linecap = (LINECAP)BUTT;
331 
332  switch( linecap )
333  {
334  case BUTT :
335  solid_dot = linethickness;
336  break;
337  case LC_ROUND :
338  zero_dist = linethickness;
339  break;
340  case SQUARE :
341  solid_dot = 0.0001;
342  zero_dist = linethickness;
343  break;
344  default : break;
345  }
346 
347  switch( m_LineDashType )
348  {
349  case SOLID :
350  sprintf( tempA.m_LineDashString, " " );
351  break;
352 
353  case DOT :
354  sprintf( tempA.m_LineDashString,
355  CreEPS_CC("%g %g"),
356  solid_dot,
357  factor * linethickness + zero_dist );
358  break;
359 
360  case DASH :
361  sprintf( tempA.m_LineDashString,
362  CreEPS_CC("%g %g"),
363  factor * linethickness + solid_dot - linethickness,
364  factor * linethickness + zero_dist );
365  break;
366 
367  case DOTDASH :
368  sprintf( tempA.m_LineDashString,
369  CreEPS_CC("%g %g %g %g"),
370  solid_dot,
371  0.5 * factor * linethickness + zero_dist,
372  factor * linethickness + solid_dot - linethickness ,
373  0.5 * factor * linethickness + zero_dist );
374  break;
375 
376  default : break;
377  }
378  }
379 
380  // inheriting the string
381  if( ( tempA.m_LineDashString[0] != 0 ||
382  tempA.m_LineDashOffset >= 0 )
383  &&
384  ( tempA.m_LineDashOffset != m_LineDashOffset ||
385  strncmp(tempA.m_LineDashString,
386  m_LineDashString,
387  CREEPS_MAX_DASH_STRING_LENGTH) )
388  )
389  {
390  memcpy( m_LineDashString,
391  tempA.m_LineDashString,
392  CREEPS_MAX_DASH_STRING_LENGTH );
393 
394  m_LineDashOffset = tempA.m_LineDashOffset;
395  if( file != NULL ) {
396  if( gsave_written == false ) {
397  fprintf( file, CreEPS_CC("x\n") );
398  }
399  fprintf( file,
400  CreEPS_CC("[%s] %g sd\n"),
401  m_LineDashString,
402  m_LineDashOffset >= 0 ? m_LineDashOffset : 0 );
403  }
404  changed = true;
405  }
406 
407  return changed;
408 }

563 CAt::inheritFillingPattern( const CAt &A,
564  FILE *file,
565  bool gsave_written )
566 {
567  bool changed = false;
568  float BoxWidth = 0,
569  BoxHeight = 0;
570 
571  if( ( A.m_FillingPattern != -1 &&
572  A.m_FillingPattern != m_FillingPattern )
573  || ( (A.m_FillingPattern == HEXDOT ||
574  A.m_FillingPattern == CHECKERBOARD ) &&
575  (m_ff1 != A.m_ff1 || m_ff2 != A.m_ff2 ) )
576  || ( (A.m_FillingPattern == STRIPE ) &&
577  (m_ff1 != A.m_ff1 || m_ff2 != A.m_ff2 ||
578  m_ff3 != A.m_ff3 )) )
579  {
580  m_FillingPattern = A.m_FillingPattern;
581  m_ff1 = A.m_ff1;
582  m_ff2 = A.m_ff2;
583  m_ff3 = A.m_ff3;
584 
585  if( file != NULL )
586  {
587  if( gsave_written == false ) fprintf( file, "x\n" );
588 
589  switch( m_FillingPattern ) {
590 
591  case HEXDOT :
592  // m_ff1 == distance between the balls
593  // m_ff2 == radius of the balls
594  BoxWidth = m_ff1;
595  BoxHeight = sqrtf(3)*m_ff1;
596  startFillingBox( file, BoxWidth, BoxHeight );
597  fprintf( file, CreEPS_CC("%g %g %g o f ")
598  CreEPS_CC("%g %g %g o f ")
599  CreEPS_CC("%g %g %g o f ")
600  CreEPS_CC("%g %g %g o f ")
601  CreEPS_CC("%g %g %g o f "),
602  0.0, 0.0, m_ff2,
603  BoxWidth, 0.0, m_ff2,
604  0.5 * BoxWidth, 0.5 * BoxHeight, m_ff2,
605  0.0, BoxHeight, m_ff2,
606  BoxWidth, BoxHeight, m_ff2 );
607  endFillingBox( file );
608  break;
609 
610  case CHECKERBOARD :
611  // m_ff1 == width of one field
612  // m_ff2 == height of one field
613  BoxWidth = 2*m_ff1;
614  BoxHeight = 2*m_ff2;
615  startFillingBox( file, BoxWidth, BoxHeight );
616  fprintf( file,
617  CreEPS_CC("0 0 %g %g rf %g %g %g %g rf"),
618  m_ff1, m_ff2, m_ff1, m_ff2, m_ff1, m_ff2 );
619  endFillingBox( file );
620  break;
621 
622  case STRIPE :
623  // m_ff1 == width
624  // m_ff2 == distance
625  // m_ff3 == angle
626  if( m_ff3 == 0 ) {
627  BoxWidth = 5,
628  BoxHeight = m_ff1 + m_ff2;
629  startFillingBox( file, BoxWidth, BoxHeight );
630  fprintf( file, CreEPS_CC("x %g sw ")
631  CreEPS_CC("0 %g m %g %g l s y "),
632  m_ff1, BoxHeight - 0.5*m_ff1,
633  BoxWidth, BoxHeight - 0.5*m_ff1 );
634  } else {
635  float diff = m_ff3 - 3.14159265358979323846 / 2;
636  if( diff < 0 ) diff = -diff;
637  // vertical lines
638  if( diff < 0.0087266 ) {
639  BoxWidth = m_ff1 + m_ff2;
640  BoxHeight = 5;
641  startFillingBox( file, BoxWidth, BoxHeight );
642  fprintf( file, CreEPS_CC("x %g sw ")
643  CreEPS_CC("%g 0 m %g %g l s y "),
644  m_ff1, 0.5*m_ff1, 0.5*m_ff1, BoxHeight );
645  // regular angle
646  } else {
647  BoxWidth = (m_ff1 + m_ff2) / sin(m_ff3);
648  BoxHeight = (m_ff1 + m_ff2) / cos(m_ff3);
649  if( BoxHeight > 0 ) {
650  startFillingBox( file, BoxWidth, BoxHeight );
651  fprintf( file,
652  CreEPS_CC("x %g sw ")
653  CreEPS_CC("0 0 m %g %g l s ")
654  CreEPS_CC("%g 0 m %g %g l s ")
655  CreEPS_CC("0 %g m %g %g l s ")
656  CreEPS_CC("y "),
657  m_ff1,
658  BoxWidth, BoxHeight,
659  -BoxWidth, BoxWidth, 2*BoxHeight,
660  -BoxHeight, 2*BoxWidth, BoxHeight );
661  } else {
662  BoxHeight = -BoxHeight;
663  startFillingBox( file, BoxWidth, BoxHeight );
664  fprintf( file,
665  CreEPS_CC("x %g sw ")
666  CreEPS_CC("0 %g m %g 0 l s ")
667  CreEPS_CC("0 %g m %g 0 l s ")
668  CreEPS_CC("%g %g m %g %g l s ")
669  CreEPS_CC("y "),
670  m_ff1,
671  BoxHeight, BoxWidth,
672  2*BoxHeight, 2*BoxWidth,
673  -BoxWidth , BoxHeight, BoxWidth, -BoxHeight );
674  }
675  }
676  }
677  endFillingBox( file );
678 
679  break;
680 
681  default : break;
682  }
683  fprintf( file,
