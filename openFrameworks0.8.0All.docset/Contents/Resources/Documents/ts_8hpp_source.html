458  int run_test_case( int expected_code, const string& _descr, F f)
459  {
460  double new_t = (double)cv::getTickCount(), dt;
461  if( test_case_idx < 0 )
462  {
463  test_case_idx = 0;
464  progress = 0;
465  dt = 0;
466  }
467  else
468  {
469  dt = (new_t - t)/(freq*1000);
470  t = new_t;
471  }
472  progress = update_progress(progress, test_case_idx, 0, dt);
473 
474  int errcount = 0;
475  bool thrown = false;
476  const char* descr = _descr.c_str() ? _descr.c_str() : "";
477 
478  try
479  {
480  f();
481  }
482  catch(const cv::Exception& e)
483  {
484  thrown = true;
485  if( e.code != expected_code )
486  {
487  ts->printf(TS::LOG, "%s (test case #%d): the error code %d is different from the expected %d\n",
488  descr, test_case_idx, e.code, expected_code);
489  errcount = 1;
490  }
491  }
492  catch(...)
493  {
494  thrown = true;
495  ts->printf(TS::LOG, "%s (test case #%d): unknown exception was thrown (the function has likely crashed)\n",
496  descr, test_case_idx);
497  errcount = 1;
498  }
499  if(!thrown)
500  {
501  ts->printf(TS::LOG, "%s (test case #%d): no expected exception was thrown\n",
502  descr, test_case_idx);
503  errcount = 1;
