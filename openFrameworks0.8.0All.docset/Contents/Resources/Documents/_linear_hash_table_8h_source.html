447  void split()
448  {
449  if (_split == _front)
450  {
451  _split = 0;
452  _front *= 2;
453  _buckets.reserve(_front*2);
454  }
455  Bucket tmp;
456  _buckets.push_back(tmp);
457  _buckets[_split].swap(tmp);
458  ++_split;
459  for (BucketIterator it = tmp.begin(); it != tmp.end(); ++it)
460  {
461  using std::swap;
462  std::size_t addr = bucketAddress(*it);
463  _buckets[addr].push_back(Value());
464  swap(*it, _buckets[addr].back());
465  }
466  }
