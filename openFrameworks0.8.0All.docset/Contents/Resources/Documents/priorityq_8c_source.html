55 /* The basic operations are insertion of a new key (pqInsert),
56 * and examination/extraction of a key whose value is minimum
57 * (pqMinimum/pqExtractMin). Deletion is also allowed (pqDelete);
58 * for this purpose pqInsert returns a "handle" which is supplied
59 * as the argument.
60 *
61 * An initial heap may be created efficiently by calling pqInsert
62 * repeatedly, then calling pqInit. In any case pqInit must be called
63 * before any operations other than pqInsert are used.
64 *
65 * If the heap is empty, pqMinimum/pqExtractMin will return a NULL key.
66 * This may also be tested with pqIsEmpty.
67 */
68

70 /* Since we support deletion the data structure is a little more
71 * complicated than an ordinary heap. "nodes" is the heap itself;
72 * active nodes are stored in the range 1..pq->size. When the
73 * heap exceeds its allocated size (pq->max), its size doubles.
74 * The children of node i are nodes 2i and 2i+1.
75 *
76 * Each node stores an index into an array "handles". Each handle
77 * stores a key, plus a pointer back to the node which currently
78 * represents that key (ie. nodes[handles[i].node].handle == i).
79 */

88 PriorityQHeap *pqHeapNewPriorityQ( TESSalloc* alloc, int size, int (*leq)(PQkey key1, PQkey key2) )
89 {

92 
93  pq->size = 0;
94  pq->max = size;
95  pq->nodes = (PQnode *)alloc->memalloc( alloc->userData, (size + 1) * sizeof(pq->nodes[0]) );
96  if (pq->nodes == NULL) {
97  alloc->memfree( alloc->userData, pq );
98  return NULL;
99  }
100 
101  pq->handles = (PQhandleElem *)alloc->memalloc( alloc->userData, (size + 1) * sizeof(pq->handles[0]) );
102  if (pq->handles == NULL) {
103  alloc->memfree( alloc->userData, pq->nodes );
104  alloc->memfree( alloc->userData, pq );
105  return NULL;
106  }
107 
108  pq->initialized = FALSE;
109  pq->freeList = 0;
110  pq->leq = leq;
111 
112  pq->nodes[1].handle = 1; /* so that Minimum() returns NULL */
113  pq->handles[1].key = NULL;
114  return pq;
115 }

126 static void FloatDown( PriorityQHeap *pq, int curr )
127 {
128  PQnode *n = pq->nodes;
129  PQhandleElem *h = pq->handles;
130  PQhandle hCurr, hChild;
131  int child;
132 
133  hCurr = n[curr].handle;
134  for( ;; ) {
135  child = curr << 1;
136  if( child < pq->size && LEQ( h[n[child+1].handle].key,
137  h[n[child].handle].key )) {
138  ++child;
139  }
140 
141  if(child > pq->max) continue;
142 
143  hChild = n[child].handle;
144  if( child > pq->size || LEQ( h[hCurr].key, h[hChild].key )) {
145  n[curr].handle = hCurr;
146  h[hCurr].node = curr;
147  break;
148  }
149  n[curr].handle = hChild;
150  h[hChild].node = curr;
151  curr = child;
152  }
153 }

156 static void FloatUp( PriorityQHeap *pq, int curr )
157 {
158  PQnode *n = pq->nodes;
159  PQhandleElem *h = pq->handles;
160  PQhandle hCurr, hParent;
161  int parent;
162 
163  hCurr = n[curr].handle;
164  for( ;; ) {
165  parent = curr >> 1;
166  hParent = n[parent].handle;
167  if( parent == 0 || LEQ( h[hParent].key, h[hCurr].key )) {
168  n[curr].handle = hCurr;
169  h[hCurr].node = curr;
170  break;
171  }
172  n[curr].handle = hParent;
173  h[hParent].node = curr;
174  curr = parent;
175  }
176 }

193 PQhandle pqHeapInsert( TESSalloc* alloc, PriorityQHeap *pq, PQkey keyNew )
194 {
195  int curr;
196  PQhandle free;
197 
198  curr = ++ pq->size;
199  if( (curr*2) > pq->max ) {
200  if (!alloc->memrealloc)
201  {
202  return INV_HANDLE;
203  }
204  else
205  {
206  PQnode *saveNodes= pq->nodes;
207  PQhandleElem *saveHandles= pq->handles;
208 
209  // If the heap overflows, double its size.
210  pq->max <<= 1;
211  pq->nodes = (PQnode *)alloc->memrealloc( alloc->userData, pq->nodes,
212  (size_t)((pq->max + 1) * sizeof( pq->nodes[0] )));
213  if (pq->nodes == NULL) {
214  pq->nodes = saveNodes; // restore ptr to free upon return
215  return INV_HANDLE;
216  }
217  pq->handles = (PQhandleElem *)alloc->memrealloc( alloc->userData, pq->handles,
218  (size_t) ((pq->max + 1) * sizeof( pq->handles[0] )));
219  if (pq->handles == NULL) {
220  pq->handles = saveHandles; // restore ptr to free upon return
221  return INV_HANDLE;
222  }
223  }
224  }
225 
226  if( pq->freeList == 0 ) {
227  free = curr;
228  } else {
229  free = pq->freeList;
230  pq->freeList = pq->handles[free].node;
231  }
232 
233  pq->nodes[curr].handle = free;
234  pq->handles[free].node = curr;
235  pq->handles[free].key = keyNew;
236 
237  if( pq->initialized ) {
238  FloatUp( pq, curr );
239  }
240  //assert(free != INV_HANDLE);

245 PQkey pqHeapExtractMin( PriorityQHeap *pq )
246 {
247  PQnode *n = pq->nodes;
248  PQhandleElem *h = pq->handles;
249  PQhandle hMin = n[1].handle;
250  PQkey min = h[hMin].key;
251 
252  if( pq->size > 0 ) {
253  n[1].handle = n[pq->size].handle;
254  h[n[1].handle].node = 1;
255 
256  h[hMin].key = NULL;
257  h[hMin].node = pq->freeList;
258  pq->freeList = hMin;
259 
260  if( -- pq->size > 0 ) {
261  FloatDown( pq, 1 );
262  }
263  }
264  return min;

267 /* really pqHeapDelete */
268 void pqHeapDelete( PriorityQHeap *pq, PQhandle hCurr )
269 {
270  PQnode *n = pq->nodes;
271  PQhandleElem *h = pq->handles;
272  int curr;
273 
274  if(!( hCurr >= 1 && hCurr <= pq->max && h[hCurr].key != NULL )) return;
275 
276  curr = h[hCurr].node;
277  n[curr].handle = n[pq->size].handle;
278  h[n[curr].handle].node = curr;
279 
280  if( curr <= -- pq->size ) {
281  if( curr <= 1 || LEQ( h[n[curr>>1].handle].key, h[n[curr].handle].key )) {
282  FloatDown( pq, curr );
283  } else {
284  FloatUp( pq, curr );
285  }
286  }
287  h[hCurr].key = NULL;
288  h[hCurr].node = pq->freeList;
289  pq->freeList = hCurr;
290 }

339 /* really tessPqSortInit */
340 int pqInit( TESSalloc* alloc, PriorityQ *pq )
341 {
342  PQkey **p, **r, **i, **j, *piv;
343  struct { PQkey **p, **r; } Stack[50], *top = Stack;
344  unsigned int seed = 2016473283;
345 
346  /* Create an array of indirect pointers to the keys, so that we
347  * the handles we have returned are still valid.
348  */
349  /*
350  pq->order = (PQkey **)memAlloc( (size_t)
351  (pq->size * sizeof(pq->order[0])) );
352  */
353  pq->order = (PQkey **)alloc->memalloc( alloc->userData,
354  (size_t)((pq->size+1) * sizeof(pq->order[0])) );
355  /* the previous line is a patch to compensate for the fact that IBM */
356  /* machines return a null on a malloc of zero bytes (unlike SGI), */
357  /* so we have to put in this defense to guard against a memory */
358  /* fault four lines down. from fossum@austin.ibm.com. */
359  if (pq->order == NULL) return 0;
360 
361  p = pq->order;
362  r = p + pq->size - 1;
363  for( piv = pq->keys, i = p; i <= r; ++piv, ++i ) {
364  *i = piv;
365  }
366 
367  /* Sort the indirect pointers in descending order,
368  * using randomized Quicksort
369  */
370  top->p = p; top->r = r; ++top;
371  while( --top >= Stack ) {
372  p = top->p;
373  r = top->r;
374  while( r > p + 10 ) {
375  seed = seed * 1539415821 + 1;
376  i = p + seed % (r - p + 1);
377  piv = *i;
378  *i = *p;
379  *p = piv;
380  i = p - 1;
381  j = r + 1;
382  do {
383  do { ++i; } while( GT( **i, *piv ));
384  do { --j; } while( LT( **j, *piv ));
385  Swap( i, j );
386  } while( i < j );
387  Swap( i, j ); /* Undo last swap */
388  if( i - p < r - j ) {
389  top->p = j+1; top->r = r; ++top;
390  r = i-1;
391  } else {
392  top->p = p; top->r = i-1; ++top;
393  p = j+1;
394  }
395  }
396  /* Insertion sort small lists */
397  for( i = p+1; i <= r; ++i ) {
398  piv = *i;
399  for( j = i; j > p && LT( **(j-1), *piv ); --j ) {
400  *j = *(j-1);
401  }
402  *j = piv;
403  }
404  }
405  pq->max = pq->size;
406  pq->initialized = TRUE;
407  pqHeapInit( pq->heap ); /* always succeeds */
408 
409 #ifndef NDEBUG
410  p = pq->order;
411  r = p + pq->size - 1;
412  for( i = p; i < r; ++i ) {
413  if(! LEQ( **(i+1), **i )) return 0;
414  }
415 #endif
416 
417  return 1;
418 }
