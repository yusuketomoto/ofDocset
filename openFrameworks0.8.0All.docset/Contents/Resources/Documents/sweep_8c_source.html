53 /*
54 * Invariants for the Edge Dictionary.
55 * - each pair of adjacent edges e2=Succ(e1) satisfies EdgeLeq(e1,e2)
56 * at any valid location of the sweep event
57 * - if EdgeLeq(e2,e1) as well (at any valid sweep event), then e1 and e2
58 * share a common endpoint
59 * - for each e, e->Dst has been processed, but not e->Org
60 * - each edge e satisfies VertLeq(e->Dst,event) && VertLeq(event,e->Org)
61 * where "event" is the current sweep line event.
62 * - no edge e has zero length
63 *
64 * Invariants for the Mesh (the processed portion).
65 * - the portion of the mesh left of the sweep line is a planar graph,
66 * ie. there is *some* way to embed it in the plane
67 * - no processed edge has zero length
68 * - no two processed vertices have identical coordinates
69 * - each "inside" region is monotone, ie. can be broken into two chains
70 * of monotonically increasing vertices according to VertLeq(v1,v2)
71 * - a non-invariant: these chains may intersect (very slightly)
72 *
73 * Invariants for the Sweep.
74 * - if none of the edges incident to the event vertex have an activeRegion
75 * (ie. none of these edges are in the edge dictionary), then the vertex
76 * has only right-going edges.
77 * - if an edge is marked "fixUpperEdge" (it is a temporary edge introduced
78 * by ConnectRightVertex), then it is the only right-going edge from
79 * its associated vertex. (This says that these edges exist only
80 * when it is necessary.)
81 */

96 static int EdgeLeq( TESStesselator *tess, ActiveRegion *reg1, ActiveRegion *reg2 )
97 /*
98 * Both edges must be directed from right to left (this is the canonical
99 * direction for the upper edge of each region).
100 *
101 * The strategy is to evaluate a "t" value for each edge at the
102 * current sweep line position, given by tess->event. The calculations
103 * are designed to be very stable, but of course they are not perfect.
104 *
105 * Special case: if both edge destinations are at the sweep event,
106 * we sort the edges by slope (they would otherwise compare equally).
107 */
108 {
109  TESSvertex *event = tess->event;
110  TESShalfEdge *e1, *e2;
111  TESSreal t1, t2;
112 
113  e1 = reg1->eUp;
114  e2 = reg2->eUp;
115 
116  if( e1->Dst == event ) {
117  if( e2->Dst == event ) {
118  /* Two edges right of the sweep line which meet at the sweep event.
119  * Sort them by slope.
120  */
121  if( VertLeq( e1->Org, e2->Org )) {
122  return EdgeSign( e2->Dst, e1->Org, e2->Org ) <= 0;
123  }
124  return EdgeSign( e1->Dst, e2->Org, e1->Org ) >= 0;
125  }
126  return EdgeSign( e2->Dst, event, e2->Org ) <= 0;
127  }
128  if( e2->Dst == event ) {
129  return EdgeSign( e1->Dst, event, e1->Org ) >= 0;
130  }
131 
132  /* General case - compute signed distance *from* e1, e2 to event */
133  t1 = EdgeEval( e1->Dst, event, e1->Org );
134  t2 = EdgeEval( e2->Dst, event, e2->Org );
135  return (t1 >= t2);
136 }

272 static TESShalfEdge *FinishLeftRegions( TESStesselator *tess,
273  ActiveRegion *regFirst, ActiveRegion *regLast )
274 /*
275 * We are given a vertex with one or more left-going edges. All affected
276 * edges should be in the edge dictionary. Starting at regFirst->eUp,
277 * we walk down deleting all regions where both edges have the same
278 * origin vOrg. At the same time we copy the "inside" flag from the
279 * active region to the face, since at this point each face will belong
280 * to at most one region (this was not necessarily true until this point
281 * in the sweep). The walk stops at the region above regLast; if regLast
282 * is NULL we walk as far as possible. At the same time we relink the
283 * mesh if necessary, so that the ordering of edges around vOrg is the
284 * same as in the dictionary.
285 */
286 {
287  ActiveRegion *reg, *regPrev;
288  TESShalfEdge *e, *ePrev;
289 
290  regPrev = regFirst;
291  ePrev = regFirst->eUp;
292  while( regPrev != regLast ) {
293  regPrev->fixUpperEdge = FALSE; /* placement was OK */
294  reg = RegionBelow( regPrev );
295  e = reg->eUp;
296  if( e->Org != ePrev->Org ) {
297  if( ! reg->fixUpperEdge ) {
298  /* Remove the last left-going edge. Even though there are no further
299  * edges in the dictionary with this origin, there may be further
300  * such edges in the mesh (if we are adding left edges to a vertex
301  * that has already been processed). Thus it is important to call
302  * FinishRegion rather than just DeleteRegion.
303  */
304  FinishRegion( tess, regPrev );
305  break;
306  }
307  /* If the edge below was a temporary edge introduced by
308  * ConnectRightVertex, now is the time to fix it.
309  */
310  e = tessMeshConnect( tess->mesh, ePrev->Lprev, e->Sym );
311  if (e == NULL) longjmp(tess->env,1);
312  if ( !FixUpperEdge( tess, reg, e ) ) longjmp(tess->env,1);
313  }
314 
315  /* Relink edges so that ePrev->Onext == e */
316  if( ePrev->Onext != e ) {
317  if ( !tessMeshSplice( tess->mesh, e->Oprev, e ) ) longjmp(tess->env,1);
318  if ( !tessMeshSplice( tess->mesh, ePrev, e ) ) longjmp(tess->env,1);
319  }
320  FinishRegion( tess, regPrev ); /* may change reg->eUp */
321  ePrev = reg->eUp;
322  regPrev = reg;
323  }
324  return ePrev;
325 }

328 static void AddRightEdges( TESStesselator *tess, ActiveRegion *regUp,
329  TESShalfEdge *eFirst, TESShalfEdge *eLast, TESShalfEdge *eTopLeft,
330  int cleanUp )
331 /*
332 * Purpose: insert right-going edges into the edge dictionary, and update
333 * winding numbers and mesh connectivity appropriately. All right-going
334 * edges share a common origin vOrg. Edges are inserted CCW starting at
335 * eFirst; the last edge inserted is eLast->Oprev. If vOrg has any
336 * left-going edges already processed, then eTopLeft must be the edge
337 * such that an imaginary upward vertical segment from vOrg would be
338 * contained between eTopLeft->Oprev and eTopLeft; otherwise eTopLeft
339 * should be NULL.
340 */
341 {
342  ActiveRegion *reg, *regPrev;
343  TESShalfEdge *e, *ePrev;
344  int firstTime = TRUE;
345 
346  /* Insert the new right-going edges in the dictionary */
347  e = eFirst;
348  do {
349  if(! VertLeq( e->Org, e->Dst )) return;
350  AddRegionBelow( tess, regUp, e->Sym );
351  e = e->Onext;
352  } while ( e != eLast );
353 
354  /* Walk *all* right-going edges from e->Org, in the dictionary order,
355  * updating the winding numbers of each region, and re-linking the mesh
356  * edges to match the dictionary ordering (if necessary).
357  */
358  if( eTopLeft == NULL ) {
359  eTopLeft = RegionBelow( regUp )->eUp->Rprev;
360  }
361  regPrev = regUp;
362  ePrev = eTopLeft;
363  for( ;; ) {
364  reg = RegionBelow( regPrev );
365  e = reg->eUp->Sym;
366  if( e->Org != ePrev->Org ) break;
367 
368  if( e->Onext != ePrev ) {
369  /* Unlink e from its current position, and relink below ePrev */
370  if ( !tessMeshSplice( tess->mesh, e->Oprev, e ) ) longjmp(tess->env,1);
371  if ( !tessMeshSplice( tess->mesh, ePrev->Oprev, e ) ) longjmp(tess->env,1);
372  }
373  /* Compute the winding number and "inside" flag for the new regions */
374  reg->windingNumber = regPrev->windingNumber - e->winding;
375  reg->inside = IsWindingInside( tess, reg->windingNumber );
376 
377  /* Check for two outgoing edges with same slope -- process these
378  * before any intersection tests (see example in tessComputeInterior).
379  */
380  regPrev->dirty = TRUE;
381  if( ! firstTime && CheckForRightSplice( tess, regPrev )) {
382  AddWinding( e, ePrev );
383  DeleteRegion( tess, regPrev );
384  if ( !tessMeshDelete( tess->mesh, ePrev ) ) longjmp(tess->env,1);
385  }
386  firstTime = FALSE;
387  regPrev = reg;
388  ePrev = e;
389  }
390  regPrev->dirty = TRUE;
391  if(!( regPrev->windingNumber - e->winding == reg->windingNumber )) return;
392 
393  if( cleanUp ) {
394  /* Check for intersections between newly adjacent edges. */
395  WalkDirtyRegions( tess, regPrev );
396  }
397 }

410 static void VertexWeights( TESSvertex *isect, TESSvertex *org, TESSvertex *dst,
411  TESSreal *weights )
412 /*
413 * Find some weights which describe how the intersection vertex is
414 * a linear combination of "org" and "dest". Each of the two edges
415 * which generated "isect" is allocated 50% of the weight; each edge
416 * splits the weight between its org and dst according to the
417 * relative distance to "isect".
418 */
419 {
420  TESSreal t1 = VertL1dist( org, isect );
421  TESSreal t2 = VertL1dist( dst, isect );
422 
423  weights[0] = (TESSreal)0.5 * t2 / (t1 + t2);
424  weights[1] = (TESSreal)0.5 * t1 / (t1 + t2);
425  isect->coords[0] += weights[0]*org->coords[0] + weights[1]*dst->coords[0];
426  isect->coords[1] += weights[0]*org->coords[1] + weights[1]*dst->coords[1];
427  isect->coords[2] += weights[0]*org->coords[2] + weights[1]*dst->coords[2];
428 }

448 static int CheckForRightSplice( TESStesselator *tess, ActiveRegion *regUp )
449 /*
450 * Check the upper and lower edge of "regUp", to make sure that the
451 * eUp->Org is above eLo, or eLo->Org is below eUp (depending on which
452 * origin is leftmost).
453 *
454 * The main purpose is to splice right-going edges with the same
455 * dest vertex and nearly identical slopes (ie. we can't distinguish
456 * the slopes numerically). However the splicing can also help us
457 * to recover from numerical errors. For example, suppose at one
458 * point we checked eUp and eLo, and decided that eUp->Org is barely
459 * above eLo. Then later, we split eLo into two edges (eg. from
460 * a splice operation like this one). This can change the result of
461 * our test so that now eUp->Org is incident to eLo, or barely below it.
462 * We must correct this condition to maintain the dictionary invariants.
463 *
464 * One possibility is to check these edges for intersection again
465 * (ie. CheckForIntersect). This is what we do if possible. However
466 * CheckForIntersect requires that tess->event lies between eUp and eLo,
467 * so that it has something to fall back on when the intersection
468 * calculation gives us an unusable answer. So, for those cases where
469 * we can't check for intersection, this routine fixes the problem
470 * by just splicing the offending vertex into the other edge.
471 * This is a guaranteed solution, no matter how degenerate things get.
472 * Basically this is a combinatorial solution to a numerical problem.
473 */
474 {
475  ActiveRegion *regLo = RegionBelow(regUp);
476  TESShalfEdge *eUp = regUp->eUp;
477  TESShalfEdge *eLo = regLo->eUp;
478 
479  if( VertLeq( eUp->Org, eLo->Org )) {
480  if( EdgeSign( eLo->Dst, eUp->Org, eLo->Org ) > 0 ) return FALSE;
481 
482  /* eUp->Org appears to be below eLo */
483  if( ! VertEq( eUp->Org, eLo->Org )) {
484  /* Splice eUp->Org into eLo */
485  if ( tessMeshSplitEdge( tess->mesh, eLo->Sym ) == NULL) longjmp(tess->env,1);
486  if ( !tessMeshSplice( tess->mesh, eUp, eLo->Oprev ) ) longjmp(tess->env,1);
487  regUp->dirty = regLo->dirty = TRUE;
488 
489  } else if( eUp->Org != eLo->Org ) {
490  /* merge the two vertices, discarding eUp->Org */
491  pqDelete( tess->pq, eUp->Org->pqHandle );
492  SpliceMergeVertices( tess, eLo->Oprev, eUp );
493  }
494  } else {
495  if( EdgeSign( eUp->Dst, eLo->Org, eUp->Org ) < 0 ) return FALSE;
496 
497  /* eLo->Org appears to be above eUp, so splice eLo->Org into eUp */
498  RegionAbove(regUp)->dirty = regUp->dirty = TRUE;
499  if (tessMeshSplitEdge( tess->mesh, eUp->Sym ) == NULL) longjmp(tess->env,1);
500  if ( !tessMeshSplice( tess->mesh, eLo->Oprev, eUp ) ) longjmp(tess->env,1);
501  }
502  return TRUE;
503 }

505 static int CheckForLeftSplice( TESStesselator *tess, ActiveRegion *regUp )
506 /*
507 * Check the upper and lower edge of "regUp", to make sure that the
508 * eUp->Dst is above eLo, or eLo->Dst is below eUp (depending on which
509 * destination is rightmost).
510 *
511 * Theoretically, this should always be true. However, splitting an edge
512 * into two pieces can change the results of previous tests. For example,
513 * suppose at one point we checked eUp and eLo, and decided that eUp->Dst
514 * is barely above eLo. Then later, we split eLo into two edges (eg. from
515 * a splice operation like this one). This can change the result of
516 * the test so that now eUp->Dst is incident to eLo, or barely below it.
517 * We must correct this condition to maintain the dictionary invariants
518 * (otherwise new edges might get inserted in the wrong place in the
519 * dictionary, and bad stuff will happen).
520 *
521 * We fix the problem by just splicing the offending vertex into the
522 * other edge.
523 */
524 {
525  ActiveRegion *regLo = RegionBelow(regUp);
526  TESShalfEdge *eUp = regUp->eUp;
527  TESShalfEdge *eLo = regLo->eUp;
528  TESShalfEdge *e;
529 
530  if( VertEq( eUp->Dst, eLo->Dst )) return FALSE;
531 
532  if( VertLeq( eUp->Dst, eLo->Dst )) {
533  if( EdgeSign( eUp->Dst, eLo->Dst, eUp->Org ) < 0 ) return FALSE;
534 
535  /* eLo->Dst is above eUp, so splice eLo->Dst into eUp */
536  RegionAbove(regUp)->dirty = regUp->dirty = TRUE;
537  e = tessMeshSplitEdge( tess->mesh, eUp );
538  if (e == NULL) longjmp(tess->env,1);
539  if ( !tessMeshSplice( tess->mesh, eLo->Sym, e ) ) longjmp(tess->env,1);
540  e->Lface->inside = regUp->inside;
541  } else {
542  if( EdgeSign( eLo->Dst, eUp->Dst, eLo->Org ) > 0 ) return FALSE;
543 
544  /* eUp->Dst is below eLo, so splice eUp->Dst into eLo */
545  regUp->dirty = regLo->dirty = TRUE;
546  e = tessMeshSplitEdge( tess->mesh, eLo );
547  if (e == NULL) longjmp(tess->env,1);
548  if ( !tessMeshSplice( tess->mesh, eUp->Lnext, eLo->Sym ) ) longjmp(tess->env,1);
549  e->Rface->inside = regUp->inside;
550  }
551  return TRUE;
552 }

555 static int CheckForIntersect( TESStesselator *tess, ActiveRegion *regUp )
556 /*
557 * Check the upper and lower edges of the given region to see if
558 * they intersect. If so, create the intersection and add it
559 * to the data structures.
560 *
561 * Returns TRUE if adding the new intersection resulted in a recursive
562 * call to AddRightEdges(); in this case all "dirty" regions have been
563 * checked for intersections, and possibly regUp has been deleted.
564 */
565 {
566  ActiveRegion *regLo = RegionBelow(regUp);
567  TESShalfEdge *eUp = regUp->eUp;
568  TESShalfEdge *eLo = regLo->eUp;
569  TESSvertex *orgUp = eUp->Org;
570  TESSvertex *orgLo = eLo->Org;
571  TESSvertex *dstUp = eUp->Dst;
572  TESSvertex *dstLo = eLo->Dst;
573  TESSreal tMinUp, tMaxLo;
574  TESSvertex isect, *orgMin;
575  TESShalfEdge *e;
576 
577  if( VertEq( dstLo, dstUp )) return FALSE;
578  if( EdgeSign( dstUp, tess->event, orgUp ) > 0 ) return FALSE;
579  if( EdgeSign( dstLo, tess->event, orgLo ) < 0 ) return FALSE;
580  if(!( orgUp != tess->event && orgLo != tess->event )) return FALSE;
581  if(!( ! regUp->fixUpperEdge && ! regLo->fixUpperEdge )) return FALSE;
582 
583  if( orgUp == orgLo ) return FALSE; /* right endpoints are the same */
584 
585  tMinUp = MIN( orgUp->t, dstUp->t );
586  tMaxLo = MAX( orgLo->t, dstLo->t );
587  if( tMinUp > tMaxLo ) return FALSE; /* t ranges do not overlap */
588 
589  if( VertLeq( orgUp, orgLo )) {
590  if( EdgeSign( dstLo, orgUp, orgLo ) > 0 ) return FALSE;
591  } else {
592  if( EdgeSign( dstUp, orgLo, orgUp ) < 0 ) return FALSE;
593  }
594 
595  /* At this point the edges intersect, at least marginally */
596  DebugEvent( tess );
597 
598  tesedgeIntersect( dstUp, orgUp, dstLo, orgLo, &isect );
599  /* The following properties are guaranteed: */
600  if(!( MIN( orgUp->t, dstUp->t ) <= isect.t )) return FALSE;
601  if(!( isect.t <= MAX( orgLo->t, dstLo->t ))) return FALSE;
602  if(!( MIN( dstLo->s, dstUp->s ) <= isect.s )) return FALSE;
603  if(!( isect.s <= MAX( orgLo->s, orgUp->s ))) return FALSE;
604 
605  if( VertLeq( &isect, tess->event )) {
606  /* The intersection point lies slightly to the left of the sweep line,
607  * so move it until it''s slightly to the right of the sweep line.
608  * (If we had perfect numerical precision, this would never happen
609  * in the first place). The easiest and safest thing to do is
610  * replace the intersection by tess->event.
611  */
612  isect.s = tess->event->s;
613  isect.t = tess->event->t;
614  }
615  /* Similarly, if the computed intersection lies to the right of the
616  * rightmost origin (which should rarely happen), it can cause
617  * unbelievable inefficiency on sufficiently degenerate inputs.
618  * (If you have the test program, try running test54.d with the
619  * "X zoom" option turned on).
620  */
621  orgMin = VertLeq( orgUp, orgLo ) ? orgUp : orgLo;
622  if( VertLeq( orgMin, &isect )) {
623  isect.s = orgMin->s;
624  isect.t = orgMin->t;
625  }
626 
627  if( VertEq( &isect, orgUp ) || VertEq( &isect, orgLo )) {
628  /* Easy case -- intersection at one of the right endpoints */
629  (void) CheckForRightSplice( tess, regUp );
630  return FALSE;
631  }
632 
633  if( (! VertEq( dstUp, tess->event )
634  && EdgeSign( dstUp, tess->event, &isect ) >= 0)
635  || (! VertEq( dstLo, tess->event )
636  && EdgeSign( dstLo, tess->event, &isect ) <= 0 ))
637  {
638  /* Very unusual -- the new upper or lower edge would pass on the
639  * wrong side of the sweep event, or through it. This can happen
640  * due to very small numerical errors in the intersection calculation.
641  */
642  if( dstLo == tess->event ) {
643  /* Splice dstLo into eUp, and process the new region(s) */
644  if (tessMeshSplitEdge( tess->mesh, eUp->Sym ) == NULL) longjmp(tess->env,1);
645  if ( !tessMeshSplice( tess->mesh, eLo->Sym, eUp ) ) longjmp(tess->env,1);
646  regUp = TopLeftRegion( tess, regUp );
647  if (regUp == NULL) longjmp(tess->env,1);
648  eUp = RegionBelow(regUp)->eUp;
649  FinishLeftRegions( tess, RegionBelow(regUp), regLo );
650  AddRightEdges( tess, regUp, eUp->Oprev, eUp, eUp, TRUE );
651  return TRUE;
652  }
653  if( dstUp == tess->event ) {
654  /* Splice dstUp into eLo, and process the new region(s) */
655  if (tessMeshSplitEdge( tess->mesh, eLo->Sym ) == NULL) longjmp(tess->env,1);
656  if ( !tessMeshSplice( tess->mesh, eUp->Lnext, eLo->Oprev ) ) longjmp(tess->env,1);
657  regLo = regUp;
658  regUp = TopRightRegion( regUp );
659  e = RegionBelow(regUp)->eUp->Rprev;
660  regLo->eUp = eLo->Oprev;
661  eLo = FinishLeftRegions( tess, regLo, NULL );
662  AddRightEdges( tess, regUp, eLo->Onext, eUp->Rprev, e, TRUE );
663  return TRUE;
664  }
665  /* Special case: called from ConnectRightVertex. If either
666  * edge passes on the wrong side of tess->event, split it
667  * (and wait for ConnectRightVertex to splice it appropriately).
668  */
669  if( EdgeSign( dstUp, tess->event, &isect ) >= 0 ) {
670  RegionAbove(regUp)->dirty = regUp->dirty = TRUE;
671  if (tessMeshSplitEdge( tess->mesh, eUp->Sym ) == NULL) longjmp(tess->env,1);
672  eUp->Org->s = tess->event->s;
673  eUp->Org->t = tess->event->t;
674  }
675  if( EdgeSign( dstLo, tess->event, &isect ) <= 0 ) {
676  regUp->dirty = regLo->dirty = TRUE;
677  if (tessMeshSplitEdge( tess->mesh, eLo->Sym ) == NULL) longjmp(tess->env,1);
678  eLo->Org->s = tess->event->s;
679  eLo->Org->t = tess->event->t;
680  }
681  /* leave the rest for ConnectRightVertex */
682  return FALSE;
683  }
684 
685  /* General case -- split both edges, splice into new vertex.
686  * When we do the splice operation, the order of the arguments is
687  * arbitrary as far as correctness goes. However, when the operation
688  * creates a new face, the work done is proportional to the size of
689  * the new face. We expect the faces in the processed part of
690  * the mesh (ie. eUp->Lface) to be smaller than the faces in the
691  * unprocessed original contours (which will be eLo->Oprev->Lface).
692  */
693  if (tessMeshSplitEdge( tess->mesh, eUp->Sym ) == NULL) longjmp(tess->env,1);
694  if (tessMeshSplitEdge( tess->mesh, eLo->Sym ) == NULL) longjmp(tess->env,1);
695  if ( !tessMeshSplice( tess->mesh, eLo->Oprev, eUp ) ) longjmp(tess->env,1);
696  eUp->Org->s = isect.s;
697  eUp->Org->t = isect.t;
698  eUp->Org->pqHandle = pqInsert( &tess->alloc, tess->pq, eUp->Org );
699  if (eUp->Org->pqHandle == INV_HANDLE) {
700  pqDeletePriorityQ( &tess->alloc, tess->pq );
701  tess->pq = NULL;
702  longjmp(tess->env,1);
703  }
704  GetIntersectData( tess, eUp->Org, orgUp, dstUp, orgLo, dstLo );
705  RegionAbove(regUp)->dirty = regUp->dirty = regLo->dirty = TRUE;
706  return FALSE;
707 }

709 static void WalkDirtyRegions( TESStesselator *tess, ActiveRegion *regUp )
710 /*
711 * When the upper or lower edge of any region changes, the region is
712 * marked "dirty". This routine walks through all the dirty regions
713 * and makes sure that the dictionary invariants are satisfied
714 * (see the comments at the beginning of this file). Of course
715 * new dirty regions can be created as we make changes to restore
716 * the invariants.
717 */
718 {
719  ActiveRegion *regLo = RegionBelow(regUp);
720  TESShalfEdge *eUp, *eLo;
721 
722  for( ;; ) {
723  /* Find the lowest dirty region (we walk from the bottom up). */
724  while( regLo->dirty ) {
725  regUp = regLo;
726  regLo = RegionBelow(regLo);
727  }
728  if( ! regUp->dirty ) {
729  regLo = regUp;
730  regUp = RegionAbove( regUp );
731  if( regUp == NULL || ! regUp->dirty ) {
732  /* We've walked all the dirty regions */
733  return;
734  }
735  }
736  regUp->dirty = FALSE;
737  eUp = regUp->eUp;
738  eLo = regLo->eUp;
739 
740  if( eUp->Dst != eLo->Dst ) {
741  /* Check that the edge ordering is obeyed at the Dst vertices. */
742  if( CheckForLeftSplice( tess, regUp )) {
743 
744  /* If the upper or lower edge was marked fixUpperEdge, then
745  * we no longer need it (since these edges are needed only for
746  * vertices which otherwise have no right-going edges).
747  */
748  if( regLo->fixUpperEdge ) {
749  DeleteRegion( tess, regLo );
750  if ( !tessMeshDelete( tess->mesh, eLo ) ) longjmp(tess->env,1);
751  regLo = RegionBelow( regUp );
752  eLo = regLo->eUp;
753  } else if( regUp->fixUpperEdge ) {
754  DeleteRegion( tess, regUp );
755  if ( !tessMeshDelete( tess->mesh, eUp ) ) longjmp(tess->env,1);
756  regUp = RegionAbove( regLo );
757  eUp = regUp->eUp;
758  }
759  }
760  }
761  if( eUp->Org != eLo->Org ) {
762  if( eUp->Dst != eLo->Dst
763  && ! regUp->fixUpperEdge && ! regLo->fixUpperEdge
764  && (eUp->Dst == tess->event || eLo->Dst == tess->event) )
765  {
766  /* When all else fails in CheckForIntersect(), it uses tess->event
767  * as the intersection location. To make this possible, it requires
768  * that tess->event lie between the upper and lower edges, and also
769  * that neither of these is marked fixUpperEdge (since in the worst
770  * case it might splice one of these edges into tess->event, and
771  * violate the invariant that fixable edges are the only right-going
772  * edge from their associated vertex).
773  */
774  if( CheckForIntersect( tess, regUp )) {
775  /* WalkDirtyRegions() was called recursively; we're done */
776  return;
777  }
778  } else {
779  /* Even though we can't use CheckForIntersect(), the Org vertices
780  * may violate the dictionary edge ordering. Check and correct this.
781  */
782  (void) CheckForRightSplice( tess, regUp );
783  }
784  }
785  if( eUp->Org == eLo->Org && eUp->Dst == eLo->Dst ) {
786  /* A degenerate loop consisting of only two edges -- delete it. */
787  AddWinding( eLo, eUp );
788  DeleteRegion( tess, regUp );
789  if ( !tessMeshDelete( tess->mesh, eUp ) ) longjmp(tess->env,1);
790  regUp = RegionAbove( regLo );
791  }
792  }
793 }

796 static void ConnectRightVertex( TESStesselator *tess, ActiveRegion *regUp,
797  TESShalfEdge *eBottomLeft )
798 /*
799 * Purpose: connect a "right" vertex vEvent (one where all edges go left)
800 * to the unprocessed portion of the mesh. Since there are no right-going
801 * edges, two regions (one above vEvent and one below) are being merged
802 * into one. "regUp" is the upper of these two regions.
803 *
804 * There are two reasons for doing this (adding a right-going edge):
805 * - if the two regions being merged are "inside", we must add an edge
806 * to keep them separated (the combined region would not be monotone).
807 * - in any case, we must leave some record of vEvent in the dictionary,
808 * so that we can merge vEvent with features that we have not seen yet.
809 * For example, maybe there is a vertical edge which passes just to
810 * the right of vEvent; we would like to splice vEvent into this edge.
811 *
812 * However, we don't want to connect vEvent to just any vertex. We don''t
813 * want the new edge to cross any other edges; otherwise we will create
814 * intersection vertices even when the input data had no self-intersections.
815 * (This is a bad thing; if the user's input data has no intersections,
816 * we don't want to generate any false intersections ourselves.)
817 *
818 * Our eventual goal is to connect vEvent to the leftmost unprocessed
819 * vertex of the combined region (the union of regUp and regLo).
820 * But because of unseen vertices with all right-going edges, and also
821 * new vertices which may be created by edge intersections, we don''t
822 * know where that leftmost unprocessed vertex is. In the meantime, we
823 * connect vEvent to the closest vertex of either chain, and mark the region
824 * as "fixUpperEdge". This flag says to delete and reconnect this edge
825 * to the next processed vertex on the boundary of the combined region.
826 * Quite possibly the vertex we connected to will turn out to be the
827 * closest one, in which case we won''t need to make any changes.
828 */
829 {
830  TESShalfEdge *eNew;
831  TESShalfEdge *eTopLeft = eBottomLeft->Onext;
832  ActiveRegion *regLo = RegionBelow(regUp);
833  TESShalfEdge *eUp = regUp->eUp;
834  TESShalfEdge *eLo = regLo->eUp;
835  int degenerate = FALSE;
836 
837  if( eUp->Dst != eLo->Dst ) {
838  (void) CheckForIntersect( tess, regUp );
839  }
840 
841  /* Possible new degeneracies: upper or lower edge of regUp may pass
842  * through vEvent, or may coincide with new intersection vertex
843  */
844  if( VertEq( eUp->Org, tess->event )) {
845  if ( !tessMeshSplice( tess->mesh, eTopLeft->Oprev, eUp ) ) longjmp(tess->env,1);
846  regUp = TopLeftRegion( tess, regUp );
847  if (regUp == NULL) longjmp(tess->env,1);
848  eTopLeft = RegionBelow( regUp )->eUp;
849  FinishLeftRegions( tess, RegionBelow(regUp), regLo );
850  degenerate = TRUE;
851  }
852  if( VertEq( eLo->Org, tess->event )) {
853  if ( !tessMeshSplice( tess->mesh, eBottomLeft, eLo->Oprev ) ) longjmp(tess->env,1);
854  eBottomLeft = FinishLeftRegions( tess, regLo, NULL );
855  degenerate = TRUE;
856  }
857  if( degenerate ) {
858  AddRightEdges( tess, regUp, eBottomLeft->Onext, eTopLeft, eTopLeft, TRUE );
859  return;
860  }
861 
862  /* Non-degenerate situation -- need to add a temporary, fixable edge.
863  * Connect to the closer of eLo->Org, eUp->Org.
864  */
865  if( VertLeq( eLo->Org, eUp->Org )) {
866  eNew = eLo->Oprev;
867  } else {
868  eNew = eUp;
869  }
870  eNew = tessMeshConnect( tess->mesh, eBottomLeft->Lprev, eNew );
871  if (eNew == NULL) longjmp(tess->env,1);
872 
873  /* Prevent cleanup, otherwise eNew might disappear before we've even
874  * had a chance to mark it as a temporary edge.
875  */
876  AddRightEdges( tess, regUp, eNew, eNew->Onext, eNew->Onext, FALSE );
877  eNew->Sym->activeRegion->fixUpperEdge = TRUE;
878  WalkDirtyRegions( tess, regUp );
879 }

881 /* Because vertices at exactly the same location are merged together
882 * before we process the sweep event, some degenerate cases can't occur.
883 * However if someone eventually makes the modifications required to
884 * merge features which are close together, the cases below marked
885 * TOLERANCE_NONZERO will be useful. They were debugged before the
886 * code to merge identical vertices in the main loop was added.
887 */
888 #define TOLERANCE_NONZERO FALSE

890 static void ConnectLeftDegenerate( TESStesselator *tess,
891  ActiveRegion *regUp, TESSvertex *vEvent )
892 /*
893 * The event vertex lies exacty on an already-processed edge or vertex.
894 * Adding the new vertex involves splicing it into the already-processed
895 * part of the mesh.
896 */
897 {
898  TESShalfEdge *e, *eTopLeft, *eTopRight, *eLast;
899  ActiveRegion *reg;
900 
901  e = regUp->eUp;
902  if( VertEq( e->Org, vEvent )) {
903  /* e->Org is an unprocessed vertex - just combine them, and wait
904  * for e->Org to be pulled from the queue
905  */
906  if(! TOLERANCE_NONZERO ) return;
907  SpliceMergeVertices( tess, e, vEvent->anEdge );
908  return;
909  }
910 
911  if( ! VertEq( e->Dst, vEvent )) {
912  /* General case -- splice vEvent into edge e which passes through it */
913  if (tessMeshSplitEdge( tess->mesh, e->Sym ) == NULL) longjmp(tess->env,1);
914  if( regUp->fixUpperEdge ) {
915  /* This edge was fixable -- delete unused portion of original edge */
916  if ( !tessMeshDelete( tess->mesh, e->Onext ) ) longjmp(tess->env,1);
917  regUp->fixUpperEdge = FALSE;
918  }
919  if ( !tessMeshSplice( tess->mesh, vEvent->anEdge, e ) ) longjmp(tess->env,1);
920  SweepEvent( tess, vEvent ); /* recurse */
921  return;
922  }
923 
924  /* vEvent coincides with e->Dst, which has already been processed.
925  * Splice in the additional right-going edges.
926  */
927  if(! TOLERANCE_NONZERO ) return;
928  regUp = TopRightRegion( regUp );
929  reg = RegionBelow( regUp );
930  eTopRight = reg->eUp->Sym;
931  eTopLeft = eLast = eTopRight->Onext;
932  if( reg->fixUpperEdge ) {
933  /* Here e->Dst has only a single fixable edge going right.
934  * We can delete it since now we have some real right-going edges.
935  */
936  if( eTopLeft == eTopRight ) return; /* there are some left edges too */
937  DeleteRegion( tess, reg );
938  if ( !tessMeshDelete( tess->mesh, eTopRight ) ) longjmp(tess->env,1);
939  eTopRight = eTopLeft->Oprev;
940  }
941  if ( !tessMeshSplice( tess->mesh, vEvent->anEdge, eTopRight ) ) longjmp(tess->env,1);
942  if( ! EdgeGoesLeft( eTopLeft )) {
943  /* e->Dst had no left-going edges -- indicate this to AddRightEdges() */
944  eTopLeft = NULL;
945  }
946  AddRightEdges( tess, regUp, eTopRight->Onext, eLast, eTopLeft, TRUE );
947 }

950 static void ConnectLeftVertex( TESStesselator *tess, TESSvertex *vEvent )
951 /*
952 * Purpose: connect a "left" vertex (one where both edges go right)
953 * to the processed portion of the mesh. Let R be the active region
954 * containing vEvent, and let U and L be the upper and lower edge
955 * chains of R. There are two possibilities:
956 *
957 * - the normal case: split R into two regions, by connecting vEvent to
958 * the rightmost vertex of U or L lying to the left of the sweep line
959 *
960 * - the degenerate case: if vEvent is close enough to U or L, we
961 * merge vEvent into that edge chain. The subcases are:
962 * - merging with the rightmost vertex of U or L
963 * - merging with the active edge of U or L
964 * - merging with an already-processed portion of U or L
965 */
966 {
967  ActiveRegion *regUp, *regLo, *reg;
968  TESShalfEdge *eUp, *eLo, *eNew;
969  ActiveRegion tmp;
970 
971  /* assert( vEvent->anEdge->Onext->Onext == vEvent->anEdge ); */
972 
973  /* Get a pointer to the active region containing vEvent */
974  tmp.eUp = vEvent->anEdge->Sym;
975  /* __GL_DICTLISTKEY */ /* tessDictListSearch */
976  regUp = (ActiveRegion *)dictKey( dictSearch( tess->dict, &tmp ));
977  regLo = RegionBelow( regUp );
978  if(!regUp) return;
979  if(!regLo) return;
980 
981  eUp = regUp->eUp;
982  eLo = regLo->eUp;
983 
984  /* Try merging with U or L first */
985  if( EdgeSign( eUp->Dst, vEvent, eUp->Org ) == 0 ) {
986  ConnectLeftDegenerate( tess, regUp, vEvent );
987  return;
988  }
989 
990  /* Connect vEvent to rightmost processed vertex of either chain.
991  * e->Dst is the vertex that we will connect to vEvent.
992  */
993  reg = VertLeq( eLo->Dst, eUp->Dst ) ? regUp : regLo;
994 
995  if( regUp->inside || reg->fixUpperEdge) {
996  if( reg == regUp ) {
997  eNew = tessMeshConnect( tess->mesh, vEvent->anEdge->Sym, eUp->Lnext );
998  if (eNew == NULL) longjmp(tess->env,1);
999  } else {
1000  TESShalfEdge *tempHalfEdge= tessMeshConnect( tess->mesh, eLo->Dnext, vEvent->anEdge);
1001  if (tempHalfEdge == NULL) longjmp(tess->env,1);
1002 
1003  eNew = tempHalfEdge->Sym;
1004  }
1005  if( reg->fixUpperEdge ) {
1006  if ( !FixUpperEdge( tess, reg, eNew ) ) longjmp(tess->env,1);
1007  } else {
1008  ComputeWinding( tess, AddRegionBelow( tess, regUp, eNew ));
1009  }
1010  SweepEvent( tess, vEvent );
1011  } else {
1012  /* The new vertex is in a region which does not belong to the polygon.
1013  * We don''t need to connect this vertex to the rest of the mesh.
1014  */
1015  AddRightEdges( tess, regUp, vEvent->anEdge, vEvent->anEdge, NULL, TRUE );
1016  }
1017 }

1020 static void SweepEvent( TESStesselator *tess, TESSvertex *vEvent )
1021 /*
1022 * Does everything necessary when the sweep line crosses a vertex.
1023 * Updates the mesh and the edge dictionary.
1024 */
1025 {
1026  ActiveRegion *regUp, *reg;
1027  TESShalfEdge *e, *eTopLeft, *eBottomLeft;
1028 
1029  tess->event = vEvent; /* for access in EdgeLeq() */
1030  DebugEvent( tess );
1031 
1032  /* Check if this vertex is the right endpoint of an edge that is
1033  * already in the dictionary. In this case we don't need to waste
1034  * time searching for the location to insert new edges.
1035  */
1036  e = vEvent->anEdge;
1037  while( e->activeRegion == NULL ) {
1038  e = e->Onext;
1039  if( e == vEvent->anEdge ) {
1040  /* All edges go right -- not incident to any processed edges */
1041  ConnectLeftVertex( tess, vEvent );
1042  return;
1043  }
1044  }
1045 
1046  /* Processing consists of two phases: first we "finish" all the
1047  * active regions where both the upper and lower edges terminate
1048  * at vEvent (ie. vEvent is closing off these regions).
1049  * We mark these faces "inside" or "outside" the polygon according
1050  * to their winding number, and delete the edges from the dictionary.
1051  * This takes care of all the left-going edges from vEvent.
1052  */
1053  regUp = TopLeftRegion( tess, e->activeRegion );
1054  if (regUp == NULL) longjmp(tess->env,1);
1055  reg = RegionBelow( regUp );
1056  eTopLeft = reg->eUp;
1057  eBottomLeft = FinishLeftRegions( tess, reg, NULL );
1058 
1059  /* Next we process all the right-going edges from vEvent. This
1060  * involves adding the edges to the dictionary, and creating the
1061  * associated "active regions" which record information about the
1062  * regions between adjacent dictionary edges.
1063  */
1064  if( eBottomLeft->Onext == eTopLeft ) {
1065  /* No right-going edges -- add a temporary "fixable" edge */
1066  ConnectRightVertex( tess, regUp, eBottomLeft );
1067  } else {
1068  AddRightEdges( tess, regUp, eBottomLeft->Onext, eTopLeft, eTopLeft, TRUE );
1069  }
1070 }

1076 
1077 static void AddSentinel( TESStesselator *tess, TESSreal smin, TESSreal smax, TESSreal t )
1078 /*
1079 * We add two sentinel edges above and below all other edges,
1080 * to avoid special cases at the top and bottom.
1081 */
1082 {
1083  TESShalfEdge *e;
1084  ActiveRegion *reg = (ActiveRegion *)bucketAlloc( tess->regionPool );
1085  if (reg == NULL) longjmp(tess->env,1);
1086 
1087  e = tessMeshMakeEdge( tess->mesh );
1088  if (e == NULL) longjmp(tess->env,1);
1089 
1090  e->Org->s = smax;
1091  e->Org->t = t;
1092  e->Dst->s = smin;
1093  e->Dst->t = t;
1094  tess->event = e->Dst; /* initialize it */
1095 
1096  reg->eUp = e;
1097  reg->windingNumber = 0;
1098  reg->inside = FALSE;
1099  reg->fixUpperEdge = FALSE;

1265 int tessComputeInterior( TESStesselator *tess )
1266 /*
1267 * tessComputeInterior( tess ) computes the planar arrangement specified
1268 * by the given contours, and further subdivides this arrangement
1269 * into regions. Each region is marked "inside" if it belongs
1270 * to the polygon, according to the rule given by tess->windingRule.
1271 * Each interior region is guaranteed be monotone.
1272 */

1275 
1276  /* Each vertex defines an event for our sweep line. Start by inserting
1277  * all the vertices in a priority queue. Events are processed in
1278  * lexicographic order, ie.
1279  *
1280  * e1 < e2 iff e1.x < e2.x || (e1.x == e2.x && e1.y < e2.y)
1281  */

1283  if ( !InitPriorityQ( tess ) ) return 0; /* if error */
1284  InitEdgeDict( tess );

1287  for( ;; ) {
1288  vNext = (TESSvertex *)pqMinimum( tess->pq );
1289  if( vNext == NULL || ! VertEq( vNext, v )) break;
1290 
1291  /* Merge together all vertices at exactly the same location.
1292  * This is more efficient than processing them one at a time,
1293  * simplifies the code (see ConnectLeftDegenerate), and is also
1294  * important for correct handling of certain degenerate cases.
1295  * For example, suppose there are two identical edges A and B
1296  * that belong to different contours (so without this code they would
1297  * be processed by separate sweep events). Suppose another edge C
1298  * crosses A and B from above. When A is processed, we split it
1299  * at its intersection point with C. However this also splits C,
1300  * so when we insert B we may compute a slightly different
1301  * intersection point. This might leave two edges with a small
1302  * gap between them. This kind of error is especially obvious
1303  * when using boundary extraction (TESS_BOUNDARY_ONLY).
1304  */
1305  vNext = (TESSvertex *)pqExtractMin( tess->pq );
1306  SpliceMergeVertices( tess, v->anEdge, vNext->anEdge );

1308  SweepEvent( tess, v );
