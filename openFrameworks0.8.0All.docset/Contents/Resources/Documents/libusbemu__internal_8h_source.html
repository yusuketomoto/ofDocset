53 struct QuickList
54 {
55  T ini;
56  T end;
57  QuickList()
58  {
59  memset(&ini, 0, sizeof(T));
60  memset(&end, 0, sizeof(T));
61  ini.prev = NULL;
62  ini.next = &end;
63  end.prev = &ini;
64  end.next = NULL;
65  }
66  // copy-constructor is required to be safely used as a plain std::map value
67  QuickList(const QuickList& rhs)
68  {
69  if (!rhs.Empty())
70  {
71  fprintf(stdout, "WARNING: Copy-constructin from a non-empty QuickList!\n");
72  return;
73  }
74  memset(&ini, 0, sizeof(T));
75  memset(&end, 0, sizeof(T));
76  ini.prev = NULL;
77  ini.next = &end;
78  end.prev = &ini;
79  end.next = NULL;
80  }
81  ~QuickList() {};
82  const bool Empty() const
83  {
84  return(ini.next == &end); // could also be (end.prev == &ini)
85  }
86  void Append(T* node)
87  {
88  if (!Orphan(node))
89  {
90  fprintf(stdout, "WARNING: Appending non-orphan node to list...\n");
91  Remove(node);
92  }
93  end.prev->next = node;
94  node->prev = end.prev;
95  node->next = &end;
96  end.prev = node;
97  node->list = this;
98  }
99  T* Head() const
100  {
101  T* head (NULL);
102  if (!Empty())
103  head = ini.next;
104  return(head);
105  }
106  T* Last() const
107  {
108  T* last (NULL);
109  if (!Empty())
110  last = end.prev;
111  return(last);
112  }
113  const bool Member(T* node) const
114  {
115  return(this == node->list);
116  }
117  static T* Prev(T* node)
118  {
119  T* prev (NULL);
120  if (NULL != node->prev->prev)
121  prev = node->prev;
122  return(prev);
123  }
124  static T* Next(T* node)
125  {
126  T* next (NULL);
127  if (NULL != node->next->next)
128  next = node->next;
129  return(next);
130  }
131  const bool Remove (T* node)
132  {
133  if (!Member(node))
134  return(false);
135  node->prev->next = node->next;
136  node->next->prev = node->prev;
137  node->prev = NULL;
138  node->next = NULL;
139  node->list = NULL;
140  return(true);
141  }
142  static void RemoveNode(T* node)
143  {
144  if (Orphan(node))
145  return;
146  node->list->Remove(node);
147  }
148  static const bool Orphan(T* node)
149  {
150  return(NULL == node->list);
151  }

310 libusb_device* libusbemu_register_device(libusb_context* ctx, struct usb_device* dev)
311 {
312  RAIIMutex lock (ctx->mutex);
313  // register the device (if not already there) ...
314  libusb_device dummy = { ctx, dev, 0, NULL, NULL };
315  libusb_context::TMapDevices::iterator it = ctx->devices.insert(std::make_pair(dev,dummy)).first;
316  // ... and increment the reference count
317  libusb_device& record (it->second);
318  record.refcount++;
319  // might as well do some paranoid checkings...
320  assert(record.ctx == ctx);
321  assert(record.device == dev);
322  return(&(it->second));
323 }

325 void libusbemu_unregister_device(libusb_device* dev)
326 {
327  libusb_context* ctx (dev->ctx);
328  RAIIMutex lock (ctx->mutex);
329  // decrement the reference count of the device ...
330  --(dev->refcount);
331  // ... and unregister device if the reference count reaches zero
332  if (0 == dev->refcount)
333  {
334  SAFE_DELETE(dev->handles);
335  // prior to device deletion, all of its transfer lists must be deleted
336  if (NULL != dev->isoTransfers)
337  {
338  libusb_device::TMapIsocTransfers& allTransfers (*(dev->isoTransfers));
339  while (!allTransfers.empty())
340  {
341  libusb_device::TMapIsocTransfers::iterator it (allTransfers.begin());
342  libusb_device::TListTransfers& listTransfers (it->second.listTransfers);
343  while (!listTransfers.Empty())
344  {
345  transfer_wrapper* transfer (listTransfers.Head());
346  // make it orphan so that it can be deleted:
347  listTransfers.Remove(transfer);
348  // the following will free the wrapper object as well:
349  libusb_free_transfer(&transfer->libusb);
350  }
351  allTransfers.erase(it);
352  }
353  SAFE_DELETE(dev->isoTransfers);
354  }
355  ctx->devices.erase(dev->device);
356  }
357 }

359 int libusbemu_setup_transfer(transfer_wrapper* wrapper)
360 {
361  void*& context = wrapper->usb;
362  // paranoid check...
363  if (NULL != context)
364  return(LIBUSB_ERROR_OTHER);
365 
366  RAIIMutex lock (wrapper->libusb.dev_handle->dev->ctx->mutex);
367 
368  int ret (LIBUSB_ERROR_OTHER);
369  libusb_transfer* transfer (&wrapper->libusb);
370  usb_dev_handle* handle (transfer->dev_handle->handle);
371  switch(transfer->type)
372  {
373  case LIBUSB_TRANSFER_TYPE_ISOCHRONOUS :
374  ret = usb_isochronous_setup_async(handle, &context, transfer->endpoint, transfer->iso_packet_desc[0].length);
375  break;
376  case LIBUSB_TRANSFER_TYPE_CONTROL :
377  // libusb-0.1 does not actually support asynchronous control transfers, but this should be
378  // very easy to emulate if necessary: just stick the transfer in a special list and then
379  // libusb_handle_events() check if the list is empty or not; if it is not empty, a thread
380  // is created temporarily just to deal with such control transfer requests until the list
381  // becomes eventually empty again and the thread terminates.
382  case LIBUSB_TRANSFER_TYPE_BULK :
383  case LIBUSB_TRANSFER_TYPE_INTERRUPT :
384  // these transfer types are not being used by libfreenect. they should be fairly simple to
385  // emulate with libusb-0.1 since it already provides support for them.
386  // usb_bulk_setup_async(translate(transfer->dev_handle), &context, transfer->endpoint);
387  // usb_interrupt_setup_async(translate(transfer->dev_handle), &context, transfer->endpoint);
388  default :
389  return(LIBUSB_ERROR_INVALID_PARAM);
390  }
391 
392  if (ret < 0)
393  {
394  // TODO: better error handling...
395  // what do the functions usb_***_setup_async() actually return on error?
396  LIBUSBEMU_ERROR_LIBUSBWIN32();
397  return(ret);
398  }
399 
400  return(LIBUSB_SUCCESS);
401 }
