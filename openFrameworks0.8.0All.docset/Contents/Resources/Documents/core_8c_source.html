85 FREENECTAPI int freenect_process_events_timeout(freenect_context *ctx, struct timeval *timeout)
86 {
87  int res = fnusb_process_events_timeout(&ctx->usb, timeout);
88  // Iterate over the devices in ctx. If any of them are flagged as
89  freenect_device* dev = ctx->first;
90  while(dev) {
91  if (dev->usb_cam.device_dead) {
92  FN_ERROR("USB camera marked dead, stopping streams\n");
93  res = -1;
94  freenect_stop_video(dev);
95  freenect_stop_depth(dev);
96  }
97 #ifdef BUILD_AUDIO
98  if (dev->usb_audio.device_dead) {
99  FN_ERROR("USB audio marked dead, stopping streams\n");
100  res = -1; // Or something else to tell the user that the device just vanished.
101  freenect_stop_audio(dev);
102  }
103 #endif
104  dev = dev->next;
105  }
106  return res;
107 }

155 FREENECTAPI int freenect_open_device(freenect_context *ctx, freenect_device **dev, int index)
156 {
157  int res;
158  freenect_device *pdev = (freenect_device*)malloc(sizeof(freenect_device));
159  if (!pdev)
160  return -1;
161 
162  memset(pdev, 0, sizeof(*pdev));
163 
164  pdev->parent = ctx;
165 
166  res = fnusb_open_subdevices(pdev, index);
167  if (res < 0) {
168  free(pdev);
169  return res;
170  }
171 
172  if (!ctx->first) {
173  ctx->first = pdev;
174  } else {
175  freenect_device *prev = ctx->first;
176  while (prev->next)
177  prev = prev->next;
178  prev->next = pdev;
179  }
180 
181  *dev = pdev;
182 
183  // Do device-specific initialization
184  if (pdev->usb_cam.dev) {
185  if (freenect_camera_init(pdev) < 0) {
186  return -1;
187  }
188  }
189 
190  return 0;
191 }

193 FREENECTAPI int freenect_open_device_by_camera_serial(freenect_context *ctx, freenect_device **dev, const char* camera_serial)
194 {
195  // This is implemented by listing the devices and seeing which index (if
196  // any) has a camera with a matching serial number, and then punting to
197  // freenect_open_device with that index.
198  struct freenect_device_attributes* attrlist;
199  struct freenect_device_attributes* item;
200  int count = fnusb_list_device_attributes(&ctx->usb, &attrlist);
201  if (count < 0) {
202  FN_ERROR("freenect_open_device_by_camera_serial: Couldn't enumerate serial numbers\n");
203  return -1;
204  }
205  int index = 0;
206  for(item = attrlist ; item != NULL; item = item->next , index++) {
207  if (strlen(item->camera_serial) == strlen(camera_serial) && strcmp(item->camera_serial, camera_serial) == 0) {
208  freenect_free_device_attributes(attrlist);
209  return freenect_open_device(ctx, dev, index);
210  }
211  }
212  freenect_free_device_attributes(attrlist);
213  FN_ERROR("freenect_open_device_by_camera_serial: Couldn't find a device with serial %s\n", camera_serial);
214  return -1;
215 }

217 FREENECTAPI int freenect_close_device(freenect_device *dev)
218 {
219  freenect_context *ctx = dev->parent;
220  int res;
221 
222  if (dev->usb_cam.dev) {
223  freenect_camera_teardown(dev);
224  }
225 
226  res = fnusb_close_subdevices(dev);
227  if (res < 0) {
228  FN_ERROR("fnusb_close_subdevices failed: %d\n", res);
229  return res;
230  }
231 
232  freenect_device *last = NULL;
233  freenect_device *cur = ctx->first;
234 
235  while (cur && cur != dev) {
236  last = cur;
237  cur = cur->next;
238  }
239 
240  if (!cur) {
241  FN_ERROR("device %p not found in linked list for this context!\n", dev);
242  return -1;
243  }
244 
245  if (last)
246  last->next = cur->next;
247  else
248  ctx->first = cur->next;
249 
250  free(dev);
251  return 0;
252 }
