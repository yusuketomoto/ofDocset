133 void ofxOscReceiver::ProcessMessage( const osc::ReceivedMessage &m, const IpEndpointName& remoteEndpoint )
134 {
135  // convert the message to an ofxOscMessage
136  ofxOscMessage* ofMessage = new ofxOscMessage();
137 
138  // set the address
139  ofMessage->setAddress( m.AddressPattern() );
140 
141  // set the sender ip/host
142  char endpoint_host[ IpEndpointName::ADDRESS_STRING_LENGTH ];
143  remoteEndpoint.AddressAsString( endpoint_host );
144  ofMessage->setRemoteEndpoint( endpoint_host, remoteEndpoint.port );
145 
146  // transfer the arguments
147  for ( osc::ReceivedMessage::const_iterator arg = m.ArgumentsBegin();
148  arg != m.ArgumentsEnd();
149  ++arg )
150  {
151  if ( arg->IsInt32() )
152  ofMessage->addIntArg( arg->AsInt32Unchecked() );
153  else if ( arg->IsInt64() )
154  ofMessage->addInt64Arg( arg->AsInt64Unchecked() );
155  else if ( arg->IsFloat() )
156  ofMessage->addFloatArg( arg->AsFloatUnchecked() );
157  else if ( arg->IsString() )
158  ofMessage->addStringArg( arg->AsStringUnchecked() );
159  else
160  {
161  ofLogError("ofxOscReceiver") << "ProcessMessage: argument in message " << m.AddressPattern() << " is not an int, float, or string";
162  }
163  }
164 
165  // now add to the queue
166 
167  // at this point we are running inside the thread created by startThread,
168  // so anyone who calls hasWaitingMessages() or getNextMessage() is coming
169  // from a different thread
170 
171  // so we have to practise shared memory management
172 
173  // grab a lock on the queue
174  grabMutex();
175 
176  // add incoming message on to the queue
177  messages.push_back( ofMessage );
178 
179  // release the lock
180  releaseMutex();
181 }

198 bool ofxOscReceiver::getNextMessage( ofxOscMessage* message )
199 {
200  // grab a lock on the queue
201  grabMutex();
202 
203  // check if there are any to be got
204  if ( messages.size() == 0 )
205  {
206  // no: release the mutex
207  releaseMutex();
208  return false;
209  }
210 
211  // copy the message from the queue to message
212  ofxOscMessage* src_message = messages.front();
213  message->copy( *src_message );
214 
215  // now delete the src message
216  delete src_message;
217  // and remove it from the queue
218  messages.pop_front();
219 
220  // release the lock on the queue
221  releaseMutex();
222 
223  // return success
224  return true;
225 }

227 bool ofxOscReceiver::getParameter(ofAbstractParameter & parameter){
228  ofxOscMessage msg;
229  if ( messages.size() == 0 ) return false;

233  vector<string> address = ofSplitString(msg.getAddress(),"/",true);
234  for(int i=0;i<address.size();i++){
235  if(address[i]==p->getName()){
236  if(p->type()==typeid(ofParameterGroup).name()){
237  if(address.size()>=i+1){
238  p = &static_cast<ofParameterGroup*>(p)->get(address[i+1]);
239  }
240  }else if(p->type()==typeid(ofParameter<int>).name() && msg.getArgType(0)==OFXOSC_TYPE_INT32){
241  p->cast<int>() = msg.getArgAsInt32(0);
242  }else if(p->type()==typeid(ofParameter<float>).name() && msg.getArgType(0)==OFXOSC_TYPE_FLOAT){
243  p->cast<float>() = msg.getArgAsFloat(0);
244  }else if(p->type()==typeid(ofParameter<bool>).name() && msg.getArgType(0)==OFXOSC_TYPE_INT32){
245  p->cast<bool>() = msg.getArgAsInt32(0);
246  }else if(msg.getArgType(0)==OFXOSC_TYPE_STRING){
247  p->fromString(msg.getArgAsString(0));
248  }
249  }
250  }
251  }
252  return true;
