159 void ofxOscSender::appendMessage( ofxOscMessage& message, osc::OutboundPacketStream& p )
160 {

169  p << message.getArgAsFloat( i );

171  p << message.getArgAsString( i ).c_str();

174  ofLogError("ofxOscSender") << "appendMessage(): bad argument type " << message.getArgType( i );
