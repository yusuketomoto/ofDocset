302 typedef struct CvStereoBMState
303 {
304  // pre-filtering (normalization of input images)
305  int preFilterType; // =CV_STEREO_BM_NORMALIZED_RESPONSE now
306  int preFilterSize; // averaging window size: ~5x5..21x21
307  int preFilterCap; // the output of pre-filtering is clipped by [-preFilterCap,preFilterCap]
308 
309  // correspondence using Sum of Absolute Difference (SAD)
310  int SADWindowSize; // ~5x5..21x21
311  int minDisparity; // minimum disparity (can be negative)
312  int numberOfDisparities; // maximum disparity - minimum disparity (> 0)
313 
314  // post-filtering
315  int textureThreshold; // the disparity is only computed for pixels
316  // with textured enough neighborhood
317  int uniquenessRatio; // accept the computed disparity d* only if
318  // SAD(d) >= SAD(d*)*(1 + uniquenessRatio/100.)
319  // for any d != d*+/-1 within the search range.
320  int speckleWindowSize; // disparity variation window
321  int speckleRange; // acceptable range of variation in window
322 
323  int trySmallerWindows; // if 1, the results may be more accurate,
324  // at the expense of slower processing
325  CvRect roi1, roi2;
326  int disp12MaxDiff;

328  // temporary buffers
329  CvMat* preFilteredImg0;
330  CvMat* preFilteredImg1;

332  CvMat* cost;
333  CvMat* disp;
334 } CvStereoBMState;
