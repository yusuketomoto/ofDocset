412 void mesh3DS::calculateNormals(){
413 
414  // Doesn't take smoothing groups into account yet
415 
416  if(DEBUG_OUTPUT) std::cout<<"Calculating normals... ";
417  m_normals.assign(m_vertices.size(), 0.0f);
418 
419  Vertex vtx1, vtx2, vtx3;
420  Vector3DS v1, v2, faceNormal;
421 
422  for(int face=0; face < int(m_faces.size()); face+=3){
423  // Calculate face normal
424  vtx1.set(m_vertices[m_faces[face]*3], m_vertices[(m_faces[face]*3)+1], m_vertices[(m_faces[face]*3)+2]);
425  vtx2.set(m_vertices[m_faces[face+1]*3], m_vertices[(m_faces[face+1]*3)+1], m_vertices[(m_faces[face+1]*3)+2]);
426  vtx3.set(m_vertices[m_faces[face+2]*3], m_vertices[(m_faces[face+2]*3)+1], m_vertices[(m_faces[face+2]*3)+2]);
427 
428  v1 = vtx2 - vtx1;
429  v2 = vtx3 - vtx1;
430 
431  faceNormal = v1.crossProduct(v2);
432 
433  // Add normal to all three vertex normals
434  m_normals[m_faces[face]*3] += faceNormal.x;
435  m_normals[(m_faces[face]*3)+1] += faceNormal.y;
436  m_normals[(m_faces[face]*3)+2] += faceNormal.z;
437 
438  m_normals[m_faces[face+1]*3] += faceNormal.x;
439  m_normals[(m_faces[face+1]*3)+1] += faceNormal.y;
440  m_normals[(m_faces[face+1]*3)+2] += faceNormal.z;
441 
442  m_normals[m_faces[face+2]*3] += faceNormal.x;
443  m_normals[(m_faces[face+2]*3)+1] += faceNormal.y;
444  m_normals[(m_faces[face+2]*3)+2] += faceNormal.z;
445 
446  }
447 
448  //normalize all normals
449  for(int n=0; n < int(m_normals.size()); n+=3){
450  faceNormal.set(m_normals[n], m_normals[n+1], m_normals[n+2]);
451  faceNormal.normalize();
452  m_normals[n] = faceNormal.x;
453  m_normals[n+1] = faceNormal.y;
454  m_normals[n+2] = faceNormal.z;
455  }
456 
457  if(DEBUG_OUTPUT) std::cout<<"done"<<std::endl;
458 }
