188 void of3dPrimitive::drawNormals(float length, bool bFaceNormals) {
189  ofNode::transformGL();
190 
191  if(getMesh().usingNormals()) {
192  vector<ofVec3f>& normals = getMesh().getNormals();
193  vector<ofVec3f>& vertices = getMesh().getVertices();
194  ofVec3f normal;
195  ofVec3f vert;
196 
197  normalsMesh.setMode( OF_PRIMITIVE_LINES );
198  normalsMesh.getVertices().resize( normals.size() * 2);
199 
200  if(bFaceNormals) {
201  for(int i = 0; i < (int)normals.size(); i++ ) {
202  if(i % 3 == 0) {
203  vert = (vertices[i]+vertices[i+1]+vertices[i+2]) / 3;
204  } else if(i % 3 == 1) {
205  vert = (vertices[i-1]+vertices[i]+vertices[i+1]) / 3;
206  } else if ( i % 3 == 2) {
207  vert = (vertices[i-2]+vertices[i-1]+vertices[i]) / 3;
208  }
209  normalsMesh.setVertex(i*2, vert);
210  normal = normals[i].getNormalized();
211  normal *= length;
212  normalsMesh.setVertex(i*2+1, normal+vert);
213  }
214  } else {
215  for(int i = 0; i < (int)normals.size(); i++) {
216  vert = vertices[i];
217  normal = normals[i].normalized();
218  normalsMesh.setVertex( i*2, vert);
219  normal *= length;
220  normalsMesh.setVertex(i*2+1, normal+vert);
221  }
222  }
223  normalsMesh.draw();
224  } else {
225  ofLogWarning("of3dPrimitive") << "drawNormals(): mesh normals are disabled";
226  }
227 
228 
229  ofNode::restoreTransformGL();
230 }

504 void ofCylinderPrimitive::set(float _radius, float _height, int radiusSegments, int heightSegments, int capSegments, bool _bCapped, ofPrimitiveMode mode) {
505  radius = _radius;
506  height = _height;
507  bCapped = _bCapped;
508  resolution.set( radiusSegments, heightSegments, capSegments );
509 
510  int resX = getResolution().x;
511  int resY = getResolution().y-1;
512  int resZ = getResolution().z-1;
513 
514  int indexStep = 2;
515  if(mode == OF_PRIMITIVE_TRIANGLES) {
516  indexStep = 6;
517  resX = resX-1;
518  }
519 
520  // 0 -> top cap
521  strides[0][0] = 0;
522  strides[0][1] = resX * resZ * indexStep;
523  vertices[0][0] = 0;
524  vertices[0][1] = getResolution().x * getResolution().z;
525 
526  // 1 -> cylinder //
527  if(bCapped) {
528  strides[1][0] = strides[0][0] + strides[0][1];
529  vertices[1][0] = vertices[0][0] + vertices[0][1];
530  } else {
531  strides[1][0] = 0;
532  vertices[1][0] = 0;
533  }
534  strides[1][1] = resX * resY * indexStep;
535  vertices[1][1] = getResolution().x * getResolution().y;
536 
537  // 2 -> bottom cap
538  strides[2][0] = strides[1][0] + strides[1][1];
539  strides[2][1] = resX * resZ * indexStep;
540  vertices[2][0] = vertices[1][0]+vertices[1][1];
541  vertices[2][1] = getResolution().x * getResolution().z;
542 
543 
544  getMesh() = ofMesh::cylinder( getRadius(), getHeight(), getResolution().x, getResolution().y, getResolution().z, getCapped(), mode );
545 
546  normalizeAndApplySavedTexCoords();
547 
548 }
