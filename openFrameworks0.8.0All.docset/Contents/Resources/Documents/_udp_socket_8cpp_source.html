135  {
136  assert( isBound_ );

139 
140  struct sockaddr_in connectSockAddr;

142

144  throw std::runtime_error("unable to connect udp socket\n");

383  void Run()
384  {
385  break_ = false;
386 
387  // configure the master fd_set for select()
388 
389  fd_set masterfds, tempfds;
390  FD_ZERO( &masterfds );
391  FD_ZERO( &tempfds );
392 
393  // in addition to listening to the inbound sockets we
394  // also listen to the asynchronous break pipe, so that AsynchronousBreak()
395  // can break us out of select() from another thread.
396  FD_SET( breakPipe_[0], &masterfds );
397  int fdmax = breakPipe_[0];
398 
399  for( std::vector< std::pair< PacketListener*, UdpSocket* > >::iterator i = socketListeners_.begin();
400  i != socketListeners_.end(); ++i ){
401 
402  if( fdmax < i->second->impl_->Socket() )
403  fdmax = i->second->impl_->Socket();
404  FD_SET( i->second->impl_->Socket(), &masterfds );
405  }
406 
407 
408  // configure the timer queue
409  double currentTimeMs = GetCurrentTimeMs();
410 
411  // expiry time ms, listener
412  std::vector< std::pair< double, AttachedTimerListener > > timerQueue_;
413  for( std::vector< AttachedTimerListener >::iterator i = timerListeners_.begin();
414  i != timerListeners_.end(); ++i )
415  timerQueue_.push_back( std::make_pair( currentTimeMs + i->initialDelayMs, *i ) );
416  std::sort( timerQueue_.begin(), timerQueue_.end(), CompareScheduledTimerCalls );
417 
418  const int MAX_BUFFER_SIZE = 4098;
419  char *data = new char[ MAX_BUFFER_SIZE ];
420  IpEndpointName remoteEndpoint;
421 
422  struct timeval timeout;
423 
424  while( !break_ ){
425  tempfds = masterfds;
426 
427  struct timeval *timeoutPtr = 0;
428  if( !timerQueue_.empty() ){
429  double timeoutMs = timerQueue_.front().first - GetCurrentTimeMs();
430  if( timeoutMs < 0 )
431  timeoutMs = 0;
432 
433  // 1000000 microseconds in a second
434  timeout.tv_sec = (long)(timeoutMs * .001);
435  timeout.tv_usec = (long)((timeoutMs - (timeout.tv_sec * 1000)) * 1000);
436  timeoutPtr = &timeout;
437  }
438 
439  if( select( fdmax + 1, &tempfds, 0, 0, timeoutPtr ) < 0 && errno != EINTR ){
440  throw std::runtime_error("select failed\n");
441  }
442 
443  if ( FD_ISSET( breakPipe_[0], &tempfds ) ){
444  // clear pending data from the asynchronous break pipe
445  char c;
446  read( breakPipe_[0], &c, 1 );
447  }
448 
449  if( break_ )
450  break;
451 
452  for( std::vector< std::pair< PacketListener*, UdpSocket* > >::iterator i = socketListeners_.begin();
453  i != socketListeners_.end(); ++i ){
454 
455  if( FD_ISSET( i->second->impl_->Socket(), &tempfds ) ){
456 
457  int size = i->second->ReceiveFrom( remoteEndpoint, data, MAX_BUFFER_SIZE );
458  if( size > 0 ){
459  i->first->ProcessPacket( data, size, remoteEndpoint );
460  if( break_ )
461  break;
462  }
463  }
464  }
465 
466  // execute any expired timers
467  currentTimeMs = GetCurrentTimeMs();
468  bool resort = false;
469  for( std::vector< std::pair< double, AttachedTimerListener > >::iterator i = timerQueue_.begin();
470  i != timerQueue_.end() && i->first <= currentTimeMs; ++i ){
471 
472  i->second.listener->TimerExpired();
473  if( break_ )
474  break;
475 
476  i->first += i->second.periodMs;
477  resort = true;
478  }
479  if( resort )
480  std::sort( timerQueue_.begin(), timerQueue_.end(), CompareScheduledTimerCalls );
481  }
482 
483  delete [] data;
484  }
