1153 void ofCairoRenderer::setSphereResolution(int res) {
1154  int n = res * 2;
1155  float ndiv2=(float)n/2;
1156  int stripVerts = (ndiv2*((n+1)*2));
1157 
1158  if((int)sphereVerts.size() != stripVerts ) {
1159  sphereVerts.clear();
1160  sphereVerts.resize( (ndiv2*((n+1)*2)) );
1161  } else {
1162  return;
1163  }
1164 
1165  /*
1166  Original code by Paul Bourke
1167  A more efficient contribution by Federico Dosil (below)
1168  Draw a point for zero radius spheres
1169  Use CCW facet ordering
1170  http://paulbourke.net/texture_colour/texturemap/
1171  */
1172 
1173  float theta2 = TWO_PI;
1174  float phi1 = -HALF_PI;
1175  float phi2 = HALF_PI;
1176  float radius = 1.f; // normalize the verts //
1177 
1178  int i, j;
1179  float j1divn,idivn,dosdivn,unodivn=1/(float)n,t1,t2,t3,cost1,cost2,cte1,cte3;
1180  cte3 = (theta2)/n;
1181  cte1 = (phi2-phi1)/ndiv2;
1182  dosdivn = 2*unodivn;
1183  ofVec3f e,p,e2,p2;
1184 
1185  // Handle special cases //
1186  if (n < 0){
1187  n = -n;
1188  ndiv2 = -ndiv2;
1189  }
1190  if (n < 4) {
1191  ofLogWarning("ofCairoRenderer") << "setSphereResolution(): ignoring low sphere resolution: " << n;
1192  }
1193 
1194  t2=phi1;
1195  cost2=cos(phi1);
1196  j1divn=0;
1197 
1198  int cindex = 0; // current index //
1199 
1200  for (j=0;j<ndiv2;j++) {
1201  t1 = t2;
1202  t2 += cte1;
1203  t3 = -cte3;
1204  cost1 = cost2;
1205  cost2 = cos(t2);
1206  e.y = sin(t1);
1207  e2.y = sin(t2);
1208  p.y = radius * e.y;
1209  p2.y = radius * e2.y;
1210 
1211  idivn=0;
1212  j1divn+=dosdivn;
1213  for (i=0;i<=n;i++) {
1214  t3 += cte3;
1215  e.x = cost1 * cos(t3);
1216  e.z = cost1 * sin(t3);
1217  p.x = radius * e.x;
1218  p.z = radius * e.z;
1219 
1220  cindex = (j * (n+1) + i) * 2;
1221  sphereVerts[cindex].set(p.x,p.y,p.z);
1222 
1223  e2.x = cost2 * cos(t3);
1224  e2.z = cost2 * sin(t3);
1225  p2.x = radius * e2.x;
1226  p2.z = radius * e2.z;
1227  cindex = (j * (n+1) + i) * 2 + 1;
1228  sphereVerts[cindex].set(p2.x,p2.y,p2.z);
1229 
1230  idivn += unodivn;
1231  }
1232  }
1233 }
