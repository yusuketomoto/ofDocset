355  ofRectangle currentViewport = getCurrentViewport();
356 
357  float viewW = currentViewport.width;
358  float viewH = currentViewport.height;
359 
360  float eyeX = viewW / 2;
361  float eyeY = viewH / 2;
362  float halfFov = PI * fov / 360;
363  float theTan = tanf(halfFov);
364  float dist = eyeY / theTan;
365  float aspect = (float) viewW / viewH;
366 
367  if(nearDist == 0) nearDist = dist / 10.0f;

369

373  persp.makePerspectiveMatrix(fov, aspect, nearDist, farDist);
