92 bool ofxKinect::init(bool infrared, bool video, bool texture) {
93  if(isConnected()) {
94  ofLogWarning("ofxKinect") << "init(): do not call init while ofxKinect is running!";
95  return false;
96  }
97 
98  clear();
99 
100  bIsVideoInfrared = infrared;
101  bGrabVideo = video;
102  videoBytesPerPixel = infrared?1:3;
103 
104  bUseTexture = texture;
105 
106  // allocate
107  depthPixelsRaw.allocate(width, height, 1);
108  depthPixelsRawBack.allocate(width, height, 1);
109  depthPixelsRawIntra.allocate(width, height, 1);
110 
111  videoPixels.allocate(width, height, videoBytesPerPixel);
112  videoPixelsBack.allocate(width, height, videoBytesPerPixel);
113  videoPixelsIntra.allocate(width, height, videoBytesPerPixel);
114 
115  depthPixels.allocate(width, height, 1);
116  distancePixels.allocate(width, height, 1);
117 
118  // set
119  depthPixelsRaw.set(0);
120  depthPixelsRawBack.set(0);
121 
122  videoPixels.set(0);
123  videoPixelsBack.set(0);
124 
125  depthPixels.set(0);
126  distancePixels.set(0);
127 
128  if(bUseTexture) {
129  depthTex.allocate(width, height, GL_LUMINANCE);
130  videoTex.allocate(width, height, infrared ? GL_LUMINANCE : GL_RGB);
131  }
132 
133  if(!kinectContext.isInited()) {
134  if(!kinectContext.init()) {
135  return false;
136  }
137  }
138 
139  bGrabberInited = true;
140 
141  return bGrabberInited;
142 }

173  if(!bGrabberInited) {
174  ofLogWarning("ofxKinect") << "open(): cannot open, init not called";
175  return false;
176  }
177 
178  if(!kinectContext.open(*this, id)) {
179  return false;
180  }
181 
182  if(serial == "0000000000000000") {
183  ofLogVerbose("ofxKinect") << "open(): device " << deviceId << " does not have motor control";
184  bHasMotorControl = false;
185  }
186  else {
187  bHasMotorControl = true;
188  }
189 
190  lastDeviceId = deviceId;
191  timeSinceOpen = ofGetElapsedTimef();
192  bGotData = false;
193 
194  freenect_set_user(kinectDevice, this);
195  freenect_set_depth_buffer(kinectDevice, depthPixelsRawBack.getPixels());
196  freenect_set_video_buffer(kinectDevice, videoPixelsBack.getPixels());
197  freenect_set_depth_callback(kinectDevice, &grabDepthFrame);

199 
200  startThread(true, false); // blocking, not verbose
201 
202  return true;

273 //----------------------------------------------------------
274 void ofxKinect::update() {
275  if(!bGrabberInited) {
276  return;
277  }
278 
279  if(!bNeedsUpdateVideo && !bNeedsUpdateDepth && !bGotData && tryCount < 5 && ofGetElapsedTimef() - timeSinceOpen > 2.0 ){
280  close();
281  ofLogWarning("ofxKinect") << "update(): device " << lastDeviceId << " isn't delivering data, reconnecting tries: " << tryCount+1;
282  kinectContext.buildDeviceList();
283  open(lastDeviceId);
284  tryCount++;
285  timeSinceOpen = ofGetElapsedTimef();
286  return;
287  }
288 
289  if(bNeedsUpdateVideo){
290  bIsFrameNewVideo = true;
291  bGotData = true;
292  tryCount = 0;
293  if(this->lock()) {
294  swap(videoPixels,videoPixelsIntra);
295  bNeedsUpdateVideo = false;
296  this->unlock();
297  }
298 
299  if(bUseTexture) {
300  videoTex.loadData(videoPixels.getPixels(), width, height, bIsVideoInfrared?GL_LUMINANCE:GL_RGB);
301  }
302  } else {
303  bIsFrameNewVideo = false;
304  }
305 
306  if(bNeedsUpdateDepth){
307  bIsFrameNewDepth = true;
308  bGotData = true;
309  tryCount = 0;
310  if(this->lock()) {
311  swap(depthPixelsRaw, depthPixelsRawIntra);
312  bNeedsUpdateDepth = false;
313  this->unlock();
314 
315  updateDepthPixels();
316  }
317 
318  if(bUseTexture) {
319  depthTex.loadData(depthPixels.getPixels(), width, height, GL_LUMINANCE);
320  }
321  } else {
322  bIsFrameNewDepth = false;
323  }
324 
325 }

644 void ofxKinect::updateDepthLookupTable() {
645  unsigned char nearColor = bNearWhite ? 255 : 0;
646  unsigned char farColor = bNearWhite ? 0 : 255;
647  unsigned int maxDepthLevels = 10001;
648  depthLookupTable.resize(maxDepthLevels);
649  depthLookupTable[0] = 0;
650  for(unsigned int i = 1; i < maxDepthLevels; i++) {
651  depthLookupTable[i] = ofMap(i, nearClipping, farClipping, nearColor, farColor, true);
652  }
653 }

694 //---------------------------------------------------------------------------
695 void ofxKinect::threadedFunction(){
696 
697  if(currentLed < 0) {
698  freenect_set_led(kinectDevice, (freenect_led_options)ofxKinect::LED_GREEN);
699  }
700 
701  freenect_frame_mode videoMode = freenect_find_video_mode(FREENECT_RESOLUTION_MEDIUM, bIsVideoInfrared?FREENECT_VIDEO_IR_8BIT:FREENECT_VIDEO_RGB);
702  freenect_set_video_mode(kinectDevice, videoMode);
703  freenect_frame_mode depthMode = freenect_find_depth_mode(FREENECT_RESOLUTION_MEDIUM, bUseRegistration?FREENECT_DEPTH_REGISTERED:FREENECT_DEPTH_MM);
704  freenect_set_depth_mode(kinectDevice, depthMode);
705 
706  ofLogVerbose("ofxKinect") << "device " << deviceId << " " << serial << " connection opened";
707 
708  freenect_start_depth(kinectDevice);
709  if(bGrabVideo) {
710  freenect_start_video(kinectDevice);
711  }
712 
713  while(isThreadRunning() && freenect_process_events(kinectContext.getContext()) >= 0) {
714 
715  if(bTiltNeedsApplying) {
716  freenect_set_tilt_degs(kinectDevice, targetTiltAngleDeg);
717  bTiltNeedsApplying = false;
718  }
719 
720  if(bLedNeedsApplying) {
721  if(currentLed == ofxKinect::LED_DEFAULT) {
722  freenect_set_led(kinectDevice, (freenect_led_options)ofxKinect::LED_GREEN);
723  }
724  else {
725  freenect_set_led(kinectDevice, (freenect_led_options)currentLed);
726  }
727  bLedNeedsApplying = false;
728  }
729 
730  freenect_update_tilt_state(kinectDevice);
731  freenect_raw_tilt_state * tilt = freenect_get_tilt_state(kinectDevice);
732  currentTiltAngleDeg = freenect_get_tilt_degs(tilt);
733 
734  rawAccel.set(tilt->accelerometer_x, tilt->accelerometer_y, tilt->accelerometer_z);
735 
736  double dx,dy,dz;
