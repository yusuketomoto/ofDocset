127 inline aiMatrix4x4& aiMatrix4x4::Inverse()
128 {
129  // Compute the reciprocal determinant
130  float det = Determinant();
131  if(det == 0.0f)
132  {
133  // Matrix not invertible. Setting all elements to nan is not really
134  // correct in a mathematical sense but it is easy to debug for the
135  // programmer.
136  const float nan = std::numeric_limits<float>::quiet_NaN();
137  *this = aiMatrix4x4(
138  nan,nan,nan,nan,
139  nan,nan,nan,nan,
140  nan,nan,nan,nan,
141  nan,nan,nan,nan);
142 
143  return *this;
144  }
145 
146  float invdet = 1.0f / det;
147 
148  aiMatrix4x4 res;
149  res.a1 = invdet * (b2 * (c3 * d4 - c4 * d3) + b3 * (c4 * d2 - c2 * d4) + b4 * (c2 * d3 - c3 * d2));
150  res.a2 = -invdet * (a2 * (c3 * d4 - c4 * d3) + a3 * (c4 * d2 - c2 * d4) + a4 * (c2 * d3 - c3 * d2));
151  res.a3 = invdet * (a2 * (b3 * d4 - b4 * d3) + a3 * (b4 * d2 - b2 * d4) + a4 * (b2 * d3 - b3 * d2));
152  res.a4 = -invdet * (a2 * (b3 * c4 - b4 * c3) + a3 * (b4 * c2 - b2 * c4) + a4 * (b2 * c3 - b3 * c2));
153  res.b1 = -invdet * (b1 * (c3 * d4 - c4 * d3) + b3 * (c4 * d1 - c1 * d4) + b4 * (c1 * d3 - c3 * d1));
154  res.b2 = invdet * (a1 * (c3 * d4 - c4 * d3) + a3 * (c4 * d1 - c1 * d4) + a4 * (c1 * d3 - c3 * d1));
155  res.b3 = -invdet * (a1 * (b3 * d4 - b4 * d3) + a3 * (b4 * d1 - b1 * d4) + a4 * (b1 * d3 - b3 * d1));
156  res.b4 = invdet * (a1 * (b3 * c4 - b4 * c3) + a3 * (b4 * c1 - b1 * c4) + a4 * (b1 * c3 - b3 * c1));
157  res.c1 = invdet * (b1 * (c2 * d4 - c4 * d2) + b2 * (c4 * d1 - c1 * d4) + b4 * (c1 * d2 - c2 * d1));
158  res.c2 = -invdet * (a1 * (c2 * d4 - c4 * d2) + a2 * (c4 * d1 - c1 * d4) + a4 * (c1 * d2 - c2 * d1));
159  res.c3 = invdet * (a1 * (b2 * d4 - b4 * d2) + a2 * (b4 * d1 - b1 * d4) + a4 * (b1 * d2 - b2 * d1));
160  res.c4 = -invdet * (a1 * (b2 * c4 - b4 * c2) + a2 * (b4 * c1 - b1 * c4) + a4 * (b1 * c2 - b2 * c1));
161  res.d1 = -invdet * (b1 * (c2 * d3 - c3 * d2) + b2 * (c3 * d1 - c1 * d3) + b3 * (c1 * d2 - c2 * d1));
162  res.d2 = invdet * (a1 * (c2 * d3 - c3 * d2) + a2 * (c3 * d1 - c1 * d3) + a3 * (c1 * d2 - c2 * d1));
163  res.d3 = -invdet * (a1 * (b2 * d3 - b3 * d2) + a2 * (b3 * d1 - b1 * d3) + a3 * (b1 * d2 - b2 * d1));
164  res.d4 = invdet * (a1 * (b2 * c3 - b3 * c2) + a2 * (b3 * c1 - b1 * c3) + a3 * (b1 * c2 - b2 * c1));
165  *this = res;
166 
167  return *this;
168 }

198 inline void aiMatrix4x4::Decompose (aiVector3D& scaling, aiQuaternion& rotation,
199  aiVector3D& position) const
200 {
201  const aiMatrix4x4& _this = *this;
202 
203  // extract translation
204  position.x = _this[0][3];
205  position.y = _this[1][3];
206  position.z = _this[2][3];
207 
208  // extract the rows of the matrix
209  aiVector3D vRows[3] = {
210  aiVector3D(_this[0][0],_this[1][0],_this[2][0]),
211  aiVector3D(_this[0][1],_this[1][1],_this[2][1]),
212  aiVector3D(_this[0][2],_this[1][2],_this[2][2])
213  };
214 
215  // extract the scaling factors
216  scaling.x = vRows[0].Length();
217  scaling.y = vRows[1].Length();
218  scaling.z = vRows[2].Length();
219 
220  // and remove all scaling from the matrix
221  if(scaling.x)
222  {
223  vRows[0] /= scaling.x;
224  }
225  if(scaling.y)
226  {
227  vRows[1] /= scaling.y;
228  }
229  if(scaling.z)
230  {
231  vRows[2] /= scaling.z;
232  }
233 
234  // build a 3x3 rotation matrix
235  aiMatrix3x3 m(vRows[0].x,vRows[1].x,vRows[2].x,
236  vRows[0].y,vRows[1].y,vRows[2].y,
237  vRows[0].z,vRows[1].z,vRows[2].z);
238 
239  // and generate the rotation quaternion from it
240  rotation = aiQuaternion(m);
241 }

265 inline aiMatrix4x4& aiMatrix4x4::FromEulerAnglesXYZ(float x, float y, float z)
266 {
267  aiMatrix4x4& _this = *this;
268 
269  float cr = cos( x );
270  float sr = sin( x );
271  float cp = cos( y );
272  float sp = sin( y );
273  float cy = cos( z );
274  float sy = sin( z );
275 
276  _this.a1 = cp*cy ;
277  _this.a2 = cp*sy;
278  _this.a3 = -sp ;
279 
280  float srsp = sr*sp;
281  float crsp = cr*sp;
282 
283  _this.b1 = srsp*cy-cr*sy ;
284  _this.b2 = srsp*sy+cr*cy ;
285  _this.b3 = sr*cp ;
286 
287  _this.c1 = crsp*cy+sr*sy ;
288  _this.c2 = crsp*sy-sr*cy ;
289  _this.c3 = cr*cp ;
290 
291  return *this;
292 }

362 // Returns a rotation matrix for a rotation around an arbitrary axis.
363 inline aiMatrix4x4& aiMatrix4x4::Rotation( float a, const aiVector3D& axis, aiMatrix4x4& out)
364 {
365  float c = cos( a), s = sin( a), t = 1 - c;
366  float x = axis.x, y = axis.y, z = axis.z;
367 
368  // Many thanks to MathWorld and Wikipedia
369  out.a1 = t*x*x + c; out.a2 = t*x*y - s*z; out.a3 = t*x*z + s*y;
370  out.b1 = t*x*y + s*z; out.b2 = t*y*y + c; out.b3 = t*y*z - s*x;
371  out.c1 = t*x*z - s*y; out.c2 = t*y*z + s*x; out.c3 = t*z*z + c;
372  out.a4 = out.b4 = out.c4 = 0.0f;
373  out.d1 = out.d2 = out.d3 = 0.0f;
374  out.d4 = 1.0f;
375 
376  return out;
377 }
