168 // DataType: STRUCT filterObject
169 // Description of the filter, which corresponds to the part of the object
170 // V - ideal (penalty = 0) position of the partial filter
171 // from the root filter position (V_i in the paper)
172 // penaltyFunction - vector describes penalty function (d_i in the paper)
173 // pf[0] * x + pf[1] * y + pf[2] * x^2 + pf[3] * y^2
174 // FILTER DESCRIPTION
175 // Rectangular map (sizeX x sizeY),
176 // every cell stores feature vector (dimension = p)
177 // H - matrix of feature vectors
178 // to set and get feature vectors (i,j)
179 // used formula H[(j * sizeX + i) * p + k], where
180 // k - component of feature vector in cell (i, j)
181 // END OF FILTER DESCRIPTION
182 typedef struct{
183  CvLSVMFilterPosition V;
184  float fineFunction[4];
185  int sizeX;
186  int sizeY;
187  int numFeatures;
188  float *H;
189 } CvLSVMFilterObject;
