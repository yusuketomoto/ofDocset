2426 inline void BruteForceMatcher<Distance>::commonKnnMatchImpl( BruteForceMatcher<Distance>& matcher,
2427  const Mat& queryDescriptors, vector<vector<DMatch> >& matches, int knn,
2428  const vector<Mat>& masks, bool compactResult )
2429 {
2430  typedef typename Distance::ValueType ValueType;
2431  typedef typename Distance::ResultType DistanceType;
2432  CV_DbgAssert( !queryDescriptors.empty() );
2433  CV_Assert( DataType<ValueType>::type == queryDescriptors.type() );
2434 
2435  int dimension = queryDescriptors.cols;
2436  matches.reserve(queryDescriptors.rows);
2437 
2438  size_t imgCount = matcher.trainDescCollection.size();

2440  for( size_t i = 0; i < imgCount; i++ )

2462  {
