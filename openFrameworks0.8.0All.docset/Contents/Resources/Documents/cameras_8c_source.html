87 {
88  if (len < 12)
89  return 0;
90 
91  struct pkt_hdr *hdr = (struct pkt_hdr*)pkt;
92  uint8_t *data = pkt + sizeof(*hdr);
93  int datalen = len - sizeof(*hdr);
94 
95  freenect_loglevel l_info = LL_INFO;
96  freenect_loglevel l_notice = LL_NOTICE;
97  freenect_loglevel l_warning = LL_WARNING;
98  if (strm->valid_frames < 2)
99  l_info = l_notice = l_warning = LL_SPEW;
100 
101  if (hdr->magic[0] != 'R' || hdr->magic[1] != 'B') {
102  FN_LOG(l_notice, "[Stream %02x] Invalid magic %02x%02x\n",
103  strm->flag, hdr->magic[0], hdr->magic[1]);
104  return 0;
105  }
106 
107  FN_FLOOD("[Stream %02x] Packet with flag: %02x\n", strm->flag, hdr->flag);
108 
109  uint8_t sof = strm->flag|1;
110  uint8_t mof = strm->flag|2;
111  uint8_t eof = strm->flag|5;
112 
113  // sync if required, dropping packets until SOF
114  if (!strm->synced) {
115  if (hdr->flag != sof) {
116  FN_SPEW("[Stream %02x] Not synced yet...\n", strm->flag);
117  return 0;
118  }
119  strm->synced = 1;
120  strm->seq = hdr->seq;
121  strm->pkt_num = 0;
122  strm->valid_pkts = 0;
123  strm->got_pkts = 0;
124  }
125 
126  int got_frame_size = 0;
127 
128  // handle lost packets
129  if (strm->seq != hdr->seq) {
130  uint8_t lost = hdr->seq - strm->seq;
131  FN_LOG(l_info, "[Stream %02x] Lost %d packets\n", strm->flag, lost);
132  if (lost > 5 || strm->variable_length) {
133  FN_LOG(l_notice, "[Stream %02x] Lost too many packets, resyncing...\n", strm->flag);
134  strm->synced = 0;
135  return 0;
136  }
137  strm->seq = hdr->seq;
138  int left = strm->pkts_per_frame - strm->pkt_num;
139  if (left <= lost) {
140  strm->pkt_num = lost - left;
141  strm->valid_pkts = strm->got_pkts;
142  strm->got_pkts = 0;
143  got_frame_size = strm->frame_size;
144  strm->timestamp = strm->last_timestamp;
145  strm->valid_frames++;
146  } else {
147  strm->pkt_num += lost;
148  }
149  }
150 
151  int expected_pkt_size = (strm->pkt_num == strm->pkts_per_frame-1) ? strm->last_pkt_size : strm->pkt_size;
152 
153  if (!strm->variable_length) {
154  // check the header to make sure it's what we expect
155  if (!(strm->pkt_num == 0 && hdr->flag == sof) &&
156  !(strm->pkt_num == strm->pkts_per_frame-1 && hdr->flag == eof) &&
157  !(strm->pkt_num > 0 && strm->pkt_num < strm->pkts_per_frame-1 && hdr->flag == mof)) {
158  FN_LOG(l_notice, "[Stream %02x] Inconsistent flag %02x with %d packets in buf (%d total), resyncing...\n",
159  strm->flag, hdr->flag, strm->pkt_num, strm->pkts_per_frame);
160  strm->synced = 0;
161  return got_frame_size;
162  }
163  // check data length
164  if (datalen > expected_pkt_size) {
165  FN_LOG(l_warning, "[Stream %02x] Expected max %d data bytes, but got %d. Dropping...\n",
166  strm->flag, expected_pkt_size, datalen);
167  return got_frame_size;
168  }
169  if (datalen < expected_pkt_size)
170  FN_LOG(l_warning, "[Stream %02x] Expected %d data bytes, but got %d\n",
171  strm->flag, expected_pkt_size, datalen);
172  } else {
173  // check the header to make sure it's what we expect
174  if (!(strm->pkt_num == 0 && hdr->flag == sof) &&
175  !(strm->pkt_num < strm->pkts_per_frame && (hdr->flag == eof || hdr->flag == mof))) {
176  FN_LOG(l_notice, "[Stream %02x] Inconsistent flag %02x with %d packets in buf (%d total), resyncing...\n",
177  strm->flag, hdr->flag, strm->pkt_num, strm->pkts_per_frame);
178  strm->synced = 0;
179  return got_frame_size;
180  }
181  // check data length
182  if (datalen > expected_pkt_size) {
183  FN_LOG(l_warning, "[Stream %02x] Expected max %d data bytes, but got %d. Resyncng...\n",
184  strm->flag, expected_pkt_size, datalen);
185  strm->synced = 0;
186  return got_frame_size;
187  }
188  if (datalen < expected_pkt_size && hdr->flag != eof) {
189  FN_LOG(l_warning, "[Stream %02x] Expected %d data bytes, but got %d. Resyncing...\n",
190  strm->flag, expected_pkt_size, datalen);
191  strm->synced = 0;
192  return got_frame_size;
193  }
194  }
195 
196  // copy data
197  uint8_t *dbuf = strm->raw_buf + strm->pkt_num * strm->pkt_size;
198  memcpy(dbuf, data, datalen);
199 
200  strm->pkt_num++;
201  strm->seq++;
202  strm->got_pkts++;
203 
204  strm->last_timestamp = fn_le32(hdr->timestamp);
205 
206  if (hdr->flag == eof) {
207  if (strm->variable_length)
208  got_frame_size = (dbuf - strm->raw_buf) + datalen;
209  else
210  got_frame_size = (dbuf - strm->raw_buf) + strm->last_pkt_size;
211  strm->pkt_num = 0;
212  strm->valid_pkts = strm->got_pkts;
213  strm->got_pkts = 0;
214  strm->timestamp = strm->last_timestamp;
215  strm->valid_frames++;
216  }
217  return got_frame_size;
218 }

294 {
295  unsigned int mask = (1 << vw) - 1;
296  uint32_t buffer = 0;
297  int bitsIn = 0;

299  while (bitsIn < vw) {
300  buffer = (buffer << 8) | *(src++);
301  bitsIn += 8;
302  }
303  bitsIn -= vw;
304  *(dest++) = (buffer >> bitsIn) & mask;
305  }

333 // Loop-unrolled version of the 11-to-16 bit unpacker. n must be a multiple of 8.
334 static void convert_packed11_to_16bit(uint8_t *raw, uint16_t *frame, int n)
335 {
336  uint16_t baseMask = (1 << 11) - 1;
337  while(n >= 8)
338  {
339  uint8_t r0 = *(raw+0);
340  uint8_t r1 = *(raw+1);
341  uint8_t r2 = *(raw+2);
342  uint8_t r3 = *(raw+3);
343  uint8_t r4 = *(raw+4);
344  uint8_t r5 = *(raw+5);
345  uint8_t r6 = *(raw+6);
346  uint8_t r7 = *(raw+7);
347  uint8_t r8 = *(raw+8);
348  uint8_t r9 = *(raw+9);
349  uint8_t r10 = *(raw+10);
350 
351  frame[0] = (r0<<3) | (r1>>5);
352  frame[1] = ((r1<<6) | (r2>>2) ) & baseMask;
353  frame[2] = ((r2<<9) | (r3<<1) | (r4>>7) ) & baseMask;
354  frame[3] = ((r4<<4) | (r5>>4) ) & baseMask;
355  frame[4] = ((r5<<7) | (r6>>1) ) & baseMask;
356  frame[5] = ((r6<<10) | (r7<<2) | (r8>>6) ) & baseMask;
357  frame[6] = ((r8<<5) | (r9>>3) ) & baseMask;
358  frame[7] = ((r9<<8) | (r10) ) & baseMask;
359 
360  n -= 8;
361  raw += 11;
362  frame += 8;
363  }
364 }

442 static void convert_bayer_to_rgb(uint8_t *raw_buf, uint8_t *proc_buf, freenect_frame_mode frame_mode)
443 {
444  int x,y;
445  /* Pixel arrangement:
446  * G R G R G R G R
447  * B G B G B G B G
448  * G R G R G R G R
449  * B G B G B G B G
450  * G R G R G R G R
451  * B G B G B G B G
452  *
453  * To convert a Bayer-pattern into RGB you have to handle four pattern
454  * configurations:
455  * 1) 2) 3) 4)
456  * B1 B1 G1 B2 R1 G1 R2 R1 <- previous line
457  * R1 G1 R2 G2 R1 G3 G2 B1 G3 B1 G1 B2 <- current line
458  * B2 B3 G4 B4 R3 G4 R4 R2 <- next line
459  * ^ ^ ^
460  * | | next pixel
461  * | current pixel
462  * previous pixel
463  *
464  * The RGB values (r,g,b) for each configuration are calculated as
465  * follows:
466  *
467  * 1) r = (R1 + R2) / 2
468  * g = G1
469  * b = (B1 + B2) / 2
470  *
471  * 2) r = R1
472  * g = (G1 + G2 + G3 + G4) / 4
473  * b = (B1 + B2 + B3 + B4) / 4
474  *
475  * 3) r = (R1 + R2 + R3 + R4) / 4
476  * g = (G1 + G2 + G3 + G4) / 4
477  * b = B1
478  *
479  * 4) r = (R1 + R2) / 2
480  * g = G1
481  * b = (B1 + B2) / 2
482  *
483  * To efficiently calculate these values, two 32bit integers are used
484  * as "shift-buffers". One integer to store the 3 horizontal bayer pixel
485  * values (previous, current, next) of the current line. The other
486  * integer to store the vertical average value of the bayer pixels
487  * (previous, current, next) of the previous and next line.
488  *
489  * The boundary conditions for the first and last line and the first
490  * and last column are solved via mirroring the second and second last
491  * line and the second and second last column.
492  *
493  * To reduce slow memory access, the values of a rgb pixel are packet
494  * into a 32bit variable and transfered together.
495  */
496 
497  uint8_t *dst = proc_buf; // pointer to destination
498 
499  uint8_t *prevLine; // pointer to previous, current and next line
500  uint8_t *curLine; // of the source bayer pattern
501  uint8_t *nextLine;
502 
503  // storing horizontal values in hVals:
504  // previous << 16, current << 8, next
505  uint32_t hVals;
506  // storing vertical averages in vSums:
507  // previous << 16, current << 8, next
508  uint32_t vSums;
509 
510  // init curLine and nextLine pointers
511  curLine = raw_buf;
512  nextLine = curLine + frame_mode.width;
513  for (y = 0; y < frame_mode.height; ++y) {
514 
515  if ((y > 0) && (y < frame_mode.height-1))
516  prevLine = curLine - frame_mode.width; // normal case
517  else if (y == 0)
518  prevLine = nextLine; // top boundary case
519  else
520  nextLine = prevLine; // bottom boundary case
521 
522  // init horizontal shift-buffer with current value
523  hVals = (*(curLine++) << 8);
524  // handle left column boundary case
525  hVals |= (*curLine << 16);
526  // init vertical average shift-buffer with current values average
527  vSums = ((*(prevLine++) + *(nextLine++)) << 7) & 0xFF00;
528  // handle left column boundary case
529  vSums |= ((*prevLine + *nextLine) << 15) & 0xFF0000;
530 
531  // store if line is odd or not
532  uint8_t yOdd = y & 1;
533  // the right column boundary case is not handled inside this loop
534  // thus the "639"
535  for (x = 0; x < frame_mode.width-1; ++x) {
536  // place next value in shift buffers
537  hVals |= *(curLine++);
538  vSums |= (*(prevLine++) + *(nextLine++)) >> 1;
539 
540  // calculate the horizontal sum as this sum is needed in
541  // any configuration
542  uint8_t hSum = ((uint8_t)(hVals >> 16) + (uint8_t)(hVals)) >> 1;
543 
544  if (yOdd == 0) {
545  if ((x & 1) == 0) {
546  // Configuration 1
547  *(dst++) = hSum; // r
548  *(dst++) = hVals >> 8; // g
549  *(dst++) = vSums >> 8; // b
550  } else {
551  // Configuration 2
552  *(dst++) = hVals >> 8;
553  *(dst++) = (hSum + (uint8_t)(vSums >> 8)) >> 1;
554  *(dst++) = ((uint8_t)(vSums >> 16) + (uint8_t)(vSums)) >> 1;
555  }
556  } else {
557  if ((x & 1) == 0) {
558  // Configuration 3
559  *(dst++) = ((uint8_t)(vSums >> 16) + (uint8_t)(vSums)) >> 1;
560  *(dst++) = (hSum + (uint8_t)(vSums >> 8)) >> 1;
561  *(dst++) = hVals >> 8;
562  } else {
563  // Configuration 4
564  *(dst++) = vSums >> 8;
565  *(dst++) = hVals >> 8;
566  *(dst++) = hSum;
567  }
568  }
569 
570  // shift the shift-buffers
571  hVals <<= 8;
572  vSums <<= 8;
573  } // end of for x loop
574  // right column boundary case, mirroring second last column
575  hVals |= (uint8_t)(hVals >> 16);
576  vSums |= (uint8_t)(vSums >> 16);
577 
578  // the horizontal sum simplifies to the second last column value
579  uint8_t hSum = (uint8_t)(hVals);
580 
581  if (yOdd == 0) {
582  if ((x & 1) == 0) {
583  *(dst++) = hSum;
584  *(dst++) = hVals >> 8;
585  *(dst++) = vSums >> 8;
586  } else {
587  *(dst++) = hVals >> 8;
588  *(dst++) = (hSum + (uint8_t)(vSums >> 8)) >> 1;
589  *(dst++) = vSums;
590  }
591  } else {
592  if ((x & 1) == 0) {
593  *(dst++) = vSums;
594  *(dst++) = (hSum + (uint8_t)(vSums >> 8)) >> 1;
595  *(dst++) = hVals >> 8;
596  } else {
597  *(dst++) = vSums >> 8;
598  *(dst++) = hVals >> 8;
599  *(dst++) = hSum;
600  }
601  }
602 
603  } // end of for y loop
604 }

660 static int send_cmd(freenect_device *dev, uint16_t cmd, void *cmdbuf, unsigned int cmd_len, void *replybuf, int reply_len)
661 {
662  freenect_context *ctx = dev->parent;
663  int res, actual_len;
664  uint8_t obuf[0x400];
665  uint8_t ibuf[0x200];
666  cam_hdr *chdr = (cam_hdr*)obuf;
667  cam_hdr *rhdr = (cam_hdr*)ibuf;
668 
669  if (cmd_len & 1 || cmd_len > (0x400 - sizeof(*chdr))) {
670  FN_ERROR("send_cmd: Invalid command length (0x%x)\n", cmd_len);
671  return -1;
672  }
673 
674  chdr->magic[0] = 0x47;
675  chdr->magic[1] = 0x4d;
676  chdr->cmd = fn_le16(cmd);
677  chdr->tag = fn_le16(dev->cam_tag);
678  chdr->len = fn_le16(cmd_len / 2);
679 
680  memcpy(obuf+sizeof(*chdr), cmdbuf, cmd_len);
681 
682  res = fnusb_control(&dev->usb_cam, 0x40, 0, 0, 0, obuf, cmd_len + sizeof(*chdr));
683  FN_SPEW("Control cmd=%04x tag=%04x len=%04x: %d\n", cmd, dev->cam_tag, cmd_len, res);
684  if (res < 0) {
685  FN_ERROR("send_cmd: Output control transfer failed (%d)\n", res);
686  return res;
687  }
688 
689  do {
690  actual_len = fnusb_control(&dev->usb_cam, 0xc0, 0, 0, 0, ibuf, 0x200);
691  FN_FLOOD("actual_len: %d\n", actual_len);
692  } while ((actual_len == 0) || (actual_len == 0x200));
693  FN_SPEW("Control reply: %d\n", res);
694  if (actual_len < (int)sizeof(*rhdr)) {
695  FN_ERROR("send_cmd: Input control transfer failed (%d)\n", res);
696  return res;
697  }
698  actual_len -= sizeof(*rhdr);
699 
700  if (rhdr->magic[0] != 0x52 || rhdr->magic[1] != 0x42) {
701  FN_ERROR("send_cmd: Bad magic %02x %02x\n", rhdr->magic[0], rhdr->magic[1]);
702  return -1;
703  }
704  if (rhdr->cmd != chdr->cmd) {
705  FN_ERROR("send_cmd: Bad cmd %02x != %02x\n", rhdr->cmd, chdr->cmd);
706  return -1;
707  }
708  if (rhdr->tag != chdr->tag) {
709  FN_ERROR("send_cmd: Bad tag %04x != %04x\n", rhdr->tag, chdr->tag);
710  return -1;
711  }
712  if (fn_le16(rhdr->len) != (actual_len/2)) {
713  FN_ERROR("send_cmd: Bad len %04x != %04x\n", fn_le16(rhdr->len), (int)(actual_len/2));
714  return -1;
715  }
716 
717  if (actual_len > reply_len) {
718  FN_WARNING("send_cmd: Data buffer is %d bytes long, but got %d bytes\n", reply_len, actual_len);
719  memcpy(replybuf, ibuf+sizeof(*rhdr), reply_len);
720  } else {
721  memcpy(replybuf, ibuf+sizeof(*rhdr), actual_len);
722  }
723 
724  dev->cam_tag++;
725 
726  return actual_len;
727 }
