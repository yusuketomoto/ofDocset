109  void chooseCentersRandom(int k, int* indices, int indices_length, int* centers, int& centers_length)
110  {
111  UniqueRandom r(indices_length);
112 
113  int index;
114  for (index=0; index<k; ++index) {
115  bool duplicate = true;
116  int rnd;
117  while (duplicate) {
118  duplicate = false;
119  rnd = r.next();
120  if (rnd<0) {
121  centers_length = index;
122  return;
123  }
124 
125  centers[index] = indices[rnd];
126 
127  for (int j=0; j<index; ++j) {
128  DistanceType sq = distance(dataset[centers[index]], dataset[centers[j]], dataset.cols);
129  if (sq<1e-16) {
130  duplicate = true;
131  }
132  }
133  }
134  }
135 
136  centers_length = index;
137  }

150  void chooseCentersGonzales(int k, int* indices, int indices_length, int* centers, int& centers_length)
151  {
152  int n = indices_length;
153 
154  int rnd = rand_int(n);
155  assert(rnd >=0 && rnd < n);
156 
157  centers[0] = indices[rnd];
158 
159  int index;
160  for (index=1; index<k; ++index) {

166  for (int i=1; i<index; ++i) {

201  void chooseCentersKMeanspp(int k, int* indices, int indices_length, int* centers, int& centers_length)
202  {
203  int n = indices_length;
204 
205  double currentPot = 0;
206  DistanceType* closestDistSq = new DistanceType[n];
207 
208  // Choose one random center and set the closestDistSq values
209  int index = rand_int(n);
210  assert(index >=0 && index < n);
211  centers[0] = indices[index];
212 
213  for (int i = 0; i < n; i++) {
214  closestDistSq[i] = distance(dataset[indices[i]], dataset[indices[index]], dataset.cols);
215  currentPot += closestDistSq[i];
216  }
217 
218 
219  const int numLocalTries = 1;
220 
221  // Choose each center
222  int centerCount;
223  for (centerCount = 1; centerCount < k; centerCount++) {
224 
225  // Repeat several trials
226  double bestNewPot = -1;
227  int bestNewIndex = 0;
228  for (int localTrial = 0; localTrial < numLocalTries; localTrial++) {
229 
230  // Choose our center - have to be slightly careful to return a valid answer even accounting
231  // for possible rounding errors
232  double randVal = rand_double(currentPot);
233  for (index = 0; index < n-1; index++) {
234  if (randVal <= closestDistSq[index]) break;
235  else randVal -= closestDistSq[index];
236  }
237 
238  // Compute the new potential
239  double newPot = 0;
240  for (int i = 0; i < n; i++) newPot += std::min( distance(dataset[indices[i]], dataset[indices[index]], dataset.cols), closestDistSq[i] );
241 
242  // Store the best result
243  if ((bestNewPot < 0)||(newPot < bestNewPot)) {
244  bestNewPot = newPot;
245  bestNewIndex = index;
246  }
247  }
248 
249  // Add the appropriate center
250  centers[centerCount] = indices[bestNewIndex];
251  currentPot = bestNewPot;
252  for (int i = 0; i < n; i++) closestDistSq[i] = std::min( distance(dataset[indices[i]], dataset[indices[bestNewIndex]], dataset.cols), closestDistSq[i] );
253  }
254 
255  centers_length = centerCount;
256 
257  delete[] closestDistSq;
258  }

416  void findNeighbors(ResultSet<DistanceType>& result, const ElementType* vec, const SearchParams& searchParams)
417  {
418 
419  int maxChecks = get_param(searchParams,"checks",32);
420 
421  // Priority queue storing intermediate branches in the best-bin-first search
422  Heap<BranchSt>* heap = new Heap<BranchSt>(size_);
423 
424  std::vector<bool> checked(size_,false);
425  int checks = 0;
426  for (int i=0; i<trees_; ++i) {
427  findNN(root[i], result, vec, checks, maxChecks, heap, checked);
428  }
429 
430  BranchSt branch;
431  while (heap->popMin(branch) && (checks<maxChecks || !result.full())) {
432  NodePtr node = branch.node;
433  findNN(node, result, vec, checks, maxChecks, heap, checked);
434  }
435  assert(result.full());
436 
437  delete heap;
438 
439  }

521  void computeLabels(int* indices, int indices_length, int* centers, int centers_length, int* labels, DistanceType& cost)
522  {
523  cost = 0;
524  for (int i=0; i<indices_length; ++i) {
525  ElementType* point = dataset[indices[i]];
526  DistanceType dist = distance(point, dataset[centers[0]], veclen_);
527  labels[i] = 0;
528  for (int j=1; j<centers_length; ++j) {
529  DistanceType new_dist = distance(point, dataset[centers[j]], veclen_);
530  if (dist>new_dist) {
531  labels[i] = j;
532  dist = new_dist;
533  }
534  }
535  cost += dist;

550  void computeClustering(NodePtr node, int* indices, int indices_length, int branching, int level)
551  {
552  node->size = indices_length;
553  node->level = level;
554 
555  if (indices_length < leaf_size_) { // leaf node
556  node->indices = indices;
557  std::sort(node->indices,node->indices+indices_length);
558  node->childs = NULL;
559  return;
560  }
561 
562  std::vector<int> centers(branching);
563  std::vector<int> labels(indices_length);
564 
565  int centers_length;
566  (this->*chooseCenters)(branching, indices, indices_length, &centers[0], centers_length);
567 
568  if (centers_length<branching) {
569  node->indices = indices;
570  std::sort(node->indices,node->indices+indices_length);
571  node->childs = NULL;
572  return;
573  }
574 
575 
576  // assign points to clusters
577  DistanceType cost;
578  computeLabels(indices, indices_length, &centers[0], centers_length, &labels[0], cost);
579 
580  node->childs = pool.allocate<NodePtr>(branching);
581  int start = 0;
582  int end = start;
583  for (int i=0; i<branching; ++i) {
584  for (int j=0; j<indices_length; ++j) {
585  if (labels[j]==i) {
586  std::swap(indices[j],indices[end]);
587  std::swap(labels[j],labels[end]);
588  end++;
589  }
590  }
591 
592  node->childs[i] = pool.allocate<Node>();
593  node->childs[i]->pivot = centers[i];
594  node->childs[i]->indices = NULL;
595  computeClustering(node->childs[i],indices+start, end-start, branching, level+1);
596  start=end;
597  }
598  }

615  void findNN(NodePtr node, ResultSet<DistanceType>& result, const ElementType* vec, int& checks, int maxChecks,
616  Heap<BranchSt>* heap, std::vector<bool>& checked)
617  {
618  if (node->childs==NULL) {
619  if (checks>=maxChecks) {
620  if (result.full()) return;
621  }
622  checks += node->size;
623  for (int i=0; i<node->size; ++i) {
624  int index = node->indices[i];
625  if (!checked[index]) {
626  DistanceType dist = distance(dataset[index], vec, veclen_);
627  result.addPoint(dist, index);
628  checked[index] = true;
629  }
630  }
631  }
632  else {
633  DistanceType* domain_distances = new DistanceType[branching_];
634  int best_index = 0;
635  domain_distances[best_index] = distance(vec, dataset[node->childs[best_index]->pivot], veclen_);
636  for (int i=1; i<branching_; ++i) {
637  domain_distances[i] = distance(vec, dataset[node->childs[i]->pivot], veclen_);
638  if (domain_distances[i]<domain_distances[best_index]) {
639  best_index = i;
640  }
641  }
642  for (int i=0; i<branching_; ++i) {
643  if (i!=best_index) {
644  heap->insert(BranchSt(node->childs[i],domain_distances[i]));
645  }
646  }
647  delete[] domain_distances;
648  findNN(node->childs[best_index],result,vec, checks, maxChecks, heap, checked);
649  }
650  }
