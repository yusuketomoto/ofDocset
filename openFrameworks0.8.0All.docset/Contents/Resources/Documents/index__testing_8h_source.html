48 inline int countCorrectMatches(int* neighbors, int* groundTruth, int n)
49 {
50  int count = 0;
51  for (int i=0; i<n; ++i) {
52  for (int k=0; k<n; ++k) {
53  if (neighbors[i]==groundTruth[k]) {
54  count++;

57  }
58  }

63 template <typename Distance>
64 typename Distance::ResultType computeDistanceRaport(const Matrix<typename Distance::ElementType>& inputData, typename Distance::ElementType* target,
65  int* neighbors, int* groundTruth, int veclen, int n, const Distance& distance)
66 {
67  typedef typename Distance::ResultType DistanceType;
68 
69  DistanceType ret = 0;
70  for (int i=0; i<n; ++i) {
71  DistanceType den = distance(inputData[groundTruth[i]], target, veclen);
72  DistanceType num = distance(inputData[neighbors[i]], target, veclen);
73 
74  if ((den==0)&&(num==0)) {
75  ret += 1;
76  }
77  else {
78  ret += num/den;
79  }
80  }
81 
82  return ret;
83 }

85 template <typename Distance>
86 float search_with_ground_truth(NNIndex<Distance>& index, const Matrix<typename Distance::ElementType>& inputData,
87  const Matrix<typename Distance::ElementType>& testData, const Matrix<int>& matches, int nn, int checks,
88  float& time, typename Distance::ResultType& dist, const Distance& distance, int skipMatches)
89 {
90  typedef typename Distance::ResultType DistanceType;
91 
92  if (matches.cols<size_t(nn)) {
93  Logger::info("matches.cols=%d, nn=%d\n",matches.cols,nn);
94 
95  throw FLANNException("Ground truth is not computed for as many neighbors as requested");
96  }
97 
98  KNNResultSet<DistanceType> resultSet(nn+skipMatches);
99  SearchParams searchParams(checks);
100 
101  int* indices = new int[nn+skipMatches];
102  DistanceType* dists = new DistanceType[nn+skipMatches];
103  int* neighbors = indices + skipMatches;
104 
105  int correct = 0;
106  DistanceType distR = 0;
107  StartStopTimer t;
108  int repeats = 0;
109  while (t.value<0.2) {
110  repeats++;
111  t.start();
112  correct = 0;
113  distR = 0;
114  for (size_t i = 0; i < testData.rows; i++) {
115  resultSet.init(indices, dists);
116  index.findNeighbors(resultSet, testData[i], searchParams);
117 
118  correct += countCorrectMatches(neighbors,matches[i], nn);
119  distR += computeDistanceRaport<Distance>(inputData, testData[i], neighbors, matches[i], testData.cols, nn, distance);
120  }
121  t.stop();
122  }
123  time = float(t.value/repeats);
124 
125  delete[] indices;
126  delete[] dists;
127 
128  float precicion = (float)correct/(nn*testData.rows);
129 
130  dist = distR/(testData.rows*nn);
131 
132  Logger::info("%8d %10.4g %10.5g %10.5g %10.5g\n",
133  checks, precicion, time, 1000.0 * time / testData.rows, dist);
134 
135  return precicion;
136 }

156 template <typename Distance>
157 float test_index_precision(NNIndex<Distance>& index, const Matrix<typename Distance::ElementType>& inputData,
158  const Matrix<typename Distance::ElementType>& testData, const Matrix<int>& matches,
159  float precision, int& checks, const Distance& distance, int nn = 1, int skipMatches = 0)
160 {
161  typedef typename Distance::ResultType DistanceType;
162  const float SEARCH_EPS = 0.001f;
163 
164  Logger::info(" Nodes Precision(%) Time(s) Time/vec(ms) Mean dist\n");
165  Logger::info("---------------------------------------------------------\n");
166 
167  int c2 = 1;
168  float p2;
169  int c1 = 1;
170  float p1;
171  float time;
172  DistanceType dist;
173 
174  p2 = search_with_ground_truth(index, inputData, testData, matches, nn, c2, time, dist, distance, skipMatches);
175 
176  if (p2>precision) {
177  Logger::info("Got as close as I can\n");
178  checks = c2;
179  return time;
180  }
181 
182  while (p2<precision) {
183  c1 = c2;
184  p1 = p2;
185  c2 *=2;
186  p2 = search_with_ground_truth(index, inputData, testData, matches, nn, c2, time, dist, distance, skipMatches);
187  }
188 
189  int cx;
190  float realPrecision;
191  if (fabs(p2-precision)>SEARCH_EPS) {
192  Logger::info("Start linear estimation\n");
193  // after we got to values in the vecinity of the desired precision
194  // use linear approximation get a better estimation
195 
196  cx = (c1+c2)/2;
197  realPrecision = search_with_ground_truth(index, inputData, testData, matches, nn, cx, time, dist, distance, skipMatches);
198  while (fabs(realPrecision-precision)>SEARCH_EPS) {
199 
200  if (realPrecision<precision) {
201  c1 = cx;
202  }
203  else {
204  c2 = cx;
205  }
206  cx = (c1+c2)/2;
207  if (cx==c1) {
208  Logger::info("Got as close as I can\n");
209  break;
210  }
211  realPrecision = search_with_ground_truth(index, inputData, testData, matches, nn, cx, time, dist, distance, skipMatches);
212  }
213 
214  c2 = cx;
215  p2 = realPrecision;
216 
217  }
218  else {
219  Logger::info("No need for linear estimation\n");
220  cx = c2;
221  realPrecision = p2;
222  }
223 
224  checks = cx;
225  return time;
