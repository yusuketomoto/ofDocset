49 /* MakeEdge creates a new pair of half-edges which form their own loop.
50 * No vertex or face structures are allocated, but these must be assigned
51 * before the current edge operation is completed.
52 */
53 static TESShalfEdge *MakeEdge( TESSmesh* mesh, TESShalfEdge *eNext )
54 {
55  TESShalfEdge *e;
56  TESShalfEdge *eSym;
57  TESShalfEdge *ePrev;
58  EdgePair *pair = (EdgePair *)bucketAlloc( mesh->edgeBucket );
59  if (pair == NULL) return NULL;
60 
61  e = &pair->e;
62  eSym = &pair->eSym;
63 
64  /* Make sure eNext points to the first edge of the edge pair */
65  if( eNext->Sym < eNext ) { eNext = eNext->Sym; }
66 
67  /* Insert in circular doubly-linked list before eNext.
68  * Note that the prev pointer is stored in Sym->next.
69  */
70  ePrev = eNext->Sym->next;
71  eSym->next = ePrev;
72  ePrev->Sym->next = e;
73  e->next = eNext;
74  eNext->Sym->next = eSym;
75 
76  e->Sym = eSym;
77  e->Onext = e;
78  e->Lnext = eSym;
79  e->Org = NULL;
80  e->Lface = NULL;
81  e->winding = 0;
82  e->activeRegion = NULL;
83 
84  eSym->Sym = e;
85  eSym->Onext = eSym;
86  eSym->Lnext = e;
87  eSym->Org = NULL;
88  eSym->Lface = NULL;
89  eSym->winding = 0;
90  eSym->activeRegion = NULL;
91 
92  return e;
93 }

95 /* Splice( a, b ) is best described by the Guibas/Stolfi paper or the
96 * CS348a notes (see mesh.h). Basically it modifies the mesh so that
97 * a->Onext and b->Onext are exchanged. This can have various effects
98 * depending on whether a and b belong to different face or vertex rings.
99 * For more explanation see tessMeshSplice() below.
100 */
101 static void Splice( TESShalfEdge *a, TESShalfEdge *b )
102 {
103  TESShalfEdge *aOnext = a->Onext;
104  TESShalfEdge *bOnext = b->Onext;
105 
106  aOnext->Sym->Lnext = b;
107  bOnext->Sym->Lnext = a;
108  a->Onext = bOnext;
109  b->Onext = aOnext;
110 }

112 /* MakeVertex( newVertex, eOrig, vNext ) attaches a new vertex and makes it the
113 * origin of all edges in the vertex loop to which eOrig belongs. "vNext" gives
114 * a place to insert the new vertex in the global vertex list. We insert
115 * the new vertex *before* vNext so that algorithms which walk the vertex
116 * list will not see the newly created vertices.
117 */
118 static void MakeVertex( TESSvertex *newVertex,
119  TESShalfEdge *eOrig, TESSvertex *vNext )
120 {
121  TESShalfEdge *e;
122  TESSvertex *vPrev;
123  TESSvertex *vNew = newVertex;
124 
125  if(vNew == NULL) return;
126 
127  /* insert in circular doubly-linked list before vNext */
128  vPrev = vNext->prev;
129  vNew->prev = vPrev;
130  vPrev->next = vNew;
131  vNew->next = vNext;
132  vNext->prev = vNew;
133 
134  vNew->anEdge = eOrig;
135  /* leave coords, s, t undefined */
136 
137  /* fix other edges on this vertex loop */
138  e = eOrig;
139  do {
140  e->Org = vNew;
141  e = e->Onext;
142  } while( e != eOrig );
143 }

145 /* MakeFace( newFace, eOrig, fNext ) attaches a new face and makes it the left
146 * face of all edges in the face loop to which eOrig belongs. "fNext" gives
147 * a place to insert the new face in the global face list. We insert
148 * the new face *before* fNext so that algorithms which walk the face
149 * list will not see the newly created faces.
150 */
151 static void MakeFace( TESSface *newFace, TESShalfEdge *eOrig, TESSface *fNext )
152 {
153  TESShalfEdge *e;
154  TESSface *fPrev;
155  TESSface *fNew = newFace;
156 
157  if(fNew == NULL) return;
158 
159  /* insert in circular doubly-linked list before fNext */
160  fPrev = fNext->prev;
161  fNew->prev = fPrev;
162  fPrev->next = fNew;
163  fNew->next = fNext;
164  fNext->prev = fNew;
165 
166  fNew->anEdge = eOrig;
167  fNew->trail = NULL;
168  fNew->marked = FALSE;
169 
170  /* The new face is marked "inside" if the old one was. This is a
171  * convenience for the common case where a face has been split in two.
172  */
173  fNew->inside = fNext->inside;
174 
175  /* fix other edges on this face loop */
176  e = eOrig;
177  do {
178  e->Lface = fNew;
179  e = e->Lnext;
180  } while( e != eOrig );
181 }

183 /* KillEdge( eDel ) destroys an edge (the half-edges eDel and eDel->Sym),
184 * and removes from the global edge list.
185 */
186 static void KillEdge( TESSmesh *mesh, TESShalfEdge *eDel )
187 {
188  TESShalfEdge *ePrev, *eNext;
189 
190  /* Half-edges are allocated in pairs, see EdgePair above */
191  if( eDel->Sym < eDel ) { eDel = eDel->Sym; }
192 
193  /* delete from circular doubly-linked list */
194  eNext = eDel->next;
195  ePrev = eDel->Sym->next;
196  eNext->Sym->next = ePrev;
197  ePrev->Sym->next = eNext;
198 
199  bucketFree( mesh->edgeBucket, eDel );
200 }

203 /* KillVertex( vDel ) destroys a vertex and removes it from the global
204 * vertex list. It updates the vertex loop to point to a given new vertex.
205 */
206 static void KillVertex( TESSmesh *mesh, TESSvertex *vDel, TESSvertex *newOrg )
207 {
208  TESShalfEdge *e, *eStart = vDel->anEdge;
209  TESSvertex *vPrev, *vNext;
210 
211  /* change the origin of all affected edges */
212  e = eStart;
213  do {
214  e->Org = newOrg;
215  e = e->Onext;
216  } while( e != eStart );
217 
218  /* delete from circular doubly-linked list */
219  vPrev = vDel->prev;
220  vNext = vDel->next;
221  vNext->prev = vPrev;
222  vPrev->next = vNext;
223 
224  bucketFree( mesh->vertexBucket, vDel );
225 }

254 /* tessMeshMakeEdge creates one edge, two vertices, and a loop (face).
255 * The loop consists of the two new half-edges.
256 */
257 TESShalfEdge *tessMeshMakeEdge( TESSmesh *mesh )
258 {
259  TESSvertex *newVertex1 = (TESSvertex*)bucketAlloc(mesh->vertexBucket);
260  TESSvertex *newVertex2 = (TESSvertex*)bucketAlloc(mesh->vertexBucket);
261  TESSface *newFace = (TESSface*)bucketAlloc(mesh->faceBucket);
262  TESShalfEdge *e;
263 
264  /* if any one is null then all get freed */
265  if (newVertex1 == NULL || newVertex2 == NULL || newFace == NULL) {
266  if (newVertex1 != NULL) bucketFree( mesh->vertexBucket, newVertex1 );
267  if (newVertex2 != NULL) bucketFree( mesh->vertexBucket, newVertex2 );
268  if (newFace != NULL) bucketFree( mesh->faceBucket, newFace );
269  return NULL;
270  }
271 
272  e = MakeEdge( mesh, &mesh->eHead );
273  if (e == NULL) return NULL;
274 
275  MakeVertex( newVertex1, e, &mesh->vHead );
276  MakeVertex( newVertex2, e->Sym, &mesh->vHead );
277  MakeFace( newFace, e, &mesh->fHead );
278  return e;
279 }

282 /* tessMeshSplice( eOrg, eDst ) is the basic operation for changing the
283 * mesh connectivity and topology. It changes the mesh so that
284 * eOrg->Onext <- OLD( eDst->Onext )
285 * eDst->Onext <- OLD( eOrg->Onext )
286 * where OLD(...) means the value before the meshSplice operation.
287 *
288 * This can have two effects on the vertex structure:
289 * - if eOrg->Org != eDst->Org, the two vertices are merged together
290 * - if eOrg->Org == eDst->Org, the origin is split into two vertices
291 * In both cases, eDst->Org is changed and eOrg->Org is untouched.
292 *
293 * Similarly (and independently) for the face structure,
294 * - if eOrg->Lface == eDst->Lface, one loop is split into two
295 * - if eOrg->Lface != eDst->Lface, two distinct loops are joined into one
296 * In both cases, eDst->Lface is changed and eOrg->Lface is unaffected.
297 *
298 * Some special cases:
299 * If eDst == eOrg, the operation has no effect.
300 * If eDst == eOrg->Lnext, the new face will have a single edge.
301 * If eDst == eOrg->Lprev, the old face will have a single edge.
302 * If eDst == eOrg->Onext, the new vertex will have a single edge.
303 * If eDst == eOrg->Oprev, the old vertex will have a single edge.
304 */
305 int tessMeshSplice( TESSmesh* mesh, TESShalfEdge *eOrg, TESShalfEdge *eDst )
306 {
307  int joiningLoops = FALSE;
308  int joiningVertices = FALSE;
309 
310  if( eOrg == eDst ) return 1;
311 
312  if( eDst->Org != eOrg->Org ) {
313  /* We are merging two disjoint vertices -- destroy eDst->Org */
314  joiningVertices = TRUE;
315  KillVertex( mesh, eDst->Org, eOrg->Org );
316  }
317  if( eDst->Lface != eOrg->Lface ) {
318  /* We are connecting two disjoint loops -- destroy eDst->Lface */
319  joiningLoops = TRUE;
320  KillFace( mesh, eDst->Lface, eOrg->Lface );
321  }
322 
323  /* Change the edge structure */
324  Splice( eDst, eOrg );
325 
326  if( ! joiningVertices ) {
327  TESSvertex *newVertex = (TESSvertex*)bucketAlloc( mesh->vertexBucket );
328  if (newVertex == NULL) return 0;
329 
330  /* We split one vertex into two -- the new vertex is eDst->Org.
331  * Make sure the old vertex points to a valid half-edge.
332  */
333  MakeVertex( newVertex, eDst, eOrg->Org );
334  eOrg->Org->anEdge = eOrg;
335  }
336  if( ! joiningLoops ) {
337  TESSface *newFace = (TESSface*)bucketAlloc( mesh->faceBucket );
338  if (newFace == NULL) return 0;
339 
340  /* We split one loop into two -- the new loop is eDst->Lface.
341  * Make sure the old face points to a valid half-edge.
342  */
343  MakeFace( newFace, eDst, eOrg->Lface );
344  eOrg->Lface->anEdge = eOrg;
345  }
346 
347  return 1;
348 }

351 /* tessMeshDelete( eDel ) removes the edge eDel. There are several cases:
352 * if (eDel->Lface != eDel->Rface), we join two loops into one; the loop
353 * eDel->Lface is deleted. Otherwise, we are splitting one loop into two;
354 * the newly created loop will contain eDel->Dst. If the deletion of eDel
355 * would create isolated vertices, those are deleted as well.
356 *
357 * This function could be implemented as two calls to tessMeshSplice
358 * plus a few calls to memFree, but this would allocate and delete
359 * unnecessary vertices and faces.
360 */
361 int tessMeshDelete( TESSmesh *mesh, TESShalfEdge *eDel )
362 {
363  TESShalfEdge *eDelSym = eDel->Sym;
364  int joiningLoops = FALSE;
365 
366  /* First step: disconnect the origin vertex eDel->Org. We make all
367  * changes to get a consistent mesh in this "intermediate" state.
368  */
369  if( eDel->Lface != eDel->Rface ) {
370  /* We are joining two loops into one -- remove the left face */
371  joiningLoops = TRUE;
372  KillFace( mesh, eDel->Lface, eDel->Rface );
373  }
374 
375  if( eDel->Onext == eDel ) {
376  KillVertex( mesh, eDel->Org, NULL );
377  } else {
378  /* Make sure that eDel->Org and eDel->Rface point to valid half-edges */
379  eDel->Rface->anEdge = eDel->Oprev;
380  eDel->Org->anEdge = eDel->Onext;
381 
382  Splice( eDel, eDel->Oprev );
383  if( ! joiningLoops ) {
384  TESSface *newFace= (TESSface*)bucketAlloc( mesh->faceBucket );
385  if (newFace == NULL) return 0;
386 
387  /* We are splitting one loop into two -- create a new loop for eDel. */
388  MakeFace( newFace, eDel, eDel->Lface );
389  }
390  }
391 
392  /* Claim: the mesh is now in a consistent state, except that eDel->Org
393  * may have been deleted. Now we disconnect eDel->Dst.
394  */
395  if( eDelSym->Onext == eDelSym ) {
396  KillVertex( mesh, eDelSym->Org, NULL );
397  KillFace( mesh, eDelSym->Lface, NULL );
398  } else {
399  /* Make sure that eDel->Dst and eDel->Lface point to valid half-edges */
400  eDel->Lface->anEdge = eDelSym->Oprev;
401  eDelSym->Org->anEdge = eDelSym->Onext;
402  Splice( eDelSym, eDelSym->Oprev );
403  }
404 
405  /* Any isolated vertices or faces have already been freed. */
406  KillEdge( mesh, eDel );
407 
408  return 1;
409 }

419 /* tessMeshAddEdgeVertex( eOrg ) creates a new edge eNew such that
420 * eNew == eOrg->Lnext, and eNew->Dst is a newly created vertex.
421 * eOrg and eNew will have the same left face.
422 */
423 TESShalfEdge *tessMeshAddEdgeVertex( TESSmesh *mesh, TESShalfEdge *eOrg )
424 {
425  TESShalfEdge *eNewSym;

427  if (eNew == NULL) return NULL;
428 
429  eNewSym = eNew->Sym;
430 
431  /* Connect the new edge appropriately */
432  Splice( eNew, eOrg->Lnext );
433 
434  /* Set the vertex and face information */
435  eNew->Org = eOrg->Dst;
436  {

438  if (newVertex == NULL) return NULL;
439

441  }

448 /* tessMeshSplitEdge( eOrg ) splits eOrg into two edges eOrg and eNew,
449 * such that eNew == eOrg->Lnext. The new vertex is eOrg->Dst == eNew->Org.
450 * eOrg and eNew will have the same left face.
451 */
452 TESShalfEdge *tessMeshSplitEdge( TESSmesh *mesh, TESShalfEdge *eOrg )
453 {
454  TESShalfEdge *eNew;
455  TESShalfEdge *tempHalfEdge= tessMeshAddEdgeVertex( mesh, eOrg );
456  if (tempHalfEdge == NULL) return NULL;
457 
458  eNew = tempHalfEdge->Sym;
459 
460  /* Disconnect eOrg from eOrg->Dst and connect it to eNew->Org */
461  Splice( eOrg->Sym, eOrg->Sym->Oprev );
462  Splice( eOrg->Sym, eNew );
463 
464  /* Set the vertex and face information */
465  eOrg->Dst = eNew->Org;
466  eNew->Dst->anEdge = eNew->Sym; /* may have pointed to eOrg->Sym */
467  eNew->Rface = eOrg->Rface;
468  eNew->winding = eOrg->winding; /* copy old winding information */
469  eNew->Sym->winding = eOrg->Sym->winding;
470 
471  return eNew;

475 /* tessMeshConnect( eOrg, eDst ) creates a new edge from eOrg->Dst
476 * to eDst->Org, and returns the corresponding half-edge eNew.
477 * If eOrg->Lface == eDst->Lface, this splits one loop into two,
478 * and the newly created loop is eNew->Lface. Otherwise, two disjoint
479 * loops are merged into one, and the loop eDst->Lface is destroyed.
480 *
481 * If (eOrg == eDst), the new face will have only two edges.
482 * If (eOrg->Lnext == eDst), the old face is reduced to a single edge.
483 * If (eOrg->Lnext->Lnext == eDst), the old face is reduced to two edges.
484 */
485 TESShalfEdge *tessMeshConnect( TESSmesh *mesh, TESShalfEdge *eOrg, TESShalfEdge *eDst )
486 {
487  TESShalfEdge *eNewSym;
488  int joiningLoops = FALSE;
489  TESShalfEdge *eNew = MakeEdge( mesh, eOrg );
490  if (eNew == NULL) return NULL;
491 
492  eNewSym = eNew->Sym;
493 
494  if( eDst->Lface != eOrg->Lface ) {
495  /* We are connecting two disjoint loops -- destroy eDst->Lface */
496  joiningLoops = TRUE;
497  KillFace( mesh, eDst->Lface, eOrg->Lface );
498  }
499 
500  /* Connect the new edge appropriately */
501  Splice( eNew, eOrg->Lnext );
502  Splice( eNewSym, eDst );
503 
504  /* Set the vertex and face information */
505  eNew->Org = eOrg->Dst;
506  eNewSym->Org = eDst->Org;
507  eNew->Lface = eNewSym->Lface = eOrg->Lface;
508 
509  /* Make sure the old face points to a valid half-edge */
510  eOrg->Lface->anEdge = eNewSym;
511 
512  if( ! joiningLoops ) {
513  TESSface *newFace= (TESSface*)bucketAlloc( mesh->faceBucket );
514  if (newFace == NULL) return NULL;
515 
516  /* We split one loop into two -- the new loop is eNew->Lface */
517  MakeFace( newFace, eNew, eOrg->Lface );
518  }
519  return eNew;
520 }

525 /* tessMeshZapFace( fZap ) destroys a face and removes it from the
526 * global face list. All edges of fZap will have a NULL pointer as their
527 * left face. Any edges which also have a NULL pointer as their right face
528 * are deleted entirely (along with any isolated vertices this produces).
529 * An entire mesh can be deleted by zapping its faces, one at a time,
530 * in any order. Zapped faces cannot be used in further mesh operations!
531 */
532 void tessMeshZapFace( TESSmesh *mesh, TESSface *fZap )
533 {
534  TESShalfEdge *eStart = fZap->anEdge;
535  TESShalfEdge *e, *eNext, *eSym;
536  TESSface *fPrev, *fNext;
537 
538  /* walk around face, deleting edges whose right face is also NULL */
539  eNext = eStart->Lnext;
540  do {
541  e = eNext;
542  eNext = e->Lnext;
543 
544  e->Lface = NULL;
545  if( e->Rface == NULL ) {
546  /* delete the edge -- see TESSmeshDelete above */
547 
548  if( e->Onext == e ) {
549  KillVertex( mesh, e->Org, NULL );
550  } else {
551  /* Make sure that e->Org points to a valid half-edge */
552  e->Org->anEdge = e->Onext;
553  Splice( e, e->Oprev );
554  }
555  eSym = e->Sym;
556  if( eSym->Onext == eSym ) {
557  KillVertex( mesh, eSym->Org, NULL );
558  } else {
559  /* Make sure that eSym->Org points to a valid half-edge */
560  eSym->Org->anEdge = eSym->Onext;
561  Splice( eSym, eSym->Oprev );
562  }
563  KillEdge( mesh, e );
564  }
565  } while( e != eStart );
566 
567  /* delete from circular doubly-linked list */
568  fPrev = fZap->prev;
569  fNext = fZap->next;
570  fNext->prev = fPrev;
571  fPrev->next = fNext;
572 
573  bucketFree( mesh->faceBucket, fZap );
574 }

577 /* tessMeshNewMesh() creates a new mesh with no edges, no vertices,
578 * and no loops (what we usually call a "face").
579 */
580 TESSmesh *tessMeshNewMesh( TESSalloc* alloc )
581 {
582  TESSvertex *v;
583  TESSface *f;
584  TESShalfEdge *e;
585  TESShalfEdge *eSym;
586  TESSmesh *mesh = (TESSmesh *)alloc->memalloc( alloc->userData, sizeof( TESSmesh ));
587  if (mesh == NULL) {
588  return NULL;
589  }
590 
591  if (alloc->meshEdgeBucketSize < 16)
592  alloc->meshEdgeBucketSize = 16;
593  if (alloc->meshEdgeBucketSize > 4096)
594  alloc->meshEdgeBucketSize = 4096;
595 
596  if (alloc->meshVertexBucketSize < 16)
597  alloc->meshVertexBucketSize = 16;
598  if (alloc->meshVertexBucketSize > 4096)
599  alloc->meshVertexBucketSize = 4096;
600 
601  if (alloc->meshFaceBucketSize < 16)
602  alloc->meshFaceBucketSize = 16;
603  if (alloc->meshFaceBucketSize > 4096)

607  mesh->vertexBucket = createBucketAlloc( alloc, "Mesh Vertices", sizeof(TESSvertex), alloc->meshVertexBucketSize );
608  mesh->faceBucket = createBucketAlloc( alloc, "Mesh Faces", sizeof(TESSface), alloc->meshFaceBucketSize );
609 
610  v = &mesh->vHead;
611  f = &mesh->fHead;
612  e = &mesh->eHead;
613  eSym = &mesh->eHeadSym;
614

616  v->anEdge = NULL;
617

698 int tessMeshMergeConvexFaces( TESSmesh *mesh, int maxVertsPerFace )
699 {
700  TESSface *f;
701  TESShalfEdge *eCur, *eNext, *eSym;
702  TESSvertex *vStart;
703  int curNv, symNv;
704 
705  for( f = mesh->fHead.next; f != &mesh->fHead; f = f->next )
706  {
707  // Skip faces which are outside the result.
708  if( !f->inside )
709  continue;
710 
711  eCur = f->anEdge;
712  vStart = eCur->Org;
713 
714  while (1)
715  {
716  eNext = eCur->Lnext;
717  eSym = eCur->Sym;
718 
719  // Try to merge if the neighbour face is valid.
720  if( eSym && eSym->Lface && eSym->Lface->inside )
721  {
722  // Try to merge the neighbour faces if the resulting polygons
723  // does not exceed maximum number of vertices.
724  curNv = CountFaceVerts( f );
725  symNv = CountFaceVerts( eSym->Lface );
726  if( (curNv+symNv-2) <= maxVertsPerFace )
727  {
728  // Merge if the resulting poly is convex.
729  if( VertCCW( eCur->Lprev->Org, eCur->Org, eSym->Lnext->Lnext->Org ) &&
730  VertCCW( eSym->Lprev->Org, eSym->Org, eCur->Lnext->Lnext->Org ) )
731  {
732  eNext = eSym->Lnext;
733  if( !tessMeshDelete( mesh, eSym ) )
734  return 0;
735  eCur = 0;
736  }
737  }
738  }
739 
740  if( eCur && eCur->Lnext->Org == vStart )
741  break;
742 
743  // Continue to next edge.
744  eCur = eNext;
745  }
746  }
747 
748  return 1;
749 }
