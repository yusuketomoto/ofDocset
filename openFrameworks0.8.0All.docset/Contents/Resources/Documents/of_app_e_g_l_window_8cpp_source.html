490 bool ofAppEGLWindow::createSurface() {
491 
492  EGLNativeWindowType nativeWindow = getNativeWindow();
493  EGLNativeDisplayType display = getNativeDisplay();
494 
495  ofLogNotice("ofAppEGLWindow") << "createSurface(): setting up EGL Display";
496  // get an EGL eglDisplay connection
497 
498  isSurfaceInited = false;
499 
500  EGLint result;
501 
502  if(display==0){
503  eglDisplay = eglGetDisplay(EGL_DEFAULT_DISPLAY);
504  }else{
505  eglDisplay = eglGetDisplay(display);
506  }
507 
508  if(eglDisplay == EGL_NO_DISPLAY) {
509  ofLogNotice("ofAppEGLWindow") << "createSurface(): eglGetDisplay returned: " << eglDisplay;
510  return false;
511  }else{
512  ofLogNotice("ofAppEGLWindow") << "createSurface(): EGL Display correctly set";
513  }
514 
515  // initialize the EGL eglDisplay connection
516  result = eglInitialize(eglDisplay,
517  &eglVersionMajor,
518  &eglVersionMinor);
519 
520  if(result == EGL_BAD_DISPLAY) {
521 // eglDisplay is not an EGL connection
522  ofLogError("ofAppEGLWindow") << "createSurface(): eglInitialize returned EGL_BAD_DISPLAY";
523  return false;
524  } else if(result == EGL_NOT_INITIALIZED) {
525  // eglDisplay cannot be intitialized
526  ofLogError("ofAppEGLWindow") << "createSurface(): eglInitialize returned EGL_NOT_INITIALIZED";
527  return false;
528  } else if(result == EGL_FALSE) {
529  // eglinitialize was not initialiezd
530  ofLogError("ofAppEGLWindow") << "createSurface(): eglInitialize returned EGL_FALSE";
531  return false;
532  } else {
533  // result == EGL_TRUE
534  // success!
535  }
536 
537  EGLint glesVersion;
538  int glesVersionForContext;
539 
540  if(ofGetCurrentRenderer()) {
541  ofLogNotice("ofAppEGLWindow") << "createSurface(): current renderer type: " << ofGetCurrentRenderer()->getType();
542  } else {
543  ofLogNotice("ofAppEGLWindow") << "createSurface(): no current renderer selected";
544  }
545 
546  if(this->glesVersion==2){
547  glesVersion = EGL_OPENGL_ES2_BIT;
548  glesVersionForContext = 2;
549  ofLogNotice("ofAppEGLWindow") << "createSurface(): GLES2 renderer detected";
550  }else{
551  glesVersion = EGL_OPENGL_ES_BIT;
552  glesVersionForContext = 1;
553  ofLogNotice("ofAppEGLWindow") << "createSurface(): default renderer detected";
554  }
555 
556  ofEGLAttributeListIterator iter, iterEnd;
557  int i;
558 
559  // each attribute has 2 values, and we need one extra for the EGL_NONE terminator
560  EGLint attribute_list_framebuffer_config[settings.frameBufferAttributes.size() * 2 + 3];
561 
562  iter = settings.frameBufferAttributes.begin();
563  iterEnd = settings.frameBufferAttributes.end();
564  i = 0;
565  for(; iter != iterEnd; iter++) {
566  attribute_list_framebuffer_config[i++] = iter->first;
567  attribute_list_framebuffer_config[i++] = iter->second;
568  }
569  attribute_list_framebuffer_config[i++] = EGL_RENDERABLE_TYPE;
570  attribute_list_framebuffer_config[i++] = glesVersion; //openGL ES version
571  attribute_list_framebuffer_config[i] = EGL_NONE; // add the terminator
572 
573  EGLint num_configs;
574 
575  // get an appropriate EGL frame buffer configuration
576  // http://www.khronos.org/registry/egl/sdk/docs/man/xhtml/eglChooseConfig.html
577  result = eglChooseConfig(eglDisplay,
578  attribute_list_framebuffer_config,
579  &eglConfig,
580  1, // we only want the first one. if we want more,
581  // we need to pass in an array.
582  // we are optimistic and don't give it more chances
583  // to find a good configuration
584  &num_configs);
585 
586  if(result == EGL_FALSE) {
587  EGLint error = eglGetError();
588  ofLogError("ofAppEGLWindow") << "createSurface(): error finding valid configuration based on settings: " << eglErrorString(error);
589  return false;
590  }
591 
592  if(num_configs <= 0 || eglConfig == NULL) {
593  ofLogError("ofAppEGLWindow") << "createSurface(): no matching configs were found, num_configs: " << num_configs;
594  return false;
595  }
596 
597 
598  // each attribute has 2 values, and we need one extra for the EGL_NONE terminator
599  EGLint attribute_list_window_surface[settings.windowSurfaceAttributes.size() * 2 + 1];
600 
601  iter = settings.windowSurfaceAttributes.begin();
602  iterEnd = settings.windowSurfaceAttributes.end();
603 
604  i = 0;
605  for(; iter != iterEnd; iter++) {
606  attribute_list_window_surface[i++] = iter->first;
607  attribute_list_window_surface[i++] = iter->second;
608  }
609  attribute_list_window_surface[i] = EGL_NONE; // add the terminator
610 
611  // create a surface
612  eglSurface = eglCreateWindowSurface( eglDisplay, // our display handle
613  eglConfig, // our first config
614  nativeWindow, // our native window
615  attribute_list_window_surface); // surface attribute list
616 
617  if(eglSurface == EGL_NO_SURFACE) {
618  EGLint error = eglGetError();
619  switch(error) {
620  case EGL_BAD_MATCH:
621  ofLogError("ofAppEGLWindow") << "createSurface(): error creating surface: EGL_BAD_MATCH " << eglErrorString(error);
622  ofLogError("ofAppEGLWindow") << "createSurface(): check window and EGLConfig attributes to determine compatibility, ";
623  ofLogError("ofAppEGLWindow") << "createSurface(): or verify that the EGLConfig supports rendering to a window";
624  break;
625  case EGL_BAD_CONFIG:
626  ofLogError("ofAppEGLWindow") << "createSurface(): error creating surface: EGL_BAD_CONFIG " << eglErrorString(error);
627  ofLogError("ofAppEGLWindow") << "createSurface(): verify that provided EGLConfig is valid";
628  break;
629  case EGL_BAD_NATIVE_WINDOW:
630  ofLogError("ofAppEGLWindow") << "createSurface(): error creating surface: EGL_BAD_NATIVE_WINDOW " << eglErrorString(error);
631  ofLogError("ofAppEGLWindow") << "createSurface(): verify that provided EGLNativeWindow is valid";
632  break;
633  case EGL_BAD_ALLOC:
634  ofLogError("ofAppEGLWindow") << "createSurface(): error creating surface: EGL_BAD_ALLOC " << eglErrorString(error);
635  ofLogError("ofAppEGLWindow") << "createSurface(): not enough resources available";
636  break;
637  default:
638  ofLogError("ofAppEGLWindow") << "createSurface(): error creating surface: << " << error << eglErrorString(error);
639  }
640 
641  return false;
642  }else{
643  ofLogNotice("ofAppEGLWindow") << "createSurface(): surface created correctly";
644  }
645 
646  // get an appropriate EGL frame buffer configuration
647  result = eglBindAPI(EGL_OPENGL_ES_API);
648 
649  if(result == EGL_FALSE) {
650  ofLogError("ofAppEGLWindow") << "createSurface(): error binding API: " << eglErrorString(eglGetError());
651  return false;
652  }else{
653  ofLogNotice("ofAppEGLWindow") << "createSurface(): API bound correctly";
654  }
655 
656  // create an EGL rendering eglContext
657  EGLint attribute_list_surface_context[] = {
658  EGL_CONTEXT_CLIENT_VERSION, glesVersionForContext,
659  EGL_NONE
660  };
661 
662  eglContext = eglCreateContext(eglDisplay,
663  eglConfig,
664  EGL_NO_CONTEXT,
665  attribute_list_surface_context);
666 
667  if(eglContext == EGL_NO_CONTEXT) {
668  EGLint error = eglGetError();
669  if(error == EGL_BAD_CONFIG) {
670  ofLogError("ofAppEGLWindow") << "createSurface(): error creating context: EGL_BAD_CONFIG " << eglErrorString(error);
671  return false;
672  } else {
673  ofLogError("ofAppEGLWindow") << "createSurface(): error creating context: " << error << " " << eglErrorString(error);
674  return false;
675  }
676  }
677 
678  // connect the eglContext to the eglSurface
679  result = eglMakeCurrent(eglDisplay,
680  eglSurface, // draw surface
681  eglSurface, // read surface
682  eglContext);
683 
684  if(eglContext == EGL_FALSE) {
685  EGLint error = eglGetError();
686  ofLogError("ofAppEGLWindow") << "createSurface(): couldn't making current surface: " << eglErrorString(error);
687  return false;
688  }
689 
690  // Set background color and clear buffers
691  glClearColor(settings.initialClearColor.r / 255.0f,
692  settings.initialClearColor.g / 255.0f,
693  settings.initialClearColor.b / 255.0f,
694  settings.initialClearColor.a / 255.0f);
695  glClear( GL_COLOR_BUFFER_BIT );
696  glClear( GL_DEPTH_BUFFER_BIT );
697 
698  ofLogNotice("ofAppEGLWindow") << "createSurface(): -----EGL-----";
699  ofLogNotice("ofAppEGLWindow") << "createSurface(): EGL_VERSION_MAJOR = " << eglVersionMajor;
700  ofLogNotice("ofAppEGLWindow") << "createSurface(): EGL_VERSION_MINOR = " << eglVersionMinor;
701  ofLogNotice("ofAppEGLWindow") << "createSurface(): EGL_CLIENT_APIS = " << eglQueryString(eglDisplay, EGL_CLIENT_APIS);
702  ofLogNotice("ofAppEGLWindow") << "createSurface(): EGL_VENDOR = " << eglQueryString(eglDisplay, EGL_VENDOR);
703  ofLogNotice("ofAppEGLWindow") << "createSurface(): EGL_VERSION = " << eglQueryString(eglDisplay, EGL_VERSION);
704  ofLogNotice("ofAppEGLWindow") << "createSurface(): EGL_EXTENSIONS = " << eglQueryString(eglDisplay, EGL_EXTENSIONS);
705  ofLogNotice("ofAppEGLWindow") << "createSurface(): GL_RENDERER = " << glGetString(GL_RENDERER);
706  ofLogNotice("ofAppEGLWindow") << "createSurface(): GL_VERSION = " << glGetString(GL_VERSION);
707  ofLogNotice("ofAppEGLWindow") << "createSurface(): GL_VENDOR = " << glGetString(GL_VENDOR);

2003 static KeySym KeyCodeToKeySym(Display * display, KeyCode keycode, unsigned int event_mask) {
2004  KeySym keysym = NoSymbol;
2005 
2006  //Get the map
2007  XkbDescPtr keyboard_map = XkbGetMap(display, XkbAllClientInfoMask, XkbUseCoreKbd);
2008  if (keyboard_map) {
2009  //What is diff between XkbKeyGroupInfo and XkbKeyNumGroups?
2010  unsigned char info = XkbKeyGroupInfo(keyboard_map, keycode);
2011  unsigned int num_groups = XkbKeyNumGroups(keyboard_map, keycode);
2012 
2013  //Get the group
2014  unsigned int group = 0x00;
2015  switch (XkbOutOfRangeGroupAction(info)) {
2016  case XkbRedirectIntoRange:
2017  /* If the RedirectIntoRange flag is set, the four least significant
2018  * bits of the groups wrap control specify the index of a group to
2019  * which all illegal groups correspond. If the specified group is
2020  * also out of range, all illegal groups map to Group1.
2021  */
2022  group = XkbOutOfRangeGroupInfo(info);
2023  if (group >= num_groups) {
2024  group = 0;
2025  }
2026  break;
2027 
2028  case XkbClampIntoRange:
2029  /* If the ClampIntoRange flag is set, out-of-range groups correspond
2030  * to the nearest legal group. Effective groups larger than the
2031  * highest supported group are mapped to the highest supported group;
2032  * effective groups less than Group1 are mapped to Group1 . For
2033  * example, a key with two groups of symbols uses Group2 type and
2034  * symbols if the global effective group is either Group3 or Group4.
2035  */
2036  group = num_groups - 1;
2037  break;
2038 
2039  case XkbWrapIntoRange:
2040  /* If neither flag is set, group is wrapped into range using integer
2041  * modulus. For example, a key with two groups of symbols for which
2042  * groups wrap uses Group1 symbols if the global effective group is
2043  * Group3 or Group2 symbols if the global effective group is Group4.
2044  */
2045  default:
2046  if (num_groups != 0) {
2047  group %= num_groups;
2048  }
2049  break;
2050  }
2051 
2052  XkbKeyTypePtr key_type = XkbKeyKeyType(keyboard_map, keycode, group);
2053  unsigned int active_mods = event_mask & key_type->mods.mask;
2054 
2055  int i, level = 0;
2056  for (i = 0; i < key_type->map_count; i++) {
2057  if (key_type->map[i].active && key_type->map[i].mods.mask == active_mods) {
2058  level = key_type->map[i].level;
2059  }
2060  }
2061 
2062  keysym = XkbKeySymEntry(keyboard_map, keycode, level, group);
2063  XkbFreeClientMap(keyboard_map, XkbAllClientInfoMask, true);
2064  }
2065 
2066  return keysym;
2067 }
