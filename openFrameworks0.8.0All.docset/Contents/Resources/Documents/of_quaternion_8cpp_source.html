71  // vectors are necessary, if only for computing the dot product.
72  // Too bad the API is that generic, it leads to performance loss.
73  // Even in the case the 2 vectors are not normalized but same length,
74  // the sqrt could be shared, but we have no way to know beforehand
75  // at this point, while the caller may know.
76  // So, we have to test... in the hope of saving at least a sqrt
77  ofVec3f sourceVector = from;
78  ofVec3f targetVector = to;
79 
80  float fromLen2 = from.lengthSquared();
81  float fromLen;
82  // normalize only when necessary, epsilon test
83  if ((fromLen2 < 1.0 - 1e-7) || (fromLen2 > 1.0 + 1e-7)) {
84  fromLen = sqrt(fromLen2);
85  sourceVector /= fromLen;
86  } else fromLen = 1.0;
87 
88  float toLen2 = to.lengthSquared();
89  // normalize only when necessary, epsilon test
90  if ((toLen2 < 1.0 - 1e-7) || (toLen2 > 1.0 + 1e-7)) {
91  float toLen;
92  // re-use fromLen for case of mapping 2 vectors of the same length
93  if ((toLen2 > fromLen2 - 1e-7) && (toLen2 < fromLen2 + 1e-7)) {
94  toLen = fromLen;
95  } else toLen = sqrt(toLen2);
96  targetVector /= toLen;
97  }
98 
99 
100  // Now let's get into the real stuff
101  // Use "dot product plus one" as test as it can be re-used later on
102  double dotProdPlus1 = 1.0 + sourceVector.dot(targetVector);
103 
104  // Check for degenerate case of full u-turn. Use epsilon for detection
105  if (dotProdPlus1 < 1e-7) {
106 
107  // Get an orthogonal vector of the given vector
108  // in a plane with maximum vector coordinates.
109  // Then use it as quaternion axis with pi angle
110  // Trick is to realize one value at least is >0.6 for a normalized vector.
111  if (fabs(sourceVector.x) < 0.6) {
112  const double norm = sqrt(1.0 - sourceVector.x * sourceVector.x);
113  _v.x = 0.0;
114  _v.y = sourceVector.z / norm;
115  _v.z = -sourceVector.y / norm;
116  _v.w = 0.0;
117  } else if (fabs(sourceVector.y) < 0.6) {
118  const double norm = sqrt(1.0 - sourceVector.y * sourceVector.y);
119  _v.x = -sourceVector.z / norm;
120  _v.y = 0.0;
121  _v.z = sourceVector.x / norm;
122  _v.w = 0.0;
123  } else {
124  const double norm = sqrt(1.0 - sourceVector.z * sourceVector.z);
125  _v.x = sourceVector.y / norm;
126  _v.y = -sourceVector.x / norm;
127  _v.z = 0.0;
128  _v.w = 0.0;
129  }
130  }
131 
132  else {
133  // Find the shortest angle quaternion that transforms normalized vectors
134  // into one other. Formula is still valid when vectors are colinear
135  const double s = sqrt(0.5 * dotProdPlus1);
136  const ofVec3f tmp = sourceVector.getCrossed(targetVector) / (2.0 * s);
137  _v.x = tmp.x;

146 // Generally take adot product to get the angle between these
147 // and then use a cross product to get the rotation axis
148 // Watch out for the two special cases of when the vectors

150 void ofQuaternion::makeRotate_original( const ofVec3f& from, const ofVec3f& to ) {
151  const float epsilon = 0.0000001f;
152 
153  float length1 = from.length();
154  float length2 = to.length();
155 
156  // dot product vec1*vec2
157  float cosangle = from.dot(to) / (length1 * length2);
158 
159  if ( fabs(cosangle - 1) < epsilon ) {
160  //osg::notify(osg::INFO)<<"*** Quat::makeRotate(from,to) with near co-linear vectors, epsilon= "<<fabs(cosangle-1)<<std::endl;
161 
162  // cosangle is close to 1, so the vectors are close to being coincident
163  // Need to generate an angle of zero with any vector we like
164  // We'll choose (1,0,0)
165  makeRotate( 0.0, 0.0, 0.0, 1.0 );
166  } else
167  if ( fabs(cosangle + 1.0) < epsilon ) {
168  // vectors are close to being opposite, so will need to find a
169  // vector orthongonal to from to rotate about.
170  ofVec3f tmp;
171  if (fabs(from.x) < fabs(from.y))
172  if (fabs(from.x) < fabs(from.z)) tmp.set(1.0, 0.0, 0.0); // use x axis.
173  else tmp.set(0.0, 0.0, 1.0);
174  else if (fabs(from.y) < fabs(from.z)) tmp.set(0.0, 1.0, 0.0);
175  else tmp.set(0.0, 0.0, 1.0);
176 
177  ofVec3f fromd(from.x, from.y, from.z);
178 
179  // find orthogonal axis.
180  ofVec3f axis(fromd.getCrossed(tmp));
181  axis.normalize();
182 
183  _v.x = axis[0]; // sin of half angle of PI is 1.0.
184  _v.y = axis[1]; // sin of half angle of PI is 1.0.
185  _v.z = axis[2]; // sin of half angle of PI is 1.0.
186  _v.w = 0; // cos of half angle of PI is zero.
187 
188  } else {
189  // This is the usual situation - take a cross-product of vec1 and vec2
190  // and that is the axis around which to rotate.
191  ofVec3f axis(from.getCrossed(to));
192  float angle = acos( cosangle );
193  makeRotate( angle, axis );
194  }

273  heading = 2 * atan2(x(), w());
