50 /* For each pair of adjacent edges crossing the sweep line, there is
51 * an ActiveRegion to represent the region between them. The active
52 * regions are kept in sorted order in a dynamic dictionary. As the
53 * sweep line crosses each vertex, we update the affected regions.
54 */
55 
56 struct ActiveRegion {
57  TESShalfEdge *eUp; /* upper edge, directed right to left */
58  DictNode *nodeUp; /* dictionary node corresponding to eUp */
59  int windingNumber; /* used to determine which regions are
60  * inside the polygon */
61  int inside; /* is this region inside the polygon? */
62  int sentinel; /* marks fake edges at t = +/-infinity */
63  int dirty; /* marks regions where the upper or lower
64  * edge has changed, but we haven't checked
65  * whether they intersect yet */
66  int fixUpperEdge; /* marks temporary edges introduced when
67  * we process a "right vertex" (one without
68  * any edges leaving to the right) */
69 };
