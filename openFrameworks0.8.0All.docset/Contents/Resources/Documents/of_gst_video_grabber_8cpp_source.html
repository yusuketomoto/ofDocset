236 static void find_framerate (ofGstVideoFormat &format, int desired_framerate = -1)
237 {
238  int framerate_numerator = 1;
239  int framerate_denominator = 1;
240  float framerate = -1;
241  float prevFramerate = -1;
242  // find nearest to desired framerate
243  if(desired_framerate!=-1){
244  float diff = 9999;
245  for(unsigned i=0; i<format.framerates.size(); i++){
246  framerate = (float)format.framerates[i].numerator / (float)format.framerates[i].denominator;
247  if( fabs((float)desired_framerate - framerate) < diff){
248  diff = fabs((float)desired_framerate - framerate );
249  framerate_numerator = format.framerates[i].numerator;
250  framerate_denominator = format.framerates[i].denominator;
251  }
252  }
253  // find fastest
254  }else{
255  for (unsigned i = 0; i < format.framerates.size(); i++){
256  prevFramerate = framerate;
257  framerate = (float)format.framerates[i].numerator / (float)format.framerates[i].denominator;
258  if (framerate > prevFramerate ){
259  framerate_numerator = format.framerates[i].numerator;
260  framerate_denominator = format.framerates[i].denominator;
261  }
262  }
263  }
264 
265  format.choosen_framerate.numerator = framerate_numerator;
266  format.choosen_framerate.denominator = framerate_denominator;

520 static void get_device_data (ofGstDevice &webcam_device, int desired_framerate)
521 {
522  string pipeline_desc = webcam_device.gstreamer_src + " name=source device=" +
523  webcam_device.video_device + " ! fakesink";
524 
525  GError * err = NULL;
526  GstElement * pipeline = gst_parse_launch (pipeline_desc.c_str(), &err);
527  if ((pipeline == NULL) || (err != NULL)){
528  if (err){
529  ofLogError("ofGstVideoGrabber") << "get_device_data(): " << err->message;
530  g_error_free (err);
531  }else{
532  ofLogError("ofGstVideoGrabber") << "get_device_data(): caouldn't get pipeline";
533  }
534  if(pipeline)
535  gst_object_unref (pipeline);
536  return;
537  }
538 
539  // TODO: try to lower seconds,
540  // Start the pipeline and wait for max. 10 seconds for it to start up
541  gst_element_set_state (pipeline, GST_STATE_READY);
542  GstStateChangeReturn ret = gst_element_get_state (pipeline, NULL, NULL, 10 * GST_SECOND);
543 
544  // Check if any error messages were posted on the bus
545  GstBus * bus = gst_element_get_bus (pipeline);
546  GstMessage * msg = gst_bus_poll (bus, GST_MESSAGE_ERROR, 0);
547  gst_object_unref (bus);
548 
549  if ((msg == NULL) && (ret == GST_STATE_CHANGE_SUCCESS)){
550 
551  GstElement *src = gst_bin_get_by_name (GST_BIN (pipeline), "source");
552  char *name;
553  g_object_get (G_OBJECT (src), "device-name", &name, (void*)NULL);
554 
555  ofLogVerbose("ofGstVideoGrabber") << "get_device_data(): device: "
556  << (name==NULL?"":name) << "(" << webcam_device.video_device << ")";
557  GstPad *pad = gst_element_get_static_pad (src, "src");
558  GstCaps *caps = gst_pad_get_allowed_caps (pad);
559  gst_object_unref (pad);
560 
561  get_supported_video_formats (webcam_device, *caps, desired_framerate);
562 
563  gst_caps_unref (caps);
564  gst_object_unref(src);
565  }else if(msg){
566  gchar *debug;
567  gst_message_parse_error(msg, &err, &debug);
568 
569  ofLogError("ofGstVideoGrabber") << "get_device_data(): module "
570  << gst_element_get_name(GST_MESSAGE_SRC (msg)) << " reported: "
571  << err->message;
572 
573  g_error_free(err);
574  g_free(debug);
575  }
576  gst_element_set_state (pipeline, GST_STATE_NULL);
577  ret = gst_element_get_state (pipeline, NULL, NULL, 10 * GST_SECOND);
578  gst_object_unref (pipeline);
